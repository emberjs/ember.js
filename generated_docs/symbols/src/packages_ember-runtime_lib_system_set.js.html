 17 /**
 18   @class
 19 
 20   An unordered collection of objects.
 21 
 22   A Set works a bit like an array except that its items are not ordered.
 23   You can create a set to efficiently test for membership for an object. You
 24   can also iterate through a set just like an array, even accessing objects
 25   by index, however there is no gaurantee as to their order.
 26 
 27   Starting with Ember 2.0 all Sets are now observable since there is no
 28   added cost to providing this support.  Sets also do away with the more
 29   specialized Set Observer API in favor of the more generic Enumerable 
 30   Observer API - which works on any enumerable object including both Sets and
 31   Arrays.
 32 
 33   ## Creating a Set
 34 
 35   You can create a set like you would most objects using 
 36   `new Ember.Set()`.  Most new sets you create will be empty, but you can 
 37   also initialize the set with some content by passing an array or other 
 38   enumerable of objects to the constructor.
 39 
 40   Finally, you can pass in an existing set and the set will be copied. You
 41   can also create a copy of a set by calling `Ember.Set#copy()`.
 42 
 43       #js
 44       // creates a new empty set
 45       var foundNames = new Ember.Set();
 46 
 47       // creates a set with four names in it.
 48       var names = new Ember.Set(["Charles", "Tom", "Juan", "Alex"]); // :P
 49 
 50       // creates a copy of the names set.
 51       var namesCopy = new Ember.Set(names);
 52 
 53       // same as above.
 54       var anotherNamesCopy = names.copy();
 55 
 56   ## Adding/Removing Objects
 57 
 58   You generally add or remove objects from a set using `add()` or 
 59   `remove()`. You can add any type of object including primitives such as 
 60   numbers, strings, and booleans.
 61 
 62   Unlike arrays, objects can only exist one time in a set. If you call `add()` 
 63   on a set with the same object multiple times, the object will only be added
 64   once. Likewise, calling `remove()` with the same object multiple times will
 65   remove the object the first time and have no effect on future calls until
 66   you add the object to the set again.
 67 
 68   NOTE: You cannot add/remove null or undefined to a set. Any attempt to do so 
 69   will be ignored.
 70 
 71   In addition to add/remove you can also call `push()`/`pop()`. Push behaves 
 72   just like `add()` but `pop()`, unlike `remove()` will pick an arbitrary 
 73   object, remove it and return it. This is a good way to use a set as a job 
 74   queue when you don't care which order the jobs are executed in.
 75 
 76   ## Testing for an Object
 77 
 78   To test for an object's presence in a set you simply call 
 79   `Ember.Set#contains()`.
 80 
 81   ## Observing changes
 82 
 83   When using `Ember.Set`, you can observe the `"[]"` property to be 
 84   alerted whenever the content changes.  You can also add an enumerable 
 85   observer to the set to be notified of specific objects that are added and
 86   removed from the set.  See `Ember.Enumerable` for more information on 
 87   enumerables.
 88 
 89   This is often unhelpful. If you are filtering sets of objects, for instance,
 90   it is very inefficient to re-filter all of the items each time the set 
 91   changes. It would be better if you could just adjust the filtered set based 
 92   on what was changed on the original set. The same issue applies to merging 
 93   sets, as well.
 94 
 95   ## Other Methods
 96 
 97   `Ember.Set` primary implements other mixin APIs.  For a complete reference
 98   on the methods you will use with `Ember.Set`, please consult these mixins.
 99   The most useful ones will be `Ember.Enumerable` and 
100   `Ember.MutableEnumerable` which implement most of the common iterator 
101   methods you are used to on Array.
102 
103   Note that you can also use the `Ember.Copyable` and `Ember.Freezable`
104   APIs on `Ember.Set` as well.  Once a set is frozen it can no longer be 
105   modified.  The benefit of this is that when you call frozenCopy() on it,
106   Ember will avoid making copies of the set.  This allows you to write
107   code that can know with certainty when the underlying set data will or 
108   will not be modified.
109 
110   @extends Ember.Enumerable
111   @extends Ember.MutableEnumerable
112   @extends Ember.Copyable
113   @extends Ember.Freezable
114 
115   @since Ember 0.9
116 */
117 Ember.Set = Ember.CoreObject.extend(Ember.MutableEnumerable, Ember.Copyable, Ember.Freezable,
118   /** @scope Ember.Set.prototype */ {
119 
120   // ..........................................................
121   // IMPLEMENT ENUMERABLE APIS
122   //
123 
124   /**
125     This property will change as the number of objects in the set changes.
126 
127     @property Number
128     @default 0
129   */
130   length: 0,
131 
132   /**
133     Clears the set.  This is useful if you want to reuse an existing set
134     without having to recreate it.
135 
136     @returns {Ember.Set}
137   */
138   clear: function() {
139     if (this.isFrozen) { throw new Error(Ember.FROZEN_ERROR); }
140     var len = get(this, 'length');
141     this.enumerableContentWillChange(len, 0);
142     set(this, 'length', 0);
143     this.enumerableContentDidChange(len, 0);
144     return this;
145   },
146 
147   /**
148     Returns true if the passed object is also an enumerable that contains the 
149     same objects as the receiver.
150 
151     @param {Ember.Set} obj the other object
152     @returns {Boolean}
153   */
154   isEqual: function(obj) {
155     // fail fast
156     if (!Ember.Enumerable.detect(obj)) return false;
157     
158     var loc = get(this, 'length');
159     if (get(obj, 'length') !== loc) return false;
160 
161     while(--loc >= 0) {
162       if (!obj.contains(this[loc])) return false;
163     }
164 
165     return true;
166   },
167   
168   /**
169     Adds an object to the set.  Only non-null objects can be added to a set 
170     and those can only be added once. If the object is already in the set or
171     the passed value is null this method will have no effect.
172 
173     This is an alias for `Ember.MutableEnumerable.addObject()`.
174 
175     @function
176     @param {Object} obj The object to add
177     @returns {Ember.Set} receiver
178   */
179   add: Ember.alias('addObject'),
180 
181   /**
182     Removes the object from the set if it is found.  If you pass a null value
183     or an object that is already not in the set, this method will have no
184     effect. This is an alias for `Ember.MutableEnumerable.removeObject()`.
185 
186     @function
187     @param {Object} obj The object to remove
188     @returns {Ember.Set} receiver
189   */
190   remove: Ember.alias('removeObject'),
191   
192   /**
193     Removes an arbitrary object from the set and returns it.
194 
195     @returns {Object} An object from the set or null
196   */
197   pop: function() {
198     if (get(this, 'isFrozen')) throw new Error(Ember.FROZEN_ERROR);
199     var obj = this.length > 0 ? this[this.length-1] : null;
200     this.remove(obj);
201     return obj;
202   },
203 
204   /**
205     This is an alias for `Ember.MutableEnumerable.addObject()`.
206 
207     @function
208   */
209   push: Ember.alias('addObject'),
210   
211   /**
212     This is an alias for `Ember.Set.pop()`.
213     @function
214   */
215   shift: Ember.alias('pop'),
216 
217   /**
218     This is an alias of `Ember.Set.push()`
219     @function
220   */
221   unshift: Ember.alias('push'),
222 
223   /**
224     This is an alias of `Ember.MutableEnumerable.addObjects()`
225     @function
226   */
227   addEach: Ember.alias('addObjects'),
228 
229   /**
230     This is an alias of `Ember.MutableEnumerable.removeObjects()`
231     @function
232   */
233   removeEach: Ember.alias('removeObjects'),

239   /** @private */
240   init: function(items) {
241     this._super();
242     if (items) this.addObjects(items);
243   },
244 
245   /** @private (nodoc) - implement Ember.Enumerable */
246   nextObject: function(idx) {
247     return this[idx];
248   },
249 
250   /** @private - more optimized version */
251   firstObject: Ember.computed(function() {
252     return this.length > 0 ? this[0] : undefined;  
253   }).property('[]').cacheable(),
254 
255   /** @private - more optimized version */
256   lastObject: Ember.computed(function() {
257     return this.length > 0 ? this[this.length-1] : undefined;
258   }).property('[]').cacheable(),
259 
260   /** @private (nodoc) - implements Ember.MutableEnumerable */
261   addObject: function(obj) {
262     if (get(this, 'isFrozen')) throw new Error(Ember.FROZEN_ERROR);
263     if (none(obj)) return this; // nothing to do
264     
265     var guid = guidFor(obj),
266         idx  = this[guid],
267         len  = get(this, 'length'),
268         added ;
269         
270     if (idx>=0 && idx<len && (this[idx] === obj)) return this; // added
271     
272     added = [obj];
273     this.enumerableContentWillChange(null, added);
274     len = get(this, 'length');
275     this[guid] = len;
276     this[len] = obj;
277     set(this, 'length', len+1);
278     this.enumerableContentDidChange(null, added);
279 
280     return this;
281   },
282   
283   /** @private (nodoc) - implements Ember.MutableEnumerable */
284   removeObject: function(obj) {
285     if (get(this, 'isFrozen')) throw new Error(Ember.FROZEN_ERROR);
286     if (none(obj)) return this; // nothing to do
287     
288     var guid = guidFor(obj),
289         idx  = this[guid],
290         len = get(this, 'length'),
291         last, removed;
292         
293     
294     if (idx>=0 && idx<len && (this[idx] === obj)) {
295       removed = [obj];
296 
297       this.enumerableContentWillChange(removed, null);
298       
299       // swap items - basically move the item to the end so it can be removed
300       if (idx < len-1) {
301         last = this[len-1];
302         this[idx] = last;
303         this[guidFor(last)] = idx;
304       }
305 
306       delete this[guid];
307       delete this[len-1];
308       set(this, 'length', len-1);
309 
310       this.enumerableContentDidChange(removed, null);
311     }
312     
313     return this;
314   },
315 
316   /** @private (nodoc) - optimized version */
317   contains: function(obj) {
318     return this[guidFor(obj)]>=0;
319   },
320   
321   /** @private (nodoc) */
322   copy: function() {
323     var C = this.constructor, ret = new C(), loc = get(this, 'length');
324     set(ret, 'length', loc);
325     while(--loc>=0) {
326       ret[loc] = this[loc];
327       ret[guidFor(this[loc])] = loc;
328     }
329     return ret;
330   },
331   
332   /** @private */
333   toString: function() {
334     var len = this.length, idx, array = [];
335     for(idx = 0; idx < len; idx++) {
336       array[idx] = this[idx];
337     }
338     return "Ember.Set<%@>".fmt(array.join(','));
339   },

361   isSet: true

365 // Support the older API 
366 var o_create = Ember.Set.create;
367 Ember.Set.create = function(items) {
368   if (items && Ember.Enumerable.detect(items)) {
369     ember_deprecate('Passing an enumerable to Ember.Set.create() is deprecated and will be removed in a future version of Ember.  Use new Ember.Set(items) instead.');
370     return new Ember.Set(items);
371   } else {
372     return o_create.apply(this, arguments);
373   }
