  6 /*globals Handlebars ember_assert */

  8 // TODO: Don't require all of this module
  9 require('ember-handlebars');
 10 require('ember-handlebars/helpers/view');
 11 
 12 var get = Ember.get, getPath = Ember.Handlebars.getPath, fmt = Ember.String.fmt;
 13 
 14 /**
 15   @name Handlebars.helpers.collection
 16   @param {String} path
 17   @param {Hash} options
 18   @returns {String} HTML string
 19 */
 20 Ember.Handlebars.registerHelper('collection', function(path, options) {
 21   // If no path is provided, treat path param as options.
 22   if (path && path.data && path.data.isRenderData) {
 23     options = path;
 24     path = undefined;
 25     ember_assert("You cannot pass more than one argument to the collection helper", arguments.length === 1);
 26   } else {
 27     ember_assert("You cannot pass more than one argument to the collection helper", arguments.length === 2);
 28   }
 29 
 30   var fn = options.fn;
 31   var data = options.data;
 32   var inverse = options.inverse;
 33 
 34   // If passed a path string, convert that into an object.
 35   // Otherwise, just default to the standard class.
 36   var collectionClass;
 37   collectionClass = path ? getPath(this, path) : Ember.CollectionView;
 38   ember_assert(fmt("%@ #collection: Could not find %@", data.view, path), !!collectionClass);
 39 
 40   var hash = options.hash, itemHash = {}, match;
 41 
 42   // Extract item view class if provided else default to the standard class
 43   var itemViewClass, itemViewPath = hash.itemViewClass;
 44   var collectionPrototype = get(collectionClass, 'proto');
 45   delete hash.itemViewClass;
 46   itemViewClass = itemViewPath ? getPath(collectionPrototype, itemViewPath) : collectionPrototype.itemViewClass;
 47   ember_assert(fmt("%@ #collection: Could not find %@", data.view, itemViewPath), !!itemViewClass);
 48 
 49   // Go through options passed to the {{collection}} helper and extract options
 50   // that configure item views instead of the collection itself.
 51   for (var prop in hash) {
 52     if (hash.hasOwnProperty(prop)) {
 53       match = prop.match(/^item(.)(.*)$/);
 54 
 55       if(match) {
 56         // Convert itemShouldFoo -> shouldFoo
 57         itemHash[match[1].toLowerCase() + match[2]] = hash[prop];
 58         // Delete from hash as this will end up getting passed to the
 59         // {{view}} helper method.
 60         delete hash[prop];
 61       }
 62     }
 63   }
 64 
 65   var tagName = hash.tagName || get(collectionClass, 'proto').tagName;
 66 
 67   if (fn) {
 68     itemHash.template = fn;
 69     delete options.fn;
 70   }
 71 
 72   if (inverse && inverse !== Handlebars.VM.noop) {
 73     hash.emptyView = Ember.View.extend({
 74       template: inverse,
 75       tagName: itemHash.tagName
 76     });
 77   }
 78 
 79   if (hash.preserveContext) {
 80     itemHash.templateContext = Ember.computed(function() {
 81       return get(this, 'content');
 82     }).property('content');
 83     delete hash.preserveContext;
 84   }
 85 
 86   hash.itemViewClass = Ember.Handlebars.ViewHelper.viewClassFromHTMLOptions(itemViewClass, itemHash);
 87 
 88   return Ember.Handlebars.helpers.view.call(this, collectionClass, options);
 89 });
