 25 /** @private */
 26 function meta(obj, writable) {
 27   var m = Ember.meta(obj, writable!==false), ret = m.mixins;
 28   if (writable===false) return ret || EMPTY_META;
 29 
 30   if (!ret) {
 31     ret = m.mixins = { __emberproto__: obj };
 32   } else if (ret.__emberproto__ !== obj) {
 33     ret = m.mixins = o_create(ret);
 34     ret.__emberproto__ = obj;
 35   }
 36   return ret;
 37 }

 39 /** @private */
 40 function initMixin(mixin, args) {
 41   if (args && args.length > 0) {
 42     mixin.mixins = a_map.call(args, function(x) {
 43       if (x instanceof Mixin) return x;
 44 
 45       // Note: Manually setup a primitive mixin here.  This is the only
 46       // way to actually get a primitive mixin.  This way normal creation
 47       // of mixins will give you combined mixins...
 48       var mixin = new Mixin();
 49       mixin.properties = x;
 50       return mixin;
 51     });
 52   }
 53   return mixin;
 54 }

 63 /** @private */
 64 function mergeMixins(mixins, m, descs, values, base) {
 65   var len = mixins.length, idx, mixin, guid, props, value, key, ovalue, concats;
 66 
 67   /** @private */
 68   function removeKeys(keyName) {
 69     delete descs[keyName];
 70     delete values[keyName];
 71   }
 72 
 73   for(idx=0;idx<len;idx++) {
 74 
 75     mixin = mixins[idx];
 76     if (!mixin) throw new Error('Null value found in Ember.mixin()');
 77 
 78     if (mixin instanceof Mixin) {
 79       guid = Ember.guidFor(mixin);
 80       if (m[guid]) continue;
 81       m[guid] = mixin;
 82       props = mixin.properties;
 83     } else {
 84       props = mixin; // apply anonymous mixin properties
 85     }
 86 
 87     if (props) {
 88 
 89       // reset before adding each new mixin to pickup concats from previous
 90       concats = values.concatenatedProperties || base.concatenatedProperties;
 91       if (props.concatenatedProperties) {
 92         concats = concats ? concats.concat(props.concatenatedProperties) : props.concatenatedProperties;
 93       }
 94 
 95       for (key in props) {
 96         if (!props.hasOwnProperty(key)) continue;
 97         value = props[key];
 98         if (value instanceof Ember.Descriptor) {
 99           if (value === REQUIRED && descs[key]) { continue; }
100 
101           descs[key]  = value;
102           values[key] = undefined;
103         } else {
104 
105           // impl super if needed...
106           if (isMethod(value)) {
107             ovalue = (descs[key] === Ember.SIMPLE_PROPERTY) && values[key];
108             if (!ovalue) ovalue = base[key];
109             if ('function' !== typeof ovalue) ovalue = null;
110             if (ovalue) {
111               var o = value.__ember_observes__, ob = value.__ember_observesBefore__;
112               value = Ember.wrap(value, ovalue);
113               value.__ember_observes__ = o;
114               value.__ember_observesBefore__ = ob;
115             }
116           } else if ((concats && concats.indexOf(key)>=0) || key === 'concatenatedProperties') {
117             var baseValue = values[key] || base[key];
118             value = baseValue ? baseValue.concat(value) : Ember.makeArray(value);
119           }
120 
121           descs[key]  = Ember.SIMPLE_PROPERTY;
122 123           values[key] = value;
124         }
125       }
126 
127       // manually copy toString() because some JS engines do not enumerate it
128       if (props.hasOwnProperty('toString')) {
129         base.toString = props.toString;
130       }
131 
132     } else if (mixin.mixins) {
133       mergeMixins(mixin.mixins, m, descs, values, base);
134       if (mixin._without) mixin._without.forEach(removeKeys);
135     }
136   }
137 }

166 /** @private */
167 function applyMixin(obj, mixins, partial) {
168   var descs = {}, values = {}, m = Ember.meta(obj), req = m.required;
169   var key, willApply, didApply, value, desc;
170 
171   var mixinBindings = Ember._mixinBindings;
172 
173   mergeMixins(mixins, meta(obj), descs, values, obj);
174 
175   if (MixinDelegate.detect(obj)) {
176     willApply = values.willApplyProperty || obj.willApplyProperty;
177     didApply  = values.didApplyProperty || obj.didApplyProperty;
178   }
179 
180   for(key in descs) {
181     if (!descs.hasOwnProperty(key)) continue;
182 
183     desc = descs[key];
184     value = values[key];
185 
186     if (desc === REQUIRED) {
187       if (!(key in obj)) {
188         if (!partial) throw new Error('Required property not defined: '+key);
189 
190         // for partial applies add to hash of required keys
191         req = writableReq(obj);
192         req.__ember_count__++;
193         req[key] = true;
194       }
195 
196     } else {
197 
198       while (desc instanceof Alias) {
199 
200         var altKey = desc.methodName;
201         if (descs[altKey]) {
202           value = values[altKey];
203           desc  = descs[altKey];
204         } else if (m.descs[altKey]) {
205           desc  = m.descs[altKey];
206           value = desc.val(obj, altKey);
207         } else {
208           value = obj[altKey];
209           desc  = Ember.SIMPLE_PROPERTY;
210         }
211       }
212 
213       if (willApply) willApply.call(obj, key);
214 
215       var observerPaths = getObserverPaths(value),
216           curObserverPaths = observerPaths && getObserverPaths(obj[key]),
217           beforeObserverPaths = getBeforeObserverPaths(value),
218           curBeforeObserverPaths = beforeObserverPaths && getBeforeObserverPaths(obj[key]),
219           len, idx;
220 
221       if (curObserverPaths) {
222         len = curObserverPaths.length;
223         for(idx=0;idx<len;idx++) {
224           Ember.removeObserver(obj, curObserverPaths[idx], null, key);
225         }
226       }
227 
228       if (curBeforeObserverPaths) {
229         len = curBeforeObserverPaths.length;
230         for(idx=0;idx<len;idx++) {
231           Ember.removeBeforeObserver(obj, curBeforeObserverPaths[idx], null,key);
232         }
233       }
234 
235       // TODO: less hacky way for ember-runtime to add bindings.
236       value = mixinBindings(obj, key, value, m);
237 
238       defineProperty(obj, key, desc, value);
239 
240       if (observerPaths) {
241         len = observerPaths.length;
242         for(idx=0;idx<len;idx++) {
243           Ember.addObserver(obj, observerPaths[idx], null, key);
244         }
245       }
246 
247       if (beforeObserverPaths) {
248         len = beforeObserverPaths.length;
249         for(idx=0;idx<len;idx++) {
250           Ember.addBeforeObserver(obj, beforeObserverPaths[idx], null, key);
251         }
252       }
253 
254       if (req && req[key]) {
255         req = writableReq(obj);
256         req.__ember_count__--;
257         req[key] = false;
258       }
259 
260       if (didApply) didApply.call(obj, key);
261 
262     }
263   }
264 
265   // Make sure no required attrs remain
266   if (!partial && req && req.__ember_count__>0) {
267     var keys = [];
268     for(key in req) {
269       if (META_SKIP[key]) continue;
270       keys.push(key);
271     }
272     throw new Error('Required properties not defined: '+keys.join(','));
273   }
274   return obj;
275 }

305 Mixin.prototype.reopen = function() {
306 
307   var mixin, tmp;
308 
309   if (this.properties) {
310     mixin = Mixin.create();
311     mixin.properties = this.properties;
312     delete this.properties;
313     this.mixins = [mixin];
314   }
315 
316   var len = arguments.length, mixins = this.mixins, idx;
317 
318   for(idx=0;idx<len;idx++) {
319     mixin = arguments[idx];
320     if (mixin instanceof Mixin) {
321       mixins.push(mixin);
322     } else {
323       tmp = Mixin.create();
324       tmp.properties = mixin;
325       mixins.push(tmp);
326     }
327   }
328 
329   return this;
330 };

347 /** @private */
348 function _detect(curMixin, targetMixin, seen) {
349   var guid = Ember.guidFor(curMixin);
350 
351   if (seen[guid]) return false;
352   seen[guid] = true;
353 
354   if (curMixin === targetMixin) return true;
355   var mixins = curMixin.mixins, loc = mixins ? mixins.length : 0;
356 357   while(--loc >= 0) {
358     if (_detect(mixins[loc], targetMixin, seen)) return true;
359   }
360   return false;
361 }

363 Mixin.prototype.detect = function(obj) {
364   if (!obj) return false;
365   if (obj instanceof Mixin) return _detect(obj, this, {});
366   return !!meta(obj, false)[Ember.guidFor(this)];
367 };

375 /** @private */
376 function _keys(ret, mixin, seen) {
377   if (seen[Ember.guidFor(mixin)]) return;
378   seen[Ember.guidFor(mixin)] = true;
379 
380   if (mixin.properties) {
381     var props = mixin.properties;
382     for(var key in props) {
383       if (props.hasOwnProperty(key)) ret[key] = true;
384     }
385   } else if (mixin.mixins) {
386     mixin.mixins.forEach(function(x) { _keys(ret, x, seen); });
387   }
388 }

390 Mixin.prototype.keys = function() {
391   var keys = {}, seen = {}, ret = [];
392   _keys(keys, this, seen);
393   for(var key in keys) {
394     if (keys.hasOwnProperty(key)) ret.push(key);
395   }
396   return ret;

404 /** @private */
405 function processNames(paths, root, seen) {
406   var idx = paths.length;
407   for(var key in root) {
408     if (!root.hasOwnProperty || !root.hasOwnProperty(key)) continue;
409     var obj = root[key];
410     paths[idx] = key;
411 
412     if (obj && obj.toString === classToString) {
413       obj[NAME_KEY] = paths.join('.');
414     } else if (obj && get(obj, 'isNamespace')) {
415       if (seen[Ember.guidFor(obj)]) continue;
416       seen[Ember.guidFor(obj)] = true;
417       processNames(paths, obj, seen);
418     }
419 
420   }
421   paths.length = idx; // cut out last item
422 }

424 /** @private */
425 function findNamespaces() {
426   var Namespace = Ember.Namespace, obj;
427 
428   if (Namespace.PROCESSED) { return; }
429 
430   for (var prop in window) {
431     //  get(window.globalStorage, 'isNamespace') would try to read the storage for domain isNamespace and cause exception in Firefox.
432     // globalStorage is a storage obsoleted by the WhatWG storage specification. See https://developer.mozilla.org/en/DOM/Storage#globalStorage
433     if (prop === "globalStorage" && window.StorageList && window.globalStorage instanceof window.StorageList) { continue; }
434     // Unfortunately, some versions of IE don't support window.hasOwnProperty
435     if (window.hasOwnProperty && !window.hasOwnProperty(prop)) { continue; }
436 
437     obj = window[prop];
438 
439     if (obj && get(obj, 'isNamespace')) {
440       obj[NAME_KEY] = prop;
441     }
442   }
443 }

447 /** @private */
448 superClassString = function(mixin) {
449   var superclass = mixin.superclass;
450   if (superclass) {
451     if (superclass[NAME_KEY]) { return superclass[NAME_KEY]; }
452     else { return superClassString(superclass); }
453   } else {
454     return;
455   }

458 /** @private */
459 classToString = function() {
460   var Namespace = Ember.Namespace, namespace;
461 
462   // TODO: Namespace should really be in Metal
463   if (Namespace) {
464     if (!this[NAME_KEY] && !classToString.processed) {
465       if (!Namespace.PROCESSED) {
466         findNamespaces();
467         Namespace.PROCESSED = true;
468       }
469 
470       classToString.processed = true;
471 
472       var namespaces = Namespace.NAMESPACES;
473       for (var i=0, l=namespaces.length; i<l; i++) {
474         namespace = namespaces[i];
475         processNames([namespace.toString()], namespace, {});
476       }
477     }
478   }
479 
480   if (this[NAME_KEY]) {
481     return this[NAME_KEY];
482   } else {
483     var str = superClassString(this);
484     if (str) {
485       return "(subclass of " + str + ")";
486     } else {
487       return "(unknown mixin)";
488     }
489   }
490 };
