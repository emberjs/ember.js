 13 var fmt = Ember.String.fmt,
 14     w   = Ember.String.w,
 15     loc = Ember.String.loc,
 16     camelize = Ember.String.camelize,
 17     decamelize = Ember.String.decamelize,
 18     dasherize = Ember.String.dasherize,
 19     underscore = Ember.String.underscore;
 20   
 21 if (Ember.EXTEND_PROTOTYPES) {
 22 
 23   /**
 24     @see Ember.String.fmt
 25   */
 26   String.prototype.fmt = function() {
 27     return fmt(this, arguments);
 28   };
 29   
 30   /**
 31     @see Ember.String.w
 32   */
 33   String.prototype.w = function() {
 34     return w(this);
 35   };
 36   
 37   /**
 38     @see Ember.String.loc
 39   */
 40   String.prototype.loc = function() {
 41     return loc(this, arguments);
 42   };
 43 
 44   /**
 45     @see Ember.String.camelize
 46   */
 47   String.prototype.camelize = function() {
 48     return camelize(this);
 49   };
 50 
 51   /**
 52     @see Ember.String.decamelize
 53   */
 54   String.prototype.decamelize = function() {
 55     return decamelize(this);
 56   };
 57 
 58   /**
 59     @see Ember.String.dasherize
 60   */
 61   String.prototype.dasherize = function() {
 62     return dasherize(this);
 63   };
 64 
 65   /**
 66     @see Ember.String.underscore
 67   */
 68   String.prototype.underscore = function() {
 69     return underscore(this);
 70   };
 71 
 72 }
