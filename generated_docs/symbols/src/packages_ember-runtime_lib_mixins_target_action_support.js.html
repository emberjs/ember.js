1 var get = Ember.get, set = Ember.set, getPath = Ember.getPath;
  2 
  3 Ember.TargetActionSupport = Ember.Mixin.create({
  4   target: null,
  5   action: null,
  6 
  7   targetObject: Ember.computed(function() {
  8     var target = get(this, 'target');
  9 
 10     if (Ember.typeOf(target) === "string") {
 11       // TODO: Remove the false when deprecation is done
 12       var value = getPath(this, target, false);
 13       if (value === undefined) { value = getPath(window, target); }
 14       return value;
 15     } else {
 16       return target;
 17     }
 18   }).property('target').cacheable(),
 19 
 20   triggerAction: function() {
 21     var action = get(this, 'action'),
 22         target = get(this, 'targetObject');
 23 
 24     if (target && action) {
 25       var ret;
 26       
 27       if (typeof target.send === 'function') {
 28         ret = target.send(action, this);
 29       } else {
 30         if (typeof action === 'string') {
 31           action = target[action];
 32         }
 33         ret = action.call(target, this);
 34       }
 35       if (ret !== false) ret = true;
 36       
 37       return ret;
 38     } else {
 39       return false;
 40     }
 41   }
 42 });
