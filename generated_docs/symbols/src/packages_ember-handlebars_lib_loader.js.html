  8 require("ember-handlebars/ext");

 10 // Find templates stored in the head tag as script tags and make them available
 11 // to Ember.CoreView in the global Ember.TEMPLATES object. This will be run as as
 12 // jQuery DOM-ready callback.
 13 //
 14 // Script tags with "text/x-handlebars" will be compiled
 15 // with Ember's Handlebars and are suitable for use as a view's template.
 16 // Those with type="text/x-raw-handlebars" will be compiled with regular
 17 // Handlebars and are suitable for use in views' computed properties.
 18 Ember.Handlebars.bootstrap = function(ctx) {
 19   var selectors = 'script[type="text/x-handlebars"], script[type="text/x-raw-handlebars"]';
 20 
 21   if (Ember.ENV.LEGACY_HANDLEBARS_TAGS) { selectors += ', script[type="text/html"]'; }
 22 
 23   ember_warn("Ember no longer parses text/html script tags by default. Set ENV.LEGACY_HANDLEBARS_TAGS = true to restore this functionality.", Ember.ENV.LEGACY_HANDLEBARS_TAGS || Ember.$('script[type="text/html"]').length === 0);
 24 
 25   Ember.$(selectors, ctx)
 26     .each(function() {
 27     // Get a reference to the script tag
 28     var script = Ember.$(this),
 29         type   = script.attr('type');
 30 
 31     if (type === 'text/html' && !Ember.ENV.LEGACY_HANDLEBARS_TAGS) { return; }
 32 
 33     var compile = (script.attr('type') === 'text/x-raw-handlebars') ?
 34                   Ember.$.proxy(Handlebars.compile, Handlebars) :
 35                   Ember.$.proxy(Ember.Handlebars.compile, Ember.Handlebars),
 36       // Get the name of the script, used by Ember.View's templateName property.
 37       // First look for data-template-name attribute, then fall back to its
 38       // id if no name is found.
 39       templateName = script.attr('data-template-name') || script.attr('id'),
 40       template = compile(script.html()),
 41       view, viewPath, tagName;
 42 
 43     if (templateName) {
 44       // For templates which have a name, we save them and then remove them from the DOM
 45       Ember.TEMPLATES[templateName] = template;
 46 
 47       // Remove script tag from DOM
 48       script.remove();
 49     } else {
 50       if (script.parents('head').length !== 0) {
 51         // don't allow inline templates in the head
 52         throw new Ember.Error("Template found in \<head\> without a name specified. " +
 53                          "Please provide a data-template-name attribute.\n" +
 54                          script.html());
 55       }
 56 
 57       // For templates which will be evaluated inline in the HTML document, instantiates a new
 58       // view, and replaces the script tag holding the template with the new
 59       // view's DOM representation.
 60       //
 61       // Users can optionally specify a custom view subclass to use by setting the
 62       // data-view attribute of the script tag.
 63       viewPath = script.attr('data-view');
 64       view = viewPath ? Ember.getPath(viewPath) : Ember.View;
 65 
 66       // Users can optionally specify a custom tag name to use by setting the
 67       // data-tag-name attribute on the script tag.
 68       tagName = script.attr('data-tag-name');
 69 
 70       view = view.create({
 71         template: template,
 72         tagName: (tagName) ? tagName : undefined
 73       });
 74 
 75       view._insertElementLater(function() {
 76         script.replaceWith(this.$());
 77 
 78         // Avoid memory leak in IE
 79         script = null;
 80       });
 81     }
 82   });
 83 };
 84 
 85 Ember.$(document).ready(
 86   function(){
 87     Ember.Handlebars.bootstrap( Ember.$(document) );
 88   }
 89 );
