  8 require('ember-views/views/states/default');
  9 
 10 var get = Ember.get, set = Ember.set, meta = Ember.meta;
 11 
 12 Ember.View.states.hasElement = {
 13   parentState: Ember.View.states._default,
 14 
 15   $: function(view, sel) {
 16     var elem = get(view, 'element');
 17     return sel ? Ember.$(sel, elem) : Ember.$(elem);
 18   },
 19 
 20   getElement: function(view) {
 21     var parent = get(view, 'parentView');
 22     if (parent) { parent = get(parent, 'element'); }
 23     if (parent) { return view.findElementInParentElement(parent); }
 24     return Ember.$("#" + get(view, 'elementId'))[0];
 25   },
 26 
 27   setElement: function(view, value) {
 28     if (value === null) {
 29       view.invalidateRecursively('element');
 30       view.transitionTo('preRender');
 31     } else {
 32       throw "You cannot set an element to a non-null value when the element is already in the DOM.";
 33     }
 34 
 35     return value;
 36   },
 37 
 38   // once the view has been inserted into the DOM, rerendering is
 39   // deferred to allow bindings to synchronize.
 40   rerender: function(view) {
 41     view.clearRenderedChildren();
 42 
 43     get(view, 'domManager').replace();
 44     return view;
 45   },
 46 
 47   // once the view is already in the DOM, destroying it removes it
 48   // from the DOM, nukes its element, and puts it back into the
 49   // preRender state.
 50   destroyElement: function(view) {
 51     view.invokeRecursively(function(view) {
 52       this.willDestroyElement();
 53     });
 54 
 55     get(view, 'domManager').remove();
 56     return view;
 57   },
 58 
 59   // Handle events from `Ember.EventDispatcher`
 60   handleEvent: function(view, eventName, evt) {
 61     var handler = view[eventName];
 62     if (Ember.typeOf(handler) === 'function') {
 63       return handler.call(view, evt);
 64     } else {
 65       return true; // continue event propagation
 66     }
 67   }
 68 };
 69 
 70 Ember.View.states.inDOM = {
 71   parentState: Ember.View.states.hasElement,
 72 
 73   insertElement: function() {
 74     throw "You can't insert an element into the DOM that has already been inserted";
 75   }
 76 };
