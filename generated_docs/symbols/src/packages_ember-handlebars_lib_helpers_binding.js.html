 18 (function() {
 19   // Binds a property into the DOM. This will create a hook in DOM that the
 20   // KVO system will look for and update if the property changes.
 21   var bind = function(property, options, preserveContext, shouldDisplay, valueNormalizer) {
 22     var data = options.data,
 23         fn = options.fn,
 24         inverse = options.inverse,
 25         view = data.view,
 26         ctx  = this;
 27 
 28     // Set up observers for observable objects
 29     if ('object' === typeof this) {
 30       // Create the view that will wrap the output of this template/property 
 31       // and add it to the nearest view's childViews array.
 32       // See the documentation of Ember._BindableSpanView for more.
 33       var bindView = view.createChildView(Ember._BindableSpanView, {
 34         preserveContext: preserveContext,
 35         shouldDisplayFunc: shouldDisplay,
 36         valueNormalizerFunc: valueNormalizer,
 37         displayTemplate: fn,
 38         inverseTemplate: inverse,
 39         property: property,
 40         previousContext: ctx,
 41         isEscaped: options.hash.escaped
 42       });
 43 
 44       view.appendChild(bindView);
 45 
 46       /** @private */
 47       var observer = function() {
 48         Ember.run.once(bindView, 'rerenderIfNeeded');
 49       };
 50 
 51       // Observes the given property on the context and
 52       // tells the Ember._BindableSpan to re-render. If property
 53       // is an empty string, we are printing the current context
 54       // object ({{this}}) so updating it is not our responsibility.
 55       if (property !== '') {
 56         Ember.addObserver(ctx, property, observer);
 57       }
 58     } else {
 59       // The object is not observable, so just render it out and
 60       // be done with it.
 61       data.buffer.push(getPath(this, property));
 62     }
 63   };

 65   /**
 66     '_triageMustache' is used internally select between a binding and helper for 
 67     the given context. Until this point, it would be hard to determine if the 
 68     mustache is a property reference or a regular helper reference. This triage
 69     helper resolves that.
 70 
 71     This would not be typically invoked by directly.
 72 
 73     @private
 74     @name Handlebars.helpers._triageMustache
 75     @param {String} property Property/helperID to triage
 76     @param {Function} fn Context to provide for rendering
 77     @returns {String} HTML string
 78   */
 79   EmberHandlebars.registerHelper('_triageMustache', function(property, fn) {
 80     ember_assert("You cannot pass more than one argument to the _triageMustache helper", arguments.length <= 2);
 81     if (helpers[property]) {
 82       return helpers[property].call(this, fn);
 83     }
 84     else {
 85  86       return helpers.bind.apply(this, arguments);
 87     }
 88   });

 90   /**
 91     `bind` can be used to display a value, then update that value if it 
 92     changes. For example, if you wanted to print the `title` property of 
 93     `content`:
 94 
 95         {{bind "content.title"}}
 96 
 97     This will return the `title` property as a string, then create a new 
 98     observer at the specified path. If it changes, it will update the value in 
 99     DOM. Note that if you need to support IE7 and IE8 you must modify the 
100     model objects properties using Ember.get() and Ember.set() for this to work as 
101     it relies on Ember's KVO system.  For all other browsers this will be handled
102     for you automatically.
103 
104     @private
105     @name Handlebars.helpers.bind
106     @param {String} property Property to bind
107     @param {Function} fn Context to provide for rendering
108     @returns {String} HTML string
109   */
110   EmberHandlebars.registerHelper('bind', function(property, fn) {
111     ember_assert("You cannot pass more than one argument to the bind helper", arguments.length <= 2);
112 
113     var context = (fn.contexts && fn.contexts[0]) || this;
114 
115     return bind.call(context, property, fn, false, function(result) {
116       return !Ember.none(result);
117     });
118   });
119 
120   /**
121     Use the `boundIf` helper to create a conditional that re-evaluates 
122     whenever the bound value changes.
123 
124         {{#boundIf "content.shouldDisplayTitle"}}
125           {{content.title}}
126         {{/boundIf}}
127 
128     @private
129     @name Handlebars.helpers.boundIf
130     @param {String} property Property to bind
131     @param {Function} fn Context to provide for rendering
132     @returns {String} HTML string
133   */
134   EmberHandlebars.registerHelper('boundIf', function(property, fn) {
135     var context = (fn.contexts && fn.contexts[0]) || this;
136     var func = function(result) {
137       if (Ember.typeOf(result) === 'array') {
138         return get(result, 'length') !== 0;
139       } else {
140         return !!result;
141       }
142     };
143 
144     return bind.call(context, property, fn, true, func, func);
145   });

193 /**
194   `bindAttr` allows you to create a binding between DOM element attributes and
195   Ember objects. For example:
196 
197       <img {{bindAttr src="imageUrl" alt="imageTitle"}}>
198 
199   @name Handlebars.helpers.bindAttr
200   @param {Hash} options
201   @returns {String} HTML string
202 */
203 EmberHandlebars.registerHelper('bindAttr', function(options) {
204 
205   var attrs = options.hash;
206 
207   ember_assert("You must specify at least one hash argument to bindAttr", !!Ember.keys(attrs).length);
208 
209   var view = options.data.view;
210   var ret = [];
211   var ctx = this;
212 
213   // Generate a unique id for this element. This will be added as a
214   // data attribute to the element so it can be looked up when
215   // the bound property changes.
216   var dataId = ++jQuery.uuid;
217 
218   // Handle classes differently, as we can bind multiple classes
219   var classBindings = attrs['class'];
220   if (classBindings !== null && classBindings !== undefined) {
221     var classResults = EmberHandlebars.bindClasses(this, classBindings, view, dataId);
222     ret.push('class="' + classResults.join(' ') + '"');
223     delete attrs['class'];
224   }
225 
226   var attrKeys = Ember.keys(attrs);
227 
228   // For each attribute passed, create an observer and emit the
229   // current value of the property as an attribute.
230   attrKeys.forEach(function(attr) {
231     var property = attrs[attr];
232 
233     ember_assert(fmt("You must provide a String for a bound attribute, not %@", [property]), typeof property === 'string');
234 
235     var value = getPath(ctx, property);
236 
237     ember_assert(fmt("Attributes must be numbers, strings or booleans, not %@", [value]), value == null || typeof value === 'number' || typeof value === 'string' || typeof value === 'boolean');
238 
239     var observer, invoker;
240 
241     /** @private */
242     observer = function observer() {
243       var result = getPath(ctx, property);
244 
245       ember_assert(fmt("Attributes must be numbers, strings or booleans, not %@", [result]), result == null || typeof result === 'number' || typeof result === 'string' || typeof result === 'boolean');
246 
247       var elem = view.$("[data-bindAttr-" + dataId + "='" + dataId + "']");
248 
249       // If we aren't able to find the element, it means the element
250       // to which we were bound has been removed from the view.
251       // In that case, we can assume the template has been re-rendered
252       // and we need to clean up the observer.
253       if (elem.length === 0) {
254         Ember.removeObserver(ctx, property, invoker);
255         return;
256       }
257 
258       Ember.View.applyAttributeBindings(elem, attr, result);
259     };
260 
261     /** @private */
262     invoker = function() {
263       Ember.run.once(observer);
264     };
265 
266     // Add an observer to the view for when the property changes.
267     // When the observer fires, find the element using the
268     // unique data id and update the attribute to the new value.
269     Ember.addObserver(ctx, property, invoker);
270 
271     // if this changes, also change the logic in ember-views/lib/views/view.js
272     var type = typeof value;
273 
274     if ((type === 'string' || (type === 'number' && !isNaN(value)))) {
275       ret.push(attr + '="' + value + '"');
276     } else if (value && type === 'boolean') {
277       ret.push(attr + '="' + attr + '"');
278     }
279   }, this);
280 
281   // Add the unique identifier
282   ret.push('data-bindAttr-' + dataId + '="' + dataId + '"');
283   return new EmberHandlebars.SafeString(ret.join(' '));
284 });

286 /**
287   Helper that, given a space-separated string of property paths and a context,
288   returns an array of class names. Calling this method also has the side 
289   effect of setting up observers at those property paths, such that if they 
290   change, the correct class name will be reapplied to the DOM element.
291 
292   For example, if you pass the string "fooBar", it will first look up the 
293   "fooBar" value of the context. If that value is YES, it will add the 
294   "foo-bar" class to the current element (i.e., the dasherized form of 
295   "fooBar"). If the value is a string, it will add that string as the class. 
296   Otherwise, it will not add any new class name.
297 
298   @param {Ember.Object} context 
299     The context from which to lookup properties
300 
301   @param {String} classBindings 
302     A string, space-separated, of class bindings to use
303 
304   @param {Ember.View} view
305     The view in which observers should look for the element to update
306 
307   @param {Srting} bindAttrId
308     Optional bindAttr id used to lookup elements
309 
310   @returns {Array} An array of class names to add
311 */
312 EmberHandlebars.bindClasses = function(context, classBindings, view, bindAttrId) {
313   var ret = [], newClass, value, elem;
314 
315   // Helper method to retrieve the property from the context and
316   // determine which class string to return, based on whether it is
317   // a Boolean or not.
318   var classStringForProperty = function(property) {
319     var split = property.split(':'),
320         className = split[1];
321 
322     property = split[0];
323 
324     var val = getPath(context, property);
325 
326     // If value is a Boolean and true, return the dasherized property
327     // name.
328     if (val === YES) {
329       if (className) { return className; }
330 
331       // Normalize property path to be suitable for use
332       // as a class name. For exaple, content.foo.barBaz
333       // becomes bar-baz.
334       var parts = property.split('.');
335       return Ember.String.dasherize(parts[parts.length-1]);
336 
337     // If the value is not NO, undefined, or null, return the current
338     // value of the property.
339     } else if (val !== NO && val !== undefined && val !== null) {
340       return val;
341 
342     // Nothing to display. Return null so that the old class is removed
343     // but no new class is added.
344     } else {
345       return null;
346     }
347   };
348 
349   // For each property passed, loop through and setup
350   // an observer.
351   classBindings.split(' ').forEach(function(binding) {
352 
353     // Variable in which the old class value is saved. The observer function
354     // closes over this variable, so it knows which string to remove when
355     // the property changes.
356     var oldClass;
357 
358     var observer, invoker;
359 
360     // Set up an observer on the context. If the property changes, toggle the
361     // class name.
362     /** @private */
363     observer = function() {
364       // Get the current value of the property
365       newClass = classStringForProperty(binding);
366       elem = bindAttrId ? view.$("[data-bindAttr-" + bindAttrId + "='" + bindAttrId + "']") : view.$();
367 
368       // If we can't find the element anymore, a parent template has been
369       // re-rendered and we've been nuked. Remove the observer.
370       if (elem.length === 0) {
371         Ember.removeObserver(context, binding, invoker);
372       } else {
373         // If we had previously added a class to the element, remove it.
374         if (oldClass) {
375           elem.removeClass(oldClass);
376         }
377 
378         // If necessary, add a new class. Make sure we keep track of it so
379         // it can be removed in the future.
380         if (newClass) {
381           elem.addClass(newClass);
382           oldClass = newClass;
383         } else {
384           oldClass = null;
385         }
386       }
387     };
388 
389     /** @private */
390     invoker = function() {
391       Ember.run.once(observer);
392     };
393 
394     property = binding.split(':')[0];
395     Ember.addObserver(context, property, invoker);
396 
397     // We've already setup the observer; now we just need to figure out the 
398     // correct behavior right now on the first pass through.
399     value = classStringForProperty(binding);
400 
401     if (value) {
402       ret.push(value);
403 
404       // Make sure we save the current value so that it can be removed if the 
405       // observer fires.
406       oldClass = value;
407     }
408   });
409 
410   return ret;
411 };
