  8 require('ember-views/views/states/default');
  9 
 10 Ember.View.states.preRender = {
 11   parentState: Ember.View.states._default,
 12 
 13   // a view leaves the preRender state once its element has been
 14   // created (createElement).
 15   insertElement: function(view, fn) {
 16     view.createElement();
 17     view._notifyWillInsertElement(true);
 18     // after createElement, the view will be in the hasElement state.
 19     fn.call(view);
 20     view.transitionTo('inDOM');
 21     view._notifyDidInsertElement();
 22   },
 23 
 24   // This exists for the removal warning, remove later
 25   $: function(view){
 26     if (view._willInsertElementAccessUnsupported) {
 27       console.error("Getting element from willInsertElement is unreliable and no longer supported.");
 28     }
 29     return Ember.$();
 30   },
 31 
 32   // This exists for the removal warning, remove later
 33   getElement: function(view){
 34     if (view._willInsertElementAccessUnsupported) {
 35       console.error("Getting element from willInsertElement is unreliable and no longer supported.");
 36     }
 37     return null;
 38   },
 39 
 40   setElement: function(view, value) {
 41     view.beginPropertyChanges();
 42     view.invalidateRecursively('element');
 43 
 44     if (value !== null) {
 45       view.transitionTo('hasElement');
 46     }
 47 
 48     view.endPropertyChanges();
 49 
 50     return value;
 51   }
 52 };
