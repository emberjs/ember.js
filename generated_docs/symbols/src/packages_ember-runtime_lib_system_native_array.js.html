 14 var get = Ember.get, set = Ember.set;
 15   
 16 // Add Ember.Array to Array.prototype.  Remove methods with native 
 17 // implementations and supply some more optimized versions of generic methods
 18 // because they are so common.
 19 var NativeArray = Ember.Mixin.create(Ember.MutableArray, Ember.Observable, Ember.Copyable, {
 20 
 21   // because length is a built-in property we need to know to just get the 
 22   // original property.
 23   get: function(key) {
 24     if (key==='length') return this.length;
 25     else if ('number' === typeof key) return this[key];
 26     else return this._super(key);  
 27   },
 28   
 29   objectAt: function(idx) {
 30     return this[idx];
 31   },
 32     
 33   // primitive for array support.
 34   replace: function(idx, amt, objects) {
 35 
 36     if (this.isFrozen) throw Ember.FROZEN_ERROR ;
 37 
 38     // if we replaced exactly the same number of items, then pass only the
 39     // replaced range.  Otherwise, pass the full remaining array length
 40     // since everything has shifted
 41     var len = objects ? get(objects, 'length') : 0;
 42     this.arrayContentWillChange(idx, amt, len);
 43     
 44     if (!objects || objects.length === 0) {
 45       this.splice(idx, amt) ;
 46     } else {
 47       var args = [idx, amt].concat(objects) ;
 48       this.splice.apply(this,args) ;
 49     }
 50 
 51     this.arrayContentDidChange(idx, amt, len);
 52     return this ;
 53   },
 54 
 55   // If you ask for an unknown property, then try to collect the value
 56   // from member items.
 57   unknownProperty: function(key, value) {
 58     var ret;// = this.reducedProperty(key, value) ;
 59     if ((value !== undefined) && ret === undefined) {
 60       ret = this[key] = value;
 61     }
 62     return ret ;
 63   },
 64 
 65   // If browser did not implement indexOf natively, then override with
 66   // specialized version
 67   indexOf: function(object, startAt) {
 68     var idx, len = this.length;
 69 
 70     if (startAt === undefined) startAt = 0;
 71     else startAt = (startAt < 0) ? Math.ceil(startAt) : Math.floor(startAt);
 72     if (startAt < 0) startAt += len;
 73 
 74     for(idx=startAt;idx<len;idx++) {
 75       if (this[idx] === object) return idx ;
 76     }
 77     return -1;
 78   },
 79 
 80   lastIndexOf: function(object, startAt) {
 81     var idx, len = this.length;
 82 
 83     if (startAt === undefined) startAt = len-1;
 84     else startAt = (startAt < 0) ? Math.ceil(startAt) : Math.floor(startAt);
 85     if (startAt < 0) startAt += len;
 86 
 87     for(idx=startAt;idx>=0;idx--) {
 88       if (this[idx] === object) return idx ;
 89     }
 90     return -1;
 91   },
 92   
 93   copy: function() {
 94     return this.slice();
 95   }
 96 });
 97 
 98 // Remove any methods implemented natively so we don't override them
 99 var ignore = ['length'];
100 NativeArray.keys().forEach(function(methodName) {
101   if (Array.prototype[methodName]) ignore.push(methodName);
102 });
103 
104 if (ignore.length>0) {
105   NativeArray = NativeArray.without.apply(NativeArray, ignore);
106 }

108 /**
109   The NativeArray mixin contains the properties needed to to make the native
110   Array support Ember.MutableArray and all of its dependent APIs.  Unless you 
111   have Ember.EXTEND_PROTOTYPES set to false, this will be applied automatically.
112   Otherwise you can apply the mixin at anytime by calling 
113   `Ember.NativeArray.activate`.
114   
115 116   @namespace
117   @extends Ember.MutableArray
118   @extends Ember.Array
119   @extends Ember.Enumerable
120   @extends Ember.MutableEnumerable
121   @extends Ember.Copyable
122   @extends Ember.Freezable
123 */
124 Ember.NativeArray = NativeArray;
125 
126 /**
127   Creates an Ember.NativeArray from an Array like object.
128   Does not modify the original object.
129 
130   @returns {Ember.NativeArray}
131 */
132 Ember.A = function(arr){
133   if (arr === undefined) { arr = []; }
134   return Ember.NativeArray.apply(arr);
135 };
136 
137 /**
138   Activates the mixin on the Array.prototype if not already applied.  Calling
139   this method more than once is safe.
140   
141   @returns {void}
142 */
143 Ember.NativeArray.activate = function() {
144   NativeArray.apply(Array.prototype);
145 
146   Ember.A = function(arr) { return arr || []; }
147 };
148 
149 if (Ember.EXTEND_PROTOTYPES) Ember.NativeArray.activate();
