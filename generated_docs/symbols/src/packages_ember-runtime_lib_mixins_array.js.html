 18 /** @private */
 19 function none(obj) { return obj===null || obj===undefined; }

 21 /** @private */
 22 function xform(target, method, params) {
 23   method.call(target, params[0], params[2], params[3], params[4]);

 29 /**
 30   @namespace
 31 
 32   This module implements Observer-friendly Array-like behavior.  This mixin is
 33   picked up by the Array class as well as other controllers, etc. that want to
 34   appear to be arrays.
 35 
 36   Unlike Ember.Enumerable, this mixin defines methods specifically for
 37   collections that provide index-ordered access to their contents.  When you
 38   are designing code that needs to accept any kind of Array-like object, you
 39   should use these methods instead of Array primitives because these will
 40   properly notify observers of changes to the array.
 41 
 42   Although these methods are efficient, they do add a layer of indirection to
 43   your application so it is a good idea to use them only when you need the
 44   flexibility of using both true JavaScript arrays and "virtual" arrays such
 45   as controllers and collections.
 46 
 47   You can use the methods defined in this module to access and modify array
 48   contents in a KVO-friendly way.  You can also be notified whenever the
 49   membership if an array changes by changing the syntax of the property to
 50   .observes('*myProperty.[]') .
 51 
 52   To support Ember.Array in your own class, you must override two
 53   primitives to use it: replace() and objectAt().
 54 
 55   Note that the Ember.Array mixin also incorporates the Ember.Enumerable mixin.  All
 56   Ember.Array-like objects are also enumerable.
 57 
 58   @extends Ember.Enumerable
 59   @since Ember 0.9.0
 60 */
 61 Ember.Array = Ember.Mixin.create(Ember.Enumerable, /** @scope Ember.Array.prototype */ {
 62 
 63   /** @private - compatibility */
 64   isSCArray: true,
 65   
 66   /**
 67     @field {Number} length
 68 
 69     Your array must support the length property.  Your replace methods should
 70     set this property whenever it changes.
 71   */
 72   length: Ember.required(),
 73 
 74   /**
 75     This is one of the primitives you must implement to support Ember.Array.
 76     Returns the object at the named index.  If your object supports retrieving
 77     the value of an array item using get() (i.e. myArray.get(0)), then you do
 78     not need to implement this method yourself.
 79 
 80     @param {Number} idx
 81       The index of the item to return.  If idx exceeds the current length,
 82       return null.
 83   */
 84   objectAt: function(idx) {
 85     if ((idx < 0) || (idx>=get(this, 'length'))) return undefined ;
 86     return get(this, idx);
 87   },
 88 
 89   /** @private (nodoc) - overrides Ember.Enumerable version */
 90   nextObject: function(idx) {
 91     return this.objectAt(idx);
 92   },
 93   
 94   /**
 95     @field []
 96 
 97     This is the handler for the special array content property.  If you get
 98     this property, it will return this.  If you set this property it a new
 99     array, it will replace the current content.
100 
101     This property overrides the default property defined in Ember.Enumerable.
102   */
103   '[]': Ember.computed(function(key, value) {
104     if (value !== undefined) this.replace(0, get(this, 'length'), value) ;
105     return this ;
106   }).property().cacheable(),
107 
108   /** @private (nodoc) - optimized version from Enumerable */
109   contains: function(obj){
110     return this.indexOf(obj) >= 0;
111   },
112 
113   // Add any extra methods to Ember.Array that are native to the built-in Array.
114   /**
115     Returns a new array that is a slice of the receiver.  This implementation
116     uses the observable array methods to retrieve the objects for the new
117     slice.
118 
119     @param beginIndex {Integer} (Optional) index to begin slicing from.
120     @param endIndex {Integer} (Optional) index to end the slice at.
121     @returns {Array} New array with specified slice
122   */
123   slice: function(beginIndex, endIndex) {
124     var ret = [];
125     var length = get(this, 'length') ;
126     if (none(beginIndex)) beginIndex = 0 ;
127     if (none(endIndex) || (endIndex > length)) endIndex = length ;
128     while(beginIndex < endIndex) {
129       ret[ret.length] = this.objectAt(beginIndex++) ;
130     }
131     return ret ;
132   },
133 
134   /**
135     Returns the index of the given object's first occurrence.
136     If no startAt argument is given, the starting location to
137     search is 0. If it's negative, will count backward from 
138     the end of the array. Returns -1 if no match is found.
139 
140     @param {Object} object the item to search for
141     @param {Number} startAt optional starting location to search, default 0
142     @returns {Number} index or -1 if not found
143 
144     @example
145     var arr = ["a", "b", "c", "d", "a"];
146     arr.indexOf("a");      =>  0
147     arr.indexOf("z");      => -1
148     arr.indexOf("a", 2);   =>  4
149     arr.indexOf("a", -1);  =>  4
150     arr.indexOf("b", 3);   => -1
151     arr.indexOf("a", 100); => -1
152   */
153   indexOf: function(object, startAt) {
154     var idx, len = get(this, 'length');
155 
156     if (startAt === undefined) startAt = 0;
157     if (startAt < 0) startAt += len;
158 
159     for(idx=startAt;idx<len;idx++) {
160       if (this.objectAt(idx, true) === object) return idx ;
161     }
162     return -1;
163   },
164 
165   /**
166     Returns the index of the given object's last occurrence.
167     If no startAt argument is given, the search starts from
168     the last position. If it's negative, will count backward 
169     from the end of the array. Returns -1 if no match is found.
170 
171     @param {Object} object the item to search for
172     @param {Number} startAt optional starting location to search, default 0
173     @returns {Number} index or -1 if not found
174 175 
    @example
176     var arr = ["a", "b", "c", "d", "a"];
177     arr.lastIndexOf("a");      =>  4
178     arr.lastIndexOf("z");      => -1
179     arr.lastIndexOf("a", 2);   =>  0
180     arr.lastIndexOf("a", -1);  =>  4
181     arr.lastIndexOf("b", 3);   =>  1
182     arr.lastIndexOf("a", 100); =>  4
183   */
184   lastIndexOf: function(object, startAt) {
185     var idx, len = get(this, 'length');
186 
187     if (startAt === undefined || startAt >= len) startAt = len-1;
188     if (startAt < 0) startAt += len;
189 
190     for(idx=startAt;idx>=0;idx--) {
191       if (this.objectAt(idx) === object) return idx ;
192     }
193     return -1;
194   },

198   // 
199   
200   /**
201     Adds an array observer to the receiving array.  The array observer object
202     normally must implement two methods:
203     
204     * `arrayWillChange(start, removeCount, addCount)` - This method will be
205       called just before the array is modified.
206     * `arrayDidChange(start, removeCount, addCount)` - This method will be
207       called just after the array is modified.
208       
209     Both callbacks will be passed the starting index of the change as well a 
210     a count of the items to be removed and added.  You can use these callbacks
211     to optionally inspect the array during the change, clear caches, or do 
212     any other bookkeeping necessary.
213     
214     In addition to passing a target, you can also include an options hash 
215     which you can use to override the method names that will be invoked on the
216     target.
217     
218     @param {Object} target
219       The observer object.
220       
221     @param {Hash} opts
222       Optional hash of configuration options including willChange, didChange,
223       and a context option.
224       
225     @returns {Ember.Array} receiver
226   */
227   addArrayObserver: function(target, opts) {
228     var willChange = (opts && opts.willChange) || 'arrayWillChange',
229         didChange  = (opts && opts.didChange) || 'arrayDidChange';
230 
231     var hasObservers = get(this, 'hasArrayObservers');
232     if (!hasObservers) Ember.propertyWillChange(this, 'hasArrayObservers');
233     Ember.addListener(this, '@array:before', target, willChange, xform);
234     Ember.addListener(this, '@array:change', target, didChange, xform);
235     if (!hasObservers) Ember.propertyDidChange(this, 'hasArrayObservers');
236     return this;
237   },
238   
239   /**
240     Removes an array observer from the object if the observer is current 
241     registered.  Calling this method multiple times with the same object will
242     have no effect.
243     
244     @param {Object} target
245       The object observing the array.
246     
247     @returns {Ember.Array} receiver
248   */
249   removeArrayObserver: function(target, opts) {
250     var willChange = (opts && opts.willChange) || 'arrayWillChange',
251         didChange  = (opts && opts.didChange) || 'arrayDidChange';
252 
253     var hasObservers = get(this, 'hasArrayObservers');
254     if (hasObservers) Ember.propertyWillChange(this, 'hasArrayObservers');
255     Ember.removeListener(this, '@array:before', target, willChange, xform);
256     Ember.removeListener(this, '@array:change', target, didChange, xform);
257     if (hasObservers) Ember.propertyDidChange(this, 'hasArrayObservers');
258     return this;
259   },
260   
261   /**
262     Becomes true whenever the array currently has observers watching changes
263     on the array.
264     
265     @property {Boolean}
266   */
267   hasArrayObservers: Ember.computed(function() {
268     return Ember.hasListeners(this, '@array:change') || Ember.hasListeners(this, '@array:before');
269   }).property().cacheable(),
270   
271   /**
272     If you are implementing an object that supports Ember.Array, call this 
273     method just before the array content changes to notify any observers and
274     invalidate any related properties.  Pass the starting index of the change
275     as well as a delta of the amounts to change.
276     
277     @param {Number} startIdx
278       The starting index in the array that will change.
279       
280     @param {Number} removeAmt
281       The number of items that will be removed.  If you pass null assumes 0
282 283     
284     @param {Number} addAmt
285       The number of items that will be added.  If you pass null assumes 0.
286       
287     @returns {Ember.Array} receiver
288   */
289   arrayContentWillChange: function(startIdx, removeAmt, addAmt) {
290 
291     // if no args are passed assume everything changes
292     if (startIdx===undefined) {
293       startIdx = 0;
294       removeAmt = addAmt = -1;
295     } else {
296       if (!removeAmt) removeAmt=0;
297       if (!addAmt) addAmt=0;
298     }
299 
300     Ember.sendEvent(this, '@array:before', startIdx, removeAmt, addAmt);
301 
302     var removing, lim;
303     if (startIdx>=0 && removeAmt>=0 && get(this, 'hasEnumerableObservers')) {
304       removing = [];
305       lim = startIdx+removeAmt;
306       for(var idx=startIdx;idx<lim;idx++) removing.push(this.objectAt(idx));
307     } else {
308       removing = removeAmt;
309     }
310     
311     this.enumerableContentWillChange(removing, addAmt);
312 
313     // Make sure the @each proxy is set up if anyone is observing @each
314     if (Ember.isWatching(this, '@each')) { get(this, '@each'); }
315     return this;
316   },
317   
318   arrayContentDidChange: function(startIdx, removeAmt, addAmt) {
319 
320     // if no args are passed assume everything changes
321     if (startIdx===undefined) {
322       startIdx = 0;
323       removeAmt = addAmt = -1;
324     } else {
325       if (!removeAmt) removeAmt=0;
326       if (!addAmt) addAmt=0;
327     }
328     
329     var adding, lim;
330     if (startIdx>=0 && addAmt>=0 && get(this, 'hasEnumerableObservers')) {
331       adding = [];
332       lim = startIdx+addAmt;
333       for(var idx=startIdx;idx<lim;idx++) adding.push(this.objectAt(idx));
334     } else {
335       adding = addAmt;
336     }
337 
338     this.enumerableContentDidChange(removeAmt, adding);
339     Ember.sendEvent(this, '@array:change', startIdx, removeAmt, addAmt);
340     return this;
341   },
