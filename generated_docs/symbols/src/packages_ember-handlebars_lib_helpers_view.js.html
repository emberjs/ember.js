 11 var get = Ember.get, set = Ember.set;
 12 var PARENT_VIEW_PATH = /^parentView\./;
 13 
 14 /** @private */
 15 Ember.Handlebars.ViewHelper = Ember.Object.create({
 16 
 17   viewClassFromHTMLOptions: function(viewClass, options, thisContext) {
 18     var extensions = {},
 19         classes = options['class'],
 20         dup = false;
 21 
 22     if (options.id) {
 23       extensions.elementId = options.id;
 24       dup = true;
 25     }
 26 
 27     if (classes) {
 28       classes = classes.split(' ');
 29       extensions.classNames = classes;
 30       dup = true;
 31     }
 32 
 33     if (options.classBinding) {
 34       extensions.classNameBindings = options.classBinding.split(' ');
 35       dup = true;
 36     }
 37 
 38     if (dup) {
 39       options = jQuery.extend({}, options);
 40       delete options.id;
 41       delete options['class'];
 42       delete options.classBinding;
 43     }
 44 
 45     // Look for bindings passed to the helper and, if they are
 46     // local, make them relative to the current context instead of the
 47     // view.
 48     var path;
 49 
 50     for (var prop in options) {
 51       if (!options.hasOwnProperty(prop)) { continue; }
 52 
 53       // Test if the property ends in "Binding"
 54       if (Ember.IS_BINDING.test(prop)) {
 55         path = options[prop];
 56         if (!Ember.isGlobalPath(path)) {
 57           if (path === 'this') {
 58             options[prop] = 'bindingContext';
 59           } else {
 60             options[prop] = 'bindingContext.'+path;
 61           }
 62         }
 63       }
 64     }
 65 
 66     // Make the current template context available to the view
 67     // for the bindings set up above.
 68     extensions.bindingContext = thisContext;
 69 
 70     return viewClass.extend(options, extensions);
 71   },
 72 
 73   helper: function(thisContext, path, options) {
 74     var inverse = options.inverse,
 75         data = options.data,
 76         view = data.view,
 77         fn = options.fn,
 78         hash = options.hash,
 79         newView;
 80 
 81     if ('string' === typeof path) {
 82       newView = Ember.Handlebars.getPath(thisContext, path);
 83       ember_assert("Unable to find view at path '" + path + "'", !!newView);
 84     } else {
 85       newView = path;
 86     }
 87 
 88     ember_assert(Ember.String.fmt('You must pass a view class to the #view helper, not %@ (%@)', [path, newView]), Ember.View.detect(newView));
 89 
 90     newView = this.viewClassFromHTMLOptions(newView, hash, thisContext);
 91     var currentView = data.view;
 92     var viewOptions = {};
 93 
 94     if (fn) {
 95       ember_assert("You cannot provide a template block if you also specified a templateName", !get(viewOptions, 'templateName') && !newView.PrototypeMixin.keys().indexOf('templateName') >= 0);
 96       viewOptions.template = fn;
 97     }
 98 
 99     currentView.appendChild(newView, viewOptions);
100   }
101 });
102 
103 /**
104   @name Handlebars.helpers.view
105   @param {String} path
106   @param {Hash} options
107   @returns {String} HTML string
108 */
109 Ember.Handlebars.registerHelper('view', function(path, options) {
110   ember_assert("The view helper only takes a single argument", arguments.length <= 2);
111 
112   // If no path is provided, treat path param as options.
113   if (path && path.data && path.data.isRenderData) {
114     options = path;
115     path = "Ember.View";
116   }
117 
118   return Ember.Handlebars.ViewHelper.helper(this, path, options);
119 });
