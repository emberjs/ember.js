 20 // ..........................................................
 21 // DEPENDENT KEYS
 22 // 
 23 
 24 // data structure:
 25 //  meta.deps = { 
 26 //   'depKey': { 
 27 //     'keyName': count,
 28 //     __emberproto__: SRC_OBJ [to detect clones]
 29 //     },
 30 //   __emberproto__: SRC_OBJ
 31 //  }
 32 
 33 /** @private */
 34 function uniqDeps(obj, depKey) {
 35   var m = meta(obj), deps, ret;
 36   deps = m.deps;
 37   if (!deps) {
 38     deps = m.deps = { __emberproto__: obj };
 39   } else if (deps.__emberproto__ !== obj) {
 40     deps = m.deps = o_create(deps);
 41     deps.__emberproto__ = obj;
 42   }
 43   
 44   ret = deps[depKey];
 45   if (!ret) {
 46     ret = deps[depKey] = { __emberproto__: obj };
 47   } else if (ret.__emberproto__ !== obj) {
 48     ret = deps[depKey] = o_create(ret);
 49     ret.__emberproto__ = obj;
 50   }
 51   
 52   return ret;
 53 }
 54 
 55 /** @private */
 56 function addDependentKey(obj, keyName, depKey) {
 57   var deps = uniqDeps(obj, depKey);
 58  59   deps[keyName] = (deps[keyName] || 0) + 1;
 60   Ember.watch(obj, depKey);
 61 }
 62 
 63 /** @private */
 64 function removeDependentKey(obj, keyName, depKey) {
 65   var deps = uniqDeps(obj, depKey);
 66   deps[keyName] = (deps[keyName] || 0) - 1;
 67   Ember.unwatch(obj, depKey);
 68 }
 69 
 70 /** @private */
 71 function addDependentKeys(desc, obj, keyName) {
 72   var keys = desc._dependentKeys, 
 73       len  = keys ? keys.length : 0;
 74   for(var idx=0;idx<len;idx++) addDependentKey(obj, keyName, keys[idx]);
 75 }

 81 /** @private */
 82 function ComputedProperty(func, opts) {
 83   this.func = func;
 84   this._cacheable = opts && opts.cacheable;
 85   this._dependentKeys = opts && opts.dependentKeys;
 86 }

 94 var CP_DESC = {
 95   configurable: true,
 96   enumerable:   true,
 97   get: function() { return undefined; }, // for when use_accessors is false.
 98   set: Ember.Descriptor.MUST_USE_SETTER  // for when use_accessors is false
 99 };
100 
101 /** @private */
102 function mkCpGetter(keyName, desc) {
103   var cacheable = desc._cacheable, 
104       func     = desc.func;
105       
106   if (cacheable) {
107     return function() {
108       var ret, cache = meta(this).cache;
109       if (keyName in cache) return cache[keyName];
110       ret = cache[keyName] = func.call(this, keyName);
111       return ret ;
112     };
113   } else {
114     return function() {
115       return func.call(this, keyName);
116     };
117   }
118 }
119 
120 /** @private */
121 function mkCpSetter(keyName, desc) {
122   var cacheable = desc._cacheable,
123       func      = desc.func;
124       
125   return function(value) {
126     var m = meta(this, cacheable),
127         watched = (m.source===this) && m.watching[keyName]>0,
128         ret, oldSuspended, lastSetValues;
129 
130     oldSuspended = desc._suspended;
131 132     desc._suspended = this;
133 
134     watched = watched && m.lastSetValues[keyName]!==guidFor(value);
135     if (watched) {
136       m.lastSetValues[keyName] = guidFor(value);
137       Ember.propertyWillChange(this, keyName);
138     }
139     
140     if (cacheable) delete m.cache[keyName];
141     ret = func.call(this, keyName, value);
142     if (cacheable) m.cache[keyName] = ret;
143     if (watched) Ember.propertyDidChange(this, keyName);
144     desc._suspended = oldSuspended;
145     return ret;
146   };
147 }

155 /**
156   Call on a computed property to set it into cacheable mode.  When in this
157   mode the computed property will automatically cache the return value of 
158   your function until one of the dependent keys changes.
159 
160   @param {Boolean} aFlag optional set to false to disable cacheing
161   @returns {Ember.ComputedProperty} receiver
162 */
163 Cp.cacheable = function(aFlag) {
164   this._cacheable = aFlag!==false;
165   return this;
166 };
167 
168 /**
169   Sets the dependent keys on this computed property.  Pass any number of 
170   arguments containing key paths that this computed property depends on.
171   
172   @param {String} path... zero or more property paths
173   @returns {Ember.ComputedProperty} receiver
174 */
175 Cp.property = function() {
176   this._dependentKeys = a_slice.call(arguments);
177   return this;
178 };
179 
180 /**
181   In some cases, you may want to annotate computed properties with additional
182   metadata about how they function or what values they operate on. For example,
183   computed property functions may close over variables that are then no longer
184   available for introspection.
185 
186   You can pass a hash of these values to a computed property like this:
187 
188       person: function() {
189         var personId = this.get('personId');
190         return App.Person.create({ id: personId });
191       }.property().meta({ type: App.Person })
192 
193   The hash that you pass to the `meta()` function will be saved on the
194   computed property descriptor under the `_meta` key. Ember runtime
195   exposes a public API for retrieving these values from classes,
196   via the `metaForProperty()` function.
197 198 */
199 
200 Cp.meta = function(meta) {
201   this._meta = meta;
202   return this;
203 };
204 
205 /** @private - impl descriptor API */
206 Cp.setup = function(obj, keyName, value) {
207   CP_DESC.get = mkCpGetter(keyName, this);
208   CP_DESC.set = mkCpSetter(keyName, this);
209   o_defineProperty(obj, keyName, CP_DESC);
210   CP_DESC.get = CP_DESC.set = null;
211   addDependentKeys(this, obj, keyName);
212 };
213 
214 /** @private - impl descriptor API */
215 Cp.teardown = function(obj, keyName) {
216   var keys = this._dependentKeys, 
217       len  = keys ? keys.length : 0;
218   for(var idx=0;idx<len;idx++) removeDependentKey(obj, keyName, keys[idx]);
219 
220   if (this._cacheable) delete meta(obj).cache[keyName];
221   
222   return null; // no value to restore
223 };
224 
225 /** @private - impl descriptor API */
226 Cp.didChange = function(obj, keyName) {
227   if (this._cacheable && (this._suspended !== obj)) {
228     delete meta(obj).cache[keyName];
229   }
230 };
231 
232 /** @private - impl descriptor API */
233 Cp.get = function(obj, keyName) {
234   var ret, cache;
235   
236   if (this._cacheable) {
237     cache = meta(obj).cache;
238     if (keyName in cache) return cache[keyName];
239     ret = cache[keyName] = this.func.call(obj, keyName);
240   } else {
241 242     ret = this.func.call(obj, keyName);
243   }
244   return ret ;
245 };
246 
247 /** @private - impl descriptor API */
248 Cp.set = function(obj, keyName, value) {
249   var cacheable = this._cacheable;
250   
251   var m = meta(obj, cacheable),
252       watched = (m.source===obj) && m.watching[keyName]>0,
253       ret, oldSuspended, lastSetValues;
254 
255   oldSuspended = this._suspended;
256   this._suspended = obj;
257 
258   watched = watched && m.lastSetValues[keyName]!==guidFor(value);
259   if (watched) {
260     m.lastSetValues[keyName] = guidFor(value);
261     Ember.propertyWillChange(obj, keyName);
262   }
263   
264   if (cacheable) delete m.cache[keyName];
265   ret = this.func.call(obj, keyName, value);
266   if (cacheable) m.cache[keyName] = ret;
267   if (watched) Ember.propertyDidChange(obj, keyName);
268   this._suspended = oldSuspended;
269   return ret;
270 };
271 
272 Cp.val = function(obj, keyName) {
273   return meta(obj, false).values[keyName];
274 };

276 if (!Ember.platform.hasPropertyAccessors) {
277   Cp.setup = function(obj, keyName, value) {
278     obj[keyName] = undefined; // so it shows up in key iteration
279     addDependentKeys(this, obj, keyName);
280   };
281   
282 } else if (!USE_ACCESSORS) {
283   Cp.setup = function(obj, keyName) {
284     // throw exception if not using Ember.get() and Ember.set() when supported
285     o_defineProperty(obj, keyName, CP_DESC);
286     addDependentKeys(this, obj, keyName);
287   };

290 /**
291   This helper returns a new property descriptor that wraps the passed 
292   computed property function.  You can use this helper to define properties
293   with mixins or via Ember.defineProperty().
294   
295   The function you pass will be used to both get and set property values.
296   The function should accept two parameters, key and value.  If value is not
297   undefined you should set the value first.  In either case return the 
298   current value of the property.
299   
300   @param {Function} func
301     The computed property function.
302     
303   @returns {Ember.ComputedProperty} property descriptor instance
304 */
305 Ember.computed = function(func) {
306   return new ComputedProperty(func);
