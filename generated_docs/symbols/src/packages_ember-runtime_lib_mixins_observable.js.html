  7 var get = Ember.get, set = Ember.set;

  9 /**
 10   @class
 11 
 12   Restores some of the Ember 1.x Ember.Observable mixin API.  The new property 
 13   observing system does not require Ember.Observable to be applied anymore.
 14   Instead, on most browsers you can just access properties directly.  For
 15   code that needs to run on IE7 or IE8 you should use Ember.get() and Ember.set()
 16   instead.
 17   
 18   If you have older code and you want to bring back the older Ember 1.x observable
 19   API, you can do so by readding Ember.Observable to Ember.Object like so:
 20   
 21       Ember.Object.reopen(Ember.Observable);
 22     
 23   You will then be able to use the traditional get(), set() and other 
 24   observable methods on your objects.
 25 
 26   @extends Ember.Mixin
 27 */
 28 Ember.Observable = Ember.Mixin.create(/** @scope Ember.Observable.prototype */ {
 29 
 30   /** @private - compatibility */
 31   isObserverable: true,
 32   
 33   /**
 34     Retrieves the value of key from the object.
 35 
 36     This method is generally very similar to using object[key] or object.key,
 37     however it supports both computed properties and the unknownProperty
 38     handler.
 39 
 40     ## Computed Properties
 41 
 42     Computed properties are methods defined with the property() modifier
 43     declared at the end, such as:
 44 
 45           fullName: function() {
 46             return this.getEach('firstName', 'lastName').compact().join(' ');
 47           }.property('firstName', 'lastName')
 48 
 49     When you call get() on a computed property, the property function will be
 50     called and the return value will be returned instead of the function
 51     itself.
 52 
 53     ## Unknown Properties
 54 
 55     Likewise, if you try to call get() on a property whose values is
 56     undefined, the unknownProperty() method will be called on the object.
 57     If this method reutrns any value other than undefined, it will be returned
 58     instead. This allows you to implement "virtual" properties that are
 59     not defined upfront.
 60 
 61     @param {String} key The property to retrieve
 62     @returns {Object} The property value or undefined.
 63   */
 64   get: function(keyName) {
 65     return get(this, keyName);
 66   },
 67 
 68   /**
 69     To get multiple properties at once, call getProperties
 70     with a list of strings:
 71 
 72           record.getProperties('firstName', 'lastName', 'zipCode');
 73 
 74     @param {String...} list of keys to get
 75     @returns {Hash}
 76   */
 77   getProperties: function() {
 78     var ret = {};
 79     for(var i = 0; i < arguments.length; i++) {
 80       ret[arguments[i]] = get(this, arguments[i]);
 81     }
 82     return ret;
 83   },
 84   
 85   /**
 86     Sets the key equal to value.
 87 
 88     This method is generally very similar to calling object[key] = value or
 89     object.key = value, except that it provides support for computed
 90     properties, the unknownProperty() method and property observers.
 91 
 92     ## Computed Properties
 93 
 94     If you try to set a value on a key that has a computed property handler
 95     defined (see the get() method for an example), then set() will call
 96     that method, passing both the value and key instead of simply changing
 97     the value itself. This is useful for those times when you need to
 98     implement a property that is composed of one or more member
 99     properties.
100 
101     ## Unknown Properties
102 
103     If you try to set a value on a key that is undefined in the target
104     object, then the unknownProperty() handler will be called instead. This
105     gives you an opportunity to implement complex "virtual" properties that
106     are not predefined on the obejct. If unknownProperty() returns
107     undefined, then set() will simply set the value on the object.
108 
109     ## Property Observers
110 
111     In addition to changing the property, set() will also register a
112     property change with the object. Unless you have placed this call
113     inside of a beginPropertyChanges() and endPropertyChanges(), any "local"
114     observers (i.e. observer methods declared on the same object), will be
115     called immediately. Any "remote" observers (i.e. observer methods
116     declared on another object) will be placed in a queue and called at a
117     later time in a coelesced manner.
118 
119     ## Chaining
120 
121     In addition to property changes, set() returns the value of the object
122     itself so you can do chaining like this:
123 
124           record.set('firstName', 'Charles').set('lastName', 'Jolley');
125 
126     @param {String} key The property to set
127     @param {Object} value The value to set or null.
128     @returns {Ember.Observable}
129   */
130   set: function(keyName, value) {
131     set(this, keyName, value);
132     return this;
133   },
134   
135   /**
136     To set multiple properties at once, call setProperties
137     with a Hash:
138 
139           record.setProperties({ firstName: 'Charles', lastName: 'Jolley' });
140 
141     @param {Hash} hash the hash of keys and values to set
142     @returns {Ember.Observable}
143   */
144   setProperties: function(hash) {
145     var self = this;
146     Ember.changeProperties(function(){
147       for(var prop in hash) {
148         if (hash.hasOwnProperty(prop)) set(self, prop, hash[prop]);
149       }
150     });
151     return this;
152   },
153 
154   /**
155     Begins a grouping of property changes.
156 
157     You can use this method to group property changes so that notifications
158     will not be sent until the changes are finished. If you plan to make a
159     large number of changes to an object at one time, you should call this
160     method at the beginning of the changes to suspend change notifications.
161     When you are done making changes, call endPropertyChanges() to allow
162     notification to resume.
163 
164     @returns {Ember.Observable}
165   */
166   beginPropertyChanges: function() {
167     Ember.beginPropertyChanges();
168     return this;
169   },
170   
171   /**
172     Ends a grouping of property changes.
173 
174     You can use this method to group property changes so that notifications
175     will not be sent until the changes are finished. If you plan to make a
176     large number of changes to an object at one time, you should call
177     beginPropertyChanges() at the beginning of the changes to suspend change
178     notifications. When you are done making changes, call this method to allow
179     notification to resume.
180 
181     @returns {Ember.Observable}
182   */
183   endPropertyChanges: function() {
184     Ember.endPropertyChanges();
185     return this;
186   },
187   
188   /**
189     Notify the observer system that a property is about to change.
190 
191     Sometimes you need to change a value directly or indirectly without
192     actually calling get() or set() on it. In this case, you can use this
193     method and propertyDidChange() instead. Calling these two methods
194     together will notify all observers that the property has potentially
195     changed value.
196 
197     Note that you must always call propertyWillChange and propertyDidChange as
198     a pair. If you do not, it may get the property change groups out of order
199     and cause notifications to be delivered more often than you would like.
200 
201     @param {String} key The property key that is about to change.
202     @returns {Ember.Observable}
203   */
204   propertyWillChange: function(keyName){
205     Ember.propertyWillChange(this, keyName);
206     return this;
207   },
208   
209   /**
210     Notify the observer system that a property has just changed.
211 
212     Sometimes you need to change a value directly or indirectly without
213     actually calling get() or set() on it. In this case, you can use this
214     method and propertyWillChange() instead. Calling these two methods
215     together will notify all observers that the property has potentially
216     changed value.
217 
218     Note that you must always call propertyWillChange and propertyDidChange as
219     a pair. If you do not, it may get the property change groups out of order
220     and cause notifications to be delivered more often than you would like.
221 
222     @param {String} key The property key that has just changed.
223     @param {Object} value The new value of the key. May be null.
224     @param {Boolean} _keepCache Private property
225     @returns {Ember.Observable}
226   */
227   propertyDidChange: function(keyName) {
228     Ember.propertyDidChange(this, keyName);
229     return this;
230   },
231   
232   notifyPropertyChange: function(keyName) {
233     this.propertyWillChange(keyName);
234     this.propertyDidChange(keyName);
235     return this;
236   }, 
237 
238   /**
239     Adds an observer on a property.
240 
241     This is the core method used to register an observer for a property.
242 
243     Once you call this method, anytime the key's value is set, your observer
244     will be notified. Note that the observers are triggered anytime the
245     value is set, regardless of whether it has actually changed. Your
246     observer should be prepared to handle that.
247 
248     You can also pass an optional context parameter to this method. The
249     context will be passed to your observer method whenever it is triggered.
250     Note that if you add the same target/method pair on a key multiple times
251     with different context parameters, your observer will only be called once
252     with the last context you passed.
253 
254     ## Observer Methods
255 
256     Observer methods you pass should generally have the following signature if
257     you do not pass a "context" parameter:
258 
259           fooDidChange: function(sender, key, value, rev);
260 
261     The sender is the object that changed. The key is the property that
262     changes. The value property is currently reserved and unused. The rev
263     is the last property revision of the object when it changed, which you can
264     use to detect if the key value has really changed or not.
265 
266     If you pass a "context" parameter, the context will be passed before the
267     revision like so:
268 
269           fooDidChange: function(sender, key, value, context, rev);
270 
271     Usually you will not need the value, context or revision parameters at
272     the end. In this case, it is common to write observer methods that take
273     only a sender and key value as parameters or, if you aren't interested in
274     any of these values, to write an observer that has no parameters at all.
275 
276     @param {String} key The key to observer
277     @param {Object} target The target object to invoke
278     @param {String|Function} method The method to invoke.
279     @returns {Ember.Object} self
280   */
281   addObserver: function(key, target, method) {
282     Ember.addObserver(this, key, target, method);
283   },
284   
285   /**
286     Remove an observer you have previously registered on this object. Pass
287     the same key, target, and method you passed to addObserver() and your
288     target will no longer receive notifications.
289 
290     @param {String} key The key to observer
291     @param {Object} target The target object to invoke
292     @param {String|Function} method The method to invoke.
293     @returns {Ember.Observable} reciever
294   */
295   removeObserver: function(key, target, method) {
296     Ember.removeObserver(this, key, target, method);
297   },
298   
299   /**
300     Returns YES if the object currently has observers registered for a
301     particular key. You can use this method to potentially defer performing
302     an expensive action until someone begins observing a particular property
303     on the object.
304 
305     @param {String} key Key to check
306     @returns {Boolean}
307   */
308   hasObserverFor: function(key) {
309     return Ember.hasListeners(this, key+':change');
310   },
311 
312   unknownProperty: function(key) {
313     return undefined;
314   },
315   
316   setUnknownProperty: function(key, value) {
317     this[key] = value;
318   },
319   
320   getPath: function(path) {
321     return Ember.getPath(this, path);
322   },
323   
324   setPath: function(path, value) {
325     Ember.setPath(this, path, value);
326     return this;
327   },
328   
329   incrementProperty: function(keyName, increment) {
330     if (!increment) { increment = 1; }
331     set(this, keyName, (get(this, keyName) || 0)+increment);
332     return get(this, keyName);
333   },
334   
335   decrementProperty: function(keyName, increment) {
336     if (!increment) { increment = 1; }
337     set(this, keyName, (get(this, keyName) || 0)-increment);
338     return get(this, keyName);
339   },
340   
341   toggleProperty: function(keyName) {
342     set(this, keyName, !get(this, keyName));
343     return get(this, keyName);
344   },
345   
346   observersForKey: function(keyName) {
347     return Ember.observersFor(this, keyName);
348   }
349     
350 });
