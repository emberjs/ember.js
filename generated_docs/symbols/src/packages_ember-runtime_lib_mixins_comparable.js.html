 11 /**
 12   @namespace
 13 
 14   Implements some standard methods for comparing objects. Add this mixin to
 15   any class you create that can compare its instances.
 16 
 17   You should implement the compare() method.
 18 
 19   @since Ember 0.9
 20 */
 21 Ember.Comparable = Ember.Mixin.create( /** @scope Ember.Comparable.prototype */{
 22 
 23   /**
 24     walk like a duck. Indicates that the object can be compared.
 25 
 26     @type Boolean
 27     @default YES
 28     @constant
 29   */
 30   isComparable: true,
 31 
 32   /**
 33     Override to return the result of the comparison of the two parameters. The
 34     compare method should return:
 35 
 36       - -1 if a < b
 37       - 0 if a == b
 38       - 1 if a > b
 39 
 40     Default implementation raises an exception.
 41 
 42     @param a {Object} the first object to compare
 43     @param b {Object} the second object to compare
 44     @returns {Integer} the result of the comparison
 45   */
 46   compare: Ember.required(Function)
 47 
 48 });
