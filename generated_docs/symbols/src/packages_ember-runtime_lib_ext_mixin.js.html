  8 require('ember-metal/mixin');
  9 
 10 var IS_BINDING = Ember.IS_BINDING = /^.+Binding$/;
 11 
 12 Ember._mixinBindings = function(obj, key, value, m) {
 13   if (IS_BINDING.test(key)) {
 14     if (!(value instanceof Ember.Binding)) {
 15       value = new Ember.Binding(key.slice(0,-7), value); // make binding
 16     } else {
 17       value.to(key.slice(0, -7));
 18     }
 19     value.connect(obj);
 20 
 21     // keep a set of bindings in the meta so that when we rewatch we can
 22     // resync them...
 23     var bindings = m.bindings;
 24     if (!bindings) {
 25       bindings = m.bindings = { __emberproto__: obj };
 26     } else if (bindings.__emberproto__ !== obj) {
 27       bindings = m.bindings = Ember.create(m.bindings);
 28       bindings.__emberproto__ = obj;
 29     }
 30 
 31     bindings[key] = true;
 32   }
 33   
 34   return value;
 35 };
