  7 require('ember-runtime/mixins/target_action_support');
  8 
  9 var get = Ember.get, set = Ember.set;
 10 
 11 Ember.Button = Ember.View.extend(Ember.TargetActionSupport, {
 12   classNames: ['ember-button'],
 13   classNameBindings: ['isActive'],
 14 
 15   tagName: 'button',
 16 
 17   propagateEvents: false,
 18 
 19   attributeBindings: ['type', 'disabled', 'href'],
 20 
 21   // Defaults to 'button' if tagName is 'input' or 'button'
 22   type: Ember.computed(function(key, value) {
 23     var tagName = this.get('tagName');
 24     if (value !== undefined) { this._type = value; }
 25     if (this._type !== undefined) { return this._type; }
 26     if (tagName === 'input' || tagName === 'button') { return 'button'; }
 27   }).property('tagName').cacheable(),
 28 
 29   disabled: false,
 30 
 31   // Allow 'a' tags to act like buttons
 32   href: Ember.computed(function() {
 33     return this.get('tagName') === 'a' ? '#' : null;
 34   }).property('tagName').cacheable(),
 35 
 36   mouseDown: function() {
 37     if (!get(this, 'disabled')) {
 38       set(this, 'isActive', true);
 39       this._mouseDown = true;
 40       this._mouseEntered = true;
 41     }
 42     return get(this, 'propagateEvents');
 43   },
 44 
 45   mouseLeave: function() {
 46     if (this._mouseDown) {
 47       set(this, 'isActive', false);
 48       this._mouseEntered = false;
 49     }
 50   },
 51 
 52   mouseEnter: function() {
 53     if (this._mouseDown) {
 54       set(this, 'isActive', true);
 55       this._mouseEntered = true;
 56     }
 57   },
 58 
 59   mouseUp: function(event) {
 60     if (get(this, 'isActive')) {
 61       // Actually invoke the button's target and action.
 62       // This method comes from the Ember.TargetActionSupport mixin.
 63       this.triggerAction();
 64       set(this, 'isActive', false);
 65     }
 66 
 67     this._mouseDown = false;
 68     this._mouseEntered = false;
 69     return get(this, 'propagateEvents');
 70   },
 71 
 72   keyDown: function(event) {
 73     // Handle space or enter
 74     if (event.keyCode === 13 || event.keyCode === 32) {
 75       this.mouseDown();
 76     }
 77   },
 78 
 79   keyUp: function(event) {
 80     // Handle space or enter
 81     if (event.keyCode === 13 || event.keyCode === 32) {
 82       this.mouseUp();
 83     }
 84   },
 85 
 86   // TODO: Handle proper touch behavior.  Including should make inactive when
 87   // finger moves more than 20x outside of the edge of the button (vs mouse
 88   // which goes inactive as soon as mouse goes out of edges.)
 89 
 90   touchStart: function(touch) {
 91     return this.mouseDown(touch);
 92   },
 93 
 94   touchEnd: function(touch) {
 95     return this.mouseUp(touch);
 96   }
 97 });
