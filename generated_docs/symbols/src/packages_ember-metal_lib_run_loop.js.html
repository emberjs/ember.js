 18 var slice = Array.prototype.slice;
 19 
 20 // invokes passed params - normalizing so you can pass target/func,
 21 // target/string or just func
 22 /** @private */
 23 function invoke(target, method, args, ignore) {
 24 
 25   if (method===undefined) {
 26     method = target;
 27     target = undefined;
 28   }
 29 
 30   if ('string'===typeof method) method = target[method];
 31   if (args && ignore>0) {
 32     args = args.length>ignore ? slice.call(args, ignore) : null;
 33   }
 34 
 35   // Unfortunately in some browsers we lose the backtrace if we rethrow the existing error,
 36   // so in the event that we don't have an `onerror` handler we don't wrap in a try/catch
 37   if ('function' === typeof Ember.onerror) {
 38     try {
 39       // IE8's Function.prototype.apply doesn't accept undefined/null arguments.
 40       return method.apply(target || this, args || []);
 41     } catch (error) {
 42       Ember.onerror(error);
 43     }
 44   } else {
 45     // IE8's Function.prototype.apply doesn't accept undefined/null arguments.
 46     return method.apply(target || this, args || []);
 47   }
 48 }

 55 var timerMark; // used by timers...

 57 /** @private */
 58 var K = function() {};
 59 
 60 /** @private */
 61 var RunLoop = function(prev) {
 62   var self;
 63 
 64   if (this instanceof RunLoop) {
 65     self = this;
 66   } else {
 67     self = new K();
 68   }
 69 
 70   self._prev = prev || null;
 71   self.onceTimers = {};
 72 
 73   return self;
 74 };
 75 
 76 K.prototype = RunLoop.prototype;
 77 
 78 RunLoop.prototype = {
 79   end: function() {
 80     this.flush();
 81   },
 82 
 83   prev: function() {
 84     return this._prev;
 85   },
 86 
 87   // ..........................................................
 88   // Delayed Actions
 89   //
 90 
 91   schedule: function(queueName, target, method) {
 92     var queues = this._queues, queue;
 93     if (!queues) queues = this._queues = {};
 94     queue = queues[queueName];
 95     if (!queue) queue = queues[queueName] = [];
 96 
 97     var args = arguments.length>3 ? slice.call(arguments, 3) : null;
 98     queue.push({ target: target, method: method, args: args });
 99     return this;
100   },
101 
102   flush: function(queueName) {
103     var queues = this._queues, queueNames, idx, len, queue, log;
104 
105     if (!queues) return this; // nothing to do
106 
107     function iter(item) {
108       invoke(item.target, item.method, item.args);
109     }
110 
111     Ember.watch.flushPending(); // make sure all chained watchers are setup
112 
113     if (queueName) {
114       while (this._queues && (queue = this._queues[queueName])) {
115         this._queues[queueName] = null;
116 
117         log = Ember.LOG_BINDINGS && queueName==='sync';
118         if (log) Ember.Logger.log('Begin: Flush Sync Queue');
119 
120         // the sync phase is to allow property changes to propogate.  don't
121         // invoke observers until that is finished.
122 123         if (queueName === 'sync') Ember.beginPropertyChanges();
124         queue.forEach(iter);
125         if (queueName === 'sync') Ember.endPropertyChanges();
126 
127         if (log) Ember.Logger.log('End: Flush Sync Queue');
128 
129       }
130 
131     } else {
132       queueNames = Ember.run.queues;
133       len = queueNames.length;
134       do {
135         this._queues = null;
136         for(idx=0;idx<len;idx++) {
137           queueName = queueNames[idx];
138           queue = queues[queueName];
139 
140           log = Ember.LOG_BINDINGS && queueName==='sync';
141           if (log) Ember.Logger.log('Begin: Flush Sync Queue');
142 
143           if (queueName === 'sync') Ember.beginPropertyChanges();
144           if (queue) queue.forEach(iter);
145           if (queueName === 'sync') Ember.endPropertyChanges();
146 
147           if (log) Ember.Logger.log('End: Flush Sync Queue');
148 
149         }
150 
151       } while (queues = this._queues); // go until queues stay clean
152     }
153 
154     timerMark = null;
155 
156     return this;
157   }
158 
159 };

167 /**
168   Runs the passed target and method inside of a runloop, ensuring any
169   deferred actions including bindings and views updates are flushed at the
170   end.
171 
172   Normally you should not need to invoke this method yourself.  However if
173   you are implementing raw event handlers when interfacing with other
174   libraries or plugins, you should probably wrap all of your code inside this
175   call.
176 
177   @function
178   @param {Object} target
179     (Optional) target of method to call
180 
181   @param {Function|String} method
182     Method to invoke.  May be a function or a string.  If you pass a string
183     then it will be looked up on the passed target.
184 
185   @param {Object...} args
186     Any additional arguments you wish to pass to the method.
187 
188   @returns {Object} return value from invoking the passed function.
189 */
190 Ember.run = function(target, method) {
191 
192   var ret, loop;
193   run.begin();
194   try {
195     if (target || method) ret = invoke(target, method, arguments, 2);
196   } finally {
197     run.end();
198   }
199   return ret;
200 };

202 /** @private */
203 var run = Ember.run;
204 
205 
206 /**
207   Begins a new RunLoop.  Any deferred actions invoked after the begin will
208   be buffered until you invoke a matching call to Ember.run.end().  This is
209   an lower-level way to use a RunLoop instead of using Ember.run().
210 
211   @returns {void}
212 */
213 Ember.run.begin = function() {
214   run.currentRunLoop = new RunLoop(run.currentRunLoop);
215 };
216 
217 /**
218   Ends a RunLoop.  This must be called sometime after you call Ember.run.begin()
219   to flush any deferred actions.  This is a lower-level way to use a RunLoop
220   instead of using Ember.run().
221 
222   @returns {void}
223 */
224 Ember.run.end = function() {
225   ember_assert('must have a current run loop', run.currentRunLoop);
226   try {
227     run.currentRunLoop.end();
228   }
229   finally {
230     run.currentRunLoop = run.currentRunLoop.prev();
231   }
232 };

234 /**
235   Array of named queues.  This array determines the order in which queues
236   are flushed at the end of the RunLoop.  You can define your own queues by
237   simply adding the queue name to this array.  Normally you should not need
238   to inspect or modify this property.
239 
240   @property {String}
241 */
242 Ember.run.queues = ['sync', 'actions', 'destroy', 'timers'];
243 
244 /**
245   Adds the passed target/method and any optional arguments to the named
246   queue to be executed at the end of the RunLoop.  If you have not already
247   started a RunLoop when calling this method one will be started for you
248   automatically.
249 
250   At the end of a RunLoop, any methods scheduled in this way will be invoked.
251   Methods will be invoked in an order matching the named queues defined in
252   the run.queues property.
253 
254   @param {String} queue
255     The name of the queue to schedule against.  Default queues are 'sync' and
256     'actions'
257 
258   @param {Object} target
259     (Optional) target object to use as the context when invoking a method.
260 
261   @param {String|Function} method
262     The method to invoke.  If you pass a string it will be resolved on the
263     target object at the time the scheduled item is invoked allowing you to
264     change the target function.
265 
266   @param {Object} arguments...
267     Optional arguments to be passed to the queued method.
268 
269   @returns {void}
270 */
271 Ember.run.schedule = function(queue, target, method) {
272   var loop = run.autorun();
273   loop.schedule.apply(loop, arguments);
274 };

278 /** @private */
279 function autorun() {
280   autorunTimer = null;
281   if (run.currentRunLoop) run.end();
282 }

284 /**
285   Begins a new RunLoop if necessary and schedules a timer to flush the
286   RunLoop at a later time.  This method is used by parts of Ember to
287   ensure the RunLoop always finishes.  You normally do not need to call this
288   method directly.  Instead use Ember.run().
289 
290   @returns {Ember.RunLoop} the new current RunLoop
291 */
292 Ember.run.autorun = function() {
293 
294   if (!run.currentRunLoop) {
295     run.begin();
296 
297     // TODO: throw during tests
298 299     if (Ember.testing) {
300       run.end();
301     } else if (!autorunTimer) {
302       autorunTimer = setTimeout(autorun, 1);
303     }
304   }
305 
306   return run.currentRunLoop;
307 };
308 
309 /**
310   Immediately flushes any events scheduled in the 'sync' queue.  Bindings
311   use this queue so this method is a useful way to immediately force all
312   bindings in the application to sync.
313 
314   You should call this method anytime you need any changed state to propogate
315   throughout the app immediately without repainting the UI.
316 
317   @returns {void}
318 */
319 Ember.run.sync = function() {
320   run.autorun();
321   run.currentRunLoop.flush('sync');
322 };

328 var timers = {}; // active timers...
329 
330 var laterScheduled = false;
331 /** @private */
332 function invokeLaterTimers() {
333   var now = (+ new Date()), earliest = -1;
334   for(var key in timers) {
335     if (!timers.hasOwnProperty(key)) continue;
336     var timer = timers[key];
337     if (timer && timer.expires) {
338       if (now >= timer.expires) {
339         delete timers[key];
340         invoke(timer.target, timer.method, timer.args, 2);
341       } else {
342         if (earliest<0 || (timer.expires < earliest)) earliest=timer.expires;
343       }
344     }
345   }
346 
347   // schedule next timeout to fire...
348   if (earliest>0) setTimeout(invokeLaterTimers, earliest-(+ new Date()));
349 }
350 
351 /**
352   Invokes the passed target/method and optional arguments after a specified
353   period if time.  The last parameter of this method must always be a number
354   of milliseconds.
355 
356   You should use this method whenever you need to run some action after a
357   period of time inside of using setTimeout().  This method will ensure that
358   items that expire during the same script execution cycle all execute
359   together, which is often more efficient than using a real setTimeout.
360 
361   @param {Object} target
362     (optional) target of method to invoke
363 
364   @param {Function|String} method
365     The method to invoke.  If you pass a string it will be resolved on the
366     target at the time the method is invoked.
367 
368   @param {Object...} args
369     Optional arguments to pass to the timeout.
370 
371   @param {Number} wait
372     Number of milliseconds to wait.
373 
374   @returns {Timer} an object you can use to cancel a timer at a later time.
375 */
376 Ember.run.later = function(target, method) {
377   var args, expires, timer, guid, wait;
378 
379   // setTimeout compatibility...
380   if (arguments.length===2 && 'function' === typeof target) {
381     wait   = method;
382     method = target;
383 384     target = undefined;
385     args   = [target, method];
386 
387   } else {
388     args = slice.call(arguments);
389     wait = args.pop();
390   }
391 
392   expires = (+ new Date())+wait;
393   timer   = { target: target, method: method, expires: expires, args: args };
394   guid    = Ember.guidFor(timer);
395   timers[guid] = timer;
396   run.once(timers, invokeLaterTimers);
397   return guid;
398 };

400 /** @private */
401 function invokeOnceTimer(guid, onceTimers) {
402   if (onceTimers[this.tguid]) delete onceTimers[this.tguid][this.mguid];
403   if (timers[guid]) invoke(this.target, this.method, this.args, 2);
404   delete timers[guid];
405 }

407 /**
408   Schedules an item to run one time during the current RunLoop.  Calling
409   this method with the same target/method combination will have no effect.
410 
411   Note that although you can pass optional arguments these will not be
412   considered when looking for duplicates.  New arguments will replace previous
413   calls.
414 
415   @param {Object} target
416     (optional) target of method to invoke
417 
418   @param {Function|String} method
419     The method to invoke.  If you pass a string it will be resolved on the
420     target at the time the method is invoked.
421 
422   @param {Object...} args
423     Optional arguments to pass to the timeout.
424 
425 
426   @returns {Object} timer
427 */
428 Ember.run.once = function(target, method) {
429   var tguid = Ember.guidFor(target), mguid = Ember.guidFor(method), guid, timer;
430 
431   var onceTimers = run.autorun().onceTimers;
432   guid = onceTimers[tguid] && onceTimers[tguid][mguid];
433   if (guid && timers[guid]) {
434     timers[guid].args = slice.call(arguments); // replace args
435 
436   } else {
437     timer = {
438       target: target,
439       method: method,
440       args:   slice.call(arguments),
441       tguid:  tguid,
442       mguid:  mguid
443     };
444 
445     guid  = Ember.guidFor(timer);
446     timers[guid] = timer;
447     if (!onceTimers[tguid]) onceTimers[tguid] = {};
448     onceTimers[tguid][mguid] = guid; // so it isn't scheduled more than once
449 
450     run.schedule('actions', timer, invokeOnceTimer, guid, onceTimers);
451   }
452 
453   return guid;
454 };

456 457 var scheduledNext = false;
458 /** @private */
459 function invokeNextTimers() {
460   scheduledNext = null;
461   for(var key in timers) {
462     if (!timers.hasOwnProperty(key)) continue;
463     var timer = timers[key];
464     if (timer.next) {
465       delete timers[key];
466       invoke(timer.target, timer.method, timer.args, 2);
467     }
468   }
469 }
470 
471 /**
472   Schedules an item to run after control has been returned to the system.
473   This is often equivalent to calling setTimeout(function...,1).
474 
475   @param {Object} target
476     (optional) target of method to invoke
477 
478   @param {Function|String} method
479     The method to invoke.  If you pass a string it will be resolved on the
480     target at the time the method is invoked.
481 
482   @param {Object...} args
483     Optional arguments to pass to the timeout.
484 
485   @returns {Object} timer
486 */
487 Ember.run.next = function(target, method) {
488   var timer, guid;
489 
490   timer = {
491     target: target,
492     method: method,
493     args: slice.call(arguments),
494     next: true
495   };
496 
497   guid = Ember.guidFor(timer);
498   timers[guid] = timer;
499 
500   if (!scheduledNext) scheduledNext = setTimeout(invokeNextTimers, 1);
501   return guid;
502 };
503 
504 /**
505   Cancels a scheduled item.  Must be a value returned by `Ember.run.later()`,
506   `Ember.run.once()`, or `Ember.run.next()`.
507 
508   @param {Object} timer
509     Timer object to cancel
510 
511   @returns {void}
512 */
513 Ember.run.cancel = function(timer) {
514   delete timers[timer];
515 };
