 15 var OUT_OF_RANGE_EXCEPTION = "Index out of range" ;
 16 var EMPTY = [];
 17 
 18 // ..........................................................
 19 // HELPERS
 20 // 
 21 
 22 var get = Ember.get, set = Ember.set;
 23 
 24 /**
 25   @class
 26 
 27   This mixin defines the API for modifying array-like objects.  These methods
 28   can be applied only to a collection that keeps its items in an ordered set.
 29   
 30   Note that an Array can change even if it does not implement this mixin.
 31   For example, a SparyArray may not be directly modified but if its 
 32   underlying enumerable changes, it will change also.
 33 
 34   @extends Ember.Mixin
 35   @extends Ember.Array
 36   @extends Ember.MutableEnumerable
 37 */
 38 Ember.MutableArray = Ember.Mixin.create(Ember.Array, Ember.MutableEnumerable,
 39   /** @scope Ember.MutableArray.prototype */ {
 40 
 41   /**
 42     __Required.__ You must implement this method to apply this mixin.
 43 
 44     This is one of the primitves you must implement to support Ember.Array.  You
 45     should replace amt objects started at idx with the objects in the passed
 46     array.  You should also call this.enumerableContentDidChange() ;
 47 
 48     @param {Number} idx
 49       Starting index in the array to replace.  If idx >= length, then append 
 50       to the end of the array.
 51 
 52     @param {Number} amt
 53       Number of elements that should be removed from the array, starting at
 54       *idx*.
 55 
 56     @param {Array} objects
 57       An array of zero or more objects that should be inserted into the array 
 58       at *idx*
 59   */
 60   replace: Ember.required(),
 61 
 62   /**
 63     This will use the primitive replace() method to insert an object at the
 64     specified index.
 65 
 66     @param {Number} idx index of insert the object at.
 67     @param {Object} object object to insert
 68   */
 69   insertAt: function(idx, object) {
 70     if (idx > get(this, 'length')) throw new Error(OUT_OF_RANGE_EXCEPTION) ;
 71     this.replace(idx, 0, [object]) ;
 72     return this ;
 73   },
 74 
 75   /**
 76     Remove an object at the specified index using the replace() primitive
 77     method.  You can pass either a single index, a start and a length or an
 78     index set.
 79 
 80     If you pass a single index or a start and length that is beyond the
 81     length this method will throw an Ember.OUT_OF_RANGE_EXCEPTION
 82 
 83     @param {Number|Ember.IndexSet} start index, start of range, or index set
 84     @param {Number} len length of passing range
 85     @returns {Object} receiver
 86   */
 87   removeAt: function(start, len) {
 88 
 89     var delta = 0;
 90 
 91     if ('number' === typeof start) {
 92 
 93       if ((start < 0) || (start >= get(this, 'length'))) {
 94         throw new Error(OUT_OF_RANGE_EXCEPTION);
 95       }
 96 
 97       // fast case
 98       if (len === undefined) len = 1;
 99       this.replace(start, len, EMPTY);
100     }
101 
102     // TODO: Reintroduce Ember.IndexSet support
103     // this.beginPropertyChanges();
104     // start.forEachRange(function(start, length) {
105     //   start -= delta ;
106     //   delta += length ;
107     //   this.replace(start, length, empty); // remove!
108     // }, this);
109     // this.endPropertyChanges();
110 
111     return this ;
112   },
113 
114   /**
115     Push the object onto the end of the array.  Works just like push() but it
116     is KVO-compliant.
117   */
118   pushObject: function(obj) {
119     this.insertAt(get(this, 'length'), obj) ;
120     return obj ;
121   },
122 
123 
124   /**
125     Add the objects in the passed numerable to the end of the array.  Defers
126     notifying observers of the change until all objects are added.
127 
128     @param {Ember.Enumerable} objects the objects to add
129     @returns {Ember.Array} receiver
130   */
131   pushObjects: function(objects) {
132     this.replace(get(this, 'length'), 0, objects);
133     return this;
134   },
135 
136   /**
137     Pop object from array or nil if none are left.  Works just like pop() but
138     it is KVO-compliant.
139   */
140   popObject: function() {
141     var len = get(this, 'length') ;
142     if (len === 0) return null ;
143 
144     var ret = this.objectAt(len-1) ;
145     this.removeAt(len-1, 1) ;
146     return ret ;
147   },
148 
149   /**
150     Shift an object from start of array or nil if none are left.  Works just
151     like shift() but it is KVO-compliant.
152   */
153   shiftObject: function() {
154     if (get(this, 'length') === 0) return null ;
155     var ret = this.objectAt(0) ;
156     this.removeAt(0) ;
157     return ret ;
158   },
159 
160   /**
161     Unshift an object to start of array.  Works just like unshift() but it is
162     KVO-compliant.
163   */
164   unshiftObject: function(obj) {
165     this.insertAt(0, obj) ;
166     return obj ;
167   },
168 
169 
170   /**
171     Adds the named objects to the beginning of the array.  Defers notifying
172     observers until all objects have been added.
173 
174     @param {Ember.Enumerable} objects the objects to add
175     @returns {Ember.Array} receiver
176   */
177   unshiftObjects: function(objects) {
178     this.beginPropertyChanges();
179     objects.forEach(function(obj) { this.unshiftObject(obj); }, this);
180     this.endPropertyChanges();
181     return this;
182   },
183   
184   // ..........................................................
185   // IMPLEMENT Ember.MutableEnumerable
186   // 
187 
188   /** @private (nodoc) */
189   removeObject: function(obj) {
190     var loc = get(this, 'length') || 0;
191     while(--loc >= 0) {
192       var curObject = this.objectAt(loc) ;
193       if (curObject === obj) this.removeAt(loc) ;
194     }
195     return this ;
196   },
197   
198   /** @private (nodoc) */
199   addObject: function(obj) {
200     if (!this.contains(obj)) this.pushObject(obj);
201     return this ;
202   }
203     
204 });
