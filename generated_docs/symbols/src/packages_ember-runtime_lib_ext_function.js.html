  8 require('ember-runtime/core');
  9 
 10 var a_slice = Array.prototype.slice;
 11 
 12 if (Ember.EXTEND_PROTOTYPES) {
 13 
 14   Function.prototype.property = function() {
 15     var ret = Ember.computed(this);
 16     return ret.property.apply(ret, arguments);
 17   };
 18 
 19   Function.prototype.observes = function() {
 20     this.__ember_observes__ = a_slice.call(arguments);
 21     return this;
 22   };
 23 
 24   Function.prototype.observesBefore = function() {
 25     this.__ember_observesBefore__ = a_slice.call(arguments);
 26     return this;
 27   };
 28 
 29 }
