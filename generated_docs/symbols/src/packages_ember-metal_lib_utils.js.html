 14 // Used for guid generation...
 15 var GUID_KEY = '__ember'+ (+ new Date());
 16 var uuid, numberCache, stringCache;
 17 
 18 uuid         = 0;
 19 numberCache  = [];
 20 stringCache  = {};
 21 
 22 var GUID_DESC = {
 23   configurable: true,
 24   writable: true,
 25   enumerable: false
 26 };
 27 
 28 var o_defineProperty = Ember.platform.defineProperty;
 29 var o_create = Ember.platform.create;
 30 
 31 /**
 32   @private
 33   @static
 34   @type String
 35   @constant
 36 
 37   A unique key used to assign guids and other private metadata to objects.
 38   If you inspect an object in your browser debugger you will often see these.
 39   They can be safely ignored.
 40 
 41   On browsers that support it, these properties are added with enumeration 
 42   disabled so they won't show up when you iterate over your properties.
 43 */
 44 Ember.GUID_KEY = GUID_KEY;
 45 
 46 /**
 47   @private
 48 
 49   Generates a new guid, optionally saving the guid to the object that you
 50   pass in.  You will rarely need to use this method.  Instead you should
 51   call Ember.guidFor(obj), which return an existing guid if available.
 52 
 53   @param {Object} obj
 54     Optional object the guid will be used for.  If passed in, the guid will
 55     be saved on the object and reused whenever you pass the same object 
 56     again.
 57 
 58     If no object is passed, just generate a new guid.
 59 
 60   @param {String} prefix
 61     Optional prefix to place in front of the guid.  Useful when you want to
 62     separate the guid into separate namespaces.
 63 
 64   @returns {String} the guid
 65 */
 66 Ember.generateGuid = function(obj, prefix) {
 67   if (!prefix) prefix = 'ember';
 68   var ret = (prefix + (uuid++));
 69   if (obj) {
 70     GUID_DESC.value = ret;
 71     o_defineProperty(obj, GUID_KEY, GUID_DESC);
 72     GUID_DESC.value = null;
 73   }
 74 
 75   return ret ;
 76 };
 77 
 78 /**
 79   @private
 80 
 81   Returns a unique id for the object.  If the object does not yet have
 82   a guid, one will be assigned to it.  You can call this on any object,
 83   Ember.Object-based or not, but be aware that it will add a _guid property.
 84 
 85   You can also use this method on DOM Element objects.
 86 
 87   @method
 88   @param obj {Object} any object, string, number, Element, or primitive
 89   @returns {String} the unique guid for this instance.
 90 */
 91 Ember.guidFor = function(obj) {
 92 
 93   // special cases where we don't want to add a key to object
 94   if (obj === undefined) return "(undefined)";
 95   if (obj === null) return "(null)";
 96 
 97   var cache, ret;
 98   var type = typeof obj;
 99 
100   // Don't allow prototype changes to String etc. to change the guidFor
101   switch(type) {
102     case 'number':
103       ret = numberCache[obj];
104       if (!ret) ret = numberCache[obj] = 'nu'+obj;
105       return ret;
106 
107     case 'string':
108       ret = stringCache[obj];
109       if (!ret) ret = stringCache[obj] = 'st'+(uuid++);
110       return ret;
111 
112     case 'boolean':
113       return obj ? '(true)' : '(false)';
114 
115     default:
116       if (obj[GUID_KEY]) return obj[GUID_KEY];
117       if (obj === Object) return '(Object)';
118       if (obj === Array)  return '(Array)';
119       return Ember.generateGuid(obj, 'ember');
120   }
121 };

128 var META_DESC = {
129   writable:    true,
130   configurable: false,
131   enumerable:  false,
132   value: null
133 };
134 
135 var META_KEY = Ember.GUID_KEY+'_meta';
136 
137 /**
138   The key used to store meta information on object for property observing.
139 
140   @static
141   @property
142 */
143 Ember.META_KEY = META_KEY;
144 
145 // Placeholder for non-writable metas.
146 var EMPTY_META = {
147   descs: {},
148   watching: {}
149 }; 
150 
151 if (Object.freeze) Object.freeze(EMPTY_META);
152 
153 /**
154   @private
155   @function
156   
157   Retrieves the meta hash for an object.  If 'writable' is true ensures the
158   hash is writable for this object as well.
159   
160   The meta object contains information about computed property descriptors as
161   well as any watched properties and other information.  You generally will
162   not access this information directly but instead work with higher level 
163   methods that manipulate this hash indirectly.
164 
165   @param {Object} obj
166     The object to retrieve meta for
167     
168   @param {Boolean} writable
169     Pass false if you do not intend to modify the meta hash, allowing the 
170     method to avoid making an unnecessary copy.
171     
172   @returns {Hash}
173 */
174 Ember.meta = function meta(obj, writable) {
175 
176   var ret = obj[META_KEY];
177   if (writable===false) return ret || EMPTY_META;
178 
179   if (!ret) {
180     o_defineProperty(obj, META_KEY, META_DESC);
181     ret = obj[META_KEY] = {
182       descs: {},
183       watching: {},
184       values: {},
185       lastSetValues: {},
186       cache:  {},
187       source: obj
188     };
189     
190     // make sure we don't accidentally try to create constructor like desc
191     ret.descs.constructor = null;
192     
193   } else if (ret.source !== obj) {
194     ret = obj[META_KEY] = o_create(ret);
195     ret.descs    = o_create(ret.descs);
196     ret.values   = o_create(ret.values);
197     ret.watching = o_create(ret.watching);
198     ret.lastSetValues = {};
199     ret.cache    = {};
200     ret.source   = obj;
201   }
202   return ret;
203 };

205 Ember.getMeta = function getMeta(obj, property) {
206   var meta = Ember.meta(obj, false);
207   return meta[property];
208 };
209 
210 Ember.setMeta = function setMeta(obj, property, value) {
211   var meta = Ember.meta(obj, true);
212   meta[property] = value;
213   return value;
214 };

216 /**
217   @private
218 
219   In order to store defaults for a class, a prototype may need to create
220   a default meta object, which will be inherited by any objects instantiated
221   from the class's constructor.
222 
223   However, the properties of that meta object are only shallow-cloned,
224   so if a property is a hash (like the event system's `listeners` hash),
225   it will by default be shared across all instances of that class.
226 
227   This method allows extensions to deeply clone a series of nested hashes or
228   other complex objects. For instance, the event system might pass
229   ['listeners', 'foo:change', 'ember157'] to `prepareMetaPath`, which will
230   walk down the keys provided.
231 
232   For each key, if the key does not exist, it is created. If it already
233   exists and it was inherited from its constructor, the constructor's
234   key is cloned.
235 
236   You can also pass false for `writable`, which will simply return
237   undefined if `prepareMetaPath` discovers any part of the path that
238   shared or undefined.
239 
240   @param {Object} obj The object whose meta we are examining
241   @param {Array} path An array of keys to walk down
242   @param {Boolean} writable whether or not to create a new meta
243     (or meta property) if one does not already exist or if it's
244     shared with its constructor
245 */
246 Ember.metaPath = function(obj, path, writable) {
247   var meta = Ember.meta(obj, writable), keyName, value;
248 
249   for (var i=0, l=path.length; i<l; i++) {
250     keyName = path[i];
251     value = meta[keyName];
252 
253     if (!value) {
254       if (!writable) { return undefined; }
255       value = meta[keyName] = { __ember_source__: obj };
256     } else if (value.__ember_source__ !== obj) {
257       if (!writable) { return undefined; }
258       value = meta[keyName] = o_create(value);
259       value.__ember_source__ = obj;
260     }
261 
262     meta = value;
263   }
264 
265   return value;
266 };

268 /**
269   @private
270 
271   Wraps the passed function so that `this._super` will point to the superFunc
272   when the function is invoked.  This is the primitive we use to implement
273   calls to super.
274   
275   @param {Function} func
276     The function to call
277     
278   @param {Function} superFunc
279     The super function.
280     
281   @returns {Function} wrapped function.
282 */
283 Ember.wrap = function(func, superFunc) {
284   
285   function K() {}
286   
287   var newFunc = function() {
288     var ret, sup = this._super;
289     this._super = superFunc || K;
290     ret = func.apply(this, arguments);
291     this._super = sup;
292     return ret;
293   };
294   
295   newFunc.base = func;
296   return newFunc;
297 };

299 /**
300   @function
301   
302   Returns YES if the passed object is an array or Array-like.
303 
304   Ember Array Protocol:
305 
306     - the object has an objectAt property
307     - the object is a native Array
308     - the object is an Object, and has a length property
309 
310   Unlike Ember.typeOf this method returns true even if the passed object is
311   not formally array but appears to be array-like (i.e. implements Ember.Array)
312 
313   @param {Object} obj The object to test
314   @returns {Boolean}
315 */
316 Ember.isArray = function(obj) {
317   if (!obj || obj.setInterval) { return false; }
318   if (Array.isArray && Array.isArray(obj)) { return true; }
319   if (Ember.Array && Ember.Array.detect(obj)) { return true; }
320   if ((obj.length !== undefined) && 'object'===typeof obj) { return true; }
321   return false;
322 };
323 
324 /**
325   Forces the passed object to be part of an array.  If the object is already
326   an array or array-like, returns the object.  Otherwise adds the object to
327   an array.  If obj is null or undefined, returns an empty array.
328   
329   @param {Object} obj the object
330   @returns {Array}
331 */
332 Ember.makeArray = function(obj) {
333   if (obj==null) return [];
334   return Ember.isArray(obj) ? obj : [obj];
335 };
