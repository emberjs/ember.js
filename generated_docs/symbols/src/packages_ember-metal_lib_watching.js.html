 43 var DEP_SKIP = { __emberproto__: true }; // skip some keys and toString
 44 
 45 /** @private */
 46 function iterDeps(method, obj, depKey, seen, meta) {
 47 
 48   var guid = guidFor(obj);
 49   if (!seen[guid]) seen[guid] = {};
 50   if (seen[guid][depKey]) return ;
 51   seen[guid][depKey] = true;
 52 
 53   var deps = meta.deps;
 54   deps = deps && deps[depKey];
 55   if (deps) {
 56     for(var key in deps) {
 57       if (DEP_SKIP[key]) continue;
 58       method(obj, key);
 59     }
 60   }
 61 }

 64 var WILL_SEEN, DID_SEEN;
 65 
 66 // called whenever a property is about to change to clear the cache of any dependent keys (and notify those properties of changes, etc...)
 67 /** @private */
 68 function dependentKeysWillChange(obj, depKey, meta) {
 69   var seen = WILL_SEEN, top = !seen;
 70   if (top) seen = WILL_SEEN = {};
 71   iterDeps(propertyWillChange, obj, depKey, seen, meta);
 72   if (top) WILL_SEEN = null;
 73 }
 74 
 75 // called whenever a property has just changed to update dependent keys
 76 /** @private */
 77 function dependentKeysDidChange(obj, depKey, meta) {
 78   var seen = DID_SEEN, top = !seen;
 79   if (top) seen = DID_SEEN = {};
 80   iterDeps(propertyDidChange, obj, depKey, seen, meta);
 81   if (top) DID_SEEN = null;
 82 }

 88 /** @private */
 89 function addChainWatcher(obj, keyName, node) {
 90   if (!obj || ('object' !== typeof obj)) return; // nothing to do
 91   var m = meta(obj);
 92   var nodes = m.chainWatchers;
 93   if (!nodes || nodes.__emberproto__ !== obj) {
 94     nodes = m.chainWatchers = { __emberproto__: obj };
 95   }
 96 
 97   if (!nodes[keyName]) nodes[keyName] = {};
 98   nodes[keyName][guidFor(node)] = node;
 99   Ember.watch(obj, keyName);
100 }
101 
102 /** @private */
103 function removeChainWatcher(obj, keyName, node) {
104   if (!obj || ('object' !== typeof obj)) return; // nothing to do
105   var m = meta(obj, false);
106   var nodes = m.chainWatchers;
107   if (!nodes || nodes.__emberproto__ !== obj) return; //nothing to do
108   if (nodes[keyName]) delete nodes[keyName][guidFor(node)];
109   Ember.unwatch(obj, keyName);
110 }

112 var pendingQueue = [];
113 
114 // attempts to add the pendingQueue chains again.  If some of them end up
115 // back in the queue and reschedule is true, schedules a timeout to try 
116 // again.
117 /** @private */
118 function flushPendingChains(reschedule) {
119   if (pendingQueue.length===0) return ; // nothing to do
120   
121   var queue = pendingQueue;
122   pendingQueue = [];
123   
124   queue.forEach(function(q) { q[0].add(q[1]); });
125   if (reschedule!==false && pendingQueue.length>0) {
126     setTimeout(flushPendingChains, 1);
127   }
128 }

135 // A ChainNode watches a single key on an object.  If you provide a starting
136 // value for the key then the node won't actually watch it.  For a root node 
137 // pass null for parent and key and object for value.
138 /** @private */
139 var ChainNode = function(parent, key, value, separator) {
140   var obj;
141   this._parent = parent;
142   this._key    = key;
143 
144   // _watching is true when calling get(this._parent, this._key) will
145   // return the value of this node.
146   //
147   // It is false for the root of a chain (because we have no parent)
148   // and for global paths (because the parent node is the object with
149   // the observer on it)
150   this._watching = value===undefined;
151 
152   this._value  = value;
153   this._separator = separator || '.';
154   this._paths = {};
155   if (this._watching) {
156     this._object = parent.value();
157     if (this._object) addChainWatcher(this._object, this._key, this);
158   }
159 
160   // Special-case: the EachProxy relies on immediate evaluation to
161   // establish its observers.
162   //
163   // TODO: Replace this with an efficient callback that the EachProxy
164   // can implement.
165   if (this._parent && this._parent._key === '@each') {
166     this.value();
167   }
168 };

171 var Wp = ChainNode.prototype;
172 
173 Wp.value = function() {
174   if (this._value === undefined && this._watching){
175     var obj = this._parent.value();
176     this._value = (obj && !isProto(obj)) ? get(obj, this._key) : undefined;
177   }
178   return this._value;
179 };
180 
181 Wp.destroy = function() {
182   if (this._watching) {
183     var obj = this._object;
184     if (obj) removeChainWatcher(obj, this._key, this);
185     this._watching = false; // so future calls do nothing
186   }
187 };
188 
189 // copies a top level object only
190 Wp.copy = function(obj) {
191   var ret = new ChainNode(null, null, obj, this._separator);
192   var paths = this._paths, path;
193   for(path in paths) {
194     if (!(paths[path] > 0)) continue; // this check will also catch non-number vals.
195     ret.add(path);
196   }
197   return ret;
198 };

200 // called on the root node of a chain to setup watchers on the specified 
201 // path.
202 Wp.add = function(path) {
203   var obj, tuple, key, src, separator, paths;
204 
205   paths = this._paths;
206   paths[path] = (paths[path] || 0) + 1 ;
207 
208   obj = this.value();
209   tuple = normalizeTuple(obj, path);
210 
211   // the path was a local path
212   if (tuple[0] && (tuple[0] === obj)) {
213     path = tuple[1];
214     key  = firstKey(path);
215     path = path.slice(key.length+1);
216 
217   // global path, but object does not exist yet.
218   // put into a queue and try to connect later.
219   } else if (!tuple[0]) {
220     pendingQueue.push([this, path]);
221     tuple.length = 0;
222     return;
223 
224   // global path, and object already exists
225   } else {
226     src  = tuple[0];
227     key  = path.slice(0, 0-(tuple[1].length+1));
228 229     separator = path.slice(key.length, key.length+1);
230     path = tuple[1];
231   }
232 
233   tuple.length = 0;
234   this.chain(key, path, src, separator);
235 };
236 
237 // called on the root node of a chain to teardown watcher on the specified
238 // path
239 Wp.remove = function(path) {
240   var obj, tuple, key, src, paths;
241 
242   paths = this._paths;
243   if (paths[path] > 0) paths[path]--;
244 
245   obj = this.value();
246   tuple = normalizeTuple(obj, path);
247   if (tuple[0] === obj) {
248     path = tuple[1];
249     key  = firstKey(path);
250     path = path.slice(key.length+1);
251 
252   } else {
253     src  = tuple[0];
254     key  = path.slice(0, 0-(tuple[1].length+1));
255     path = tuple[1];
256   }
257 
258   tuple.length = 0;
259   this.unchain(key, path);
260 };
261 
262 Wp.count = 0;
263 
264 Wp.chain = function(key, path, src, separator) {
265   var chains = this._chains, node;
266   if (!chains) chains = this._chains = {};
267 
268   node = chains[key];
269   if (!node) node = chains[key] = new ChainNode(this, key, src, separator);
270   node.count++; // count chains...
271 
272   // chain rest of path if there is one
273   if (path && path.length>0) {
274     key = firstKey(path);
275     path = path.slice(key.length+1);
276     node.chain(key, path); // NOTE: no src means it will observe changes...
277   }
278 };
279 
280 Wp.unchain = function(key, path) {
281   var chains = this._chains, node = chains[key];
282 
283   // unchain rest of path first...
284   if (path && path.length>1) {
285     key  = firstKey(path);
286     path = path.slice(key.length+1);
287     node.unchain(key, path);
288   }
289 
290   // delete node if needed.
291   node.count--;
292   if (node.count<=0) {
293     delete chains[node._key];
294     node.destroy();
295   }
296   
297 };

299 Wp.willChange = function() {
300   var chains = this._chains;
301   if (chains) {
302     for(var key in chains) {
303       if (!chains.hasOwnProperty(key)) continue;
304       chains[key].willChange();
305     }
306   }
307   
308   if (this._parent) this._parent.chainWillChange(this, this._key, 1);
309 };
310 
311 Wp.chainWillChange = function(chain, path, depth) {
312   if (this._key) path = this._key+this._separator+path;
313 
314   if (this._parent) {
315     this._parent.chainWillChange(this, path, depth+1);
316   } else {
317     if (depth>1) Ember.propertyWillChange(this.value(), path);
318     path = 'this.'+path;
319     if (this._paths[path]>0) Ember.propertyWillChange(this.value(), path);
320   }
321 };
322 
323 Wp.chainDidChange = function(chain, path, depth) {
324   if (this._key) path = this._key+this._separator+path;
325   if (this._parent) {
326     this._parent.chainDidChange(this, path, depth+1);
327   } else {
328     if (depth>1) Ember.propertyDidChange(this.value(), path);
329     path = 'this.'+path;
330     if (this._paths[path]>0) Ember.propertyDidChange(this.value(), path);
331   }
332 };
333 
334 Wp.didChange = function(suppressEvent) {
335   // invalidate my own value first.
336   if (this._watching) {
337     var obj = this._parent.value();
338     if (obj !== this._object) {
339       removeChainWatcher(this._object, this._key, this);
340       this._object = obj;
341       addChainWatcher(obj, this._key, this);
342     }
343     this._value  = undefined;
344 
345     // Special-case: the EachProxy relies on immediate evaluation to
346     // establish its observers.
347     if (this._parent && this._parent._key === '@each')
348       this.value();
349   }
350   
351   // then notify chains...
352   var chains = this._chains;
353   if (chains) {
354     for(var key in chains) {
355       if (!chains.hasOwnProperty(key)) continue;
356       chains[key].didChange(suppressEvent);
357     }
358   }
359 
360   if (suppressEvent) return;
361 
362   // and finally tell parent about my path changing...
363   if (this._parent) this._parent.chainDidChange(this, this._key, 1);
364 };

366 // get the chains for the current object.  If the current object has 
367 // chains inherited from the proto they will be cloned and reconfigured for
368 // the current object.
369 /** @private */
370 function chainsFor(obj) {
371   var m   = meta(obj), ret = m.chains;
372   if (!ret) {
373     ret = m.chains = new ChainNode(null, null, obj);
374   } else if (ret.value() !== obj) {
375     ret = m.chains = ret.copy(obj);
376   }
377   return ret ;
378 }

381 /** @private */
382 function notifyChains(obj, m, keyName, methodName, arg) {
383   var nodes = m.chainWatchers;
384 
385   if (!nodes || nodes.__emberproto__ !== obj) return; // nothing to do
386 
387   nodes = nodes[keyName];
388   if (!nodes) return;
389 
390   for(var key in nodes) {
391     if (!nodes.hasOwnProperty(key)) continue;
392     nodes[key][methodName](arg);
393   }
394 }

397   notifyChains(obj, m, keyName, 'didChange', true);
398 }

414 var WATCHED_PROPERTY = Ember.SIMPLE_PROPERTY.watched;
415 
416 /**
417   @private
418 
419   Starts watching a property on an object.  Whenever the property changes,
420   invokes Ember.propertyWillChange and Ember.propertyDidChange.  This is the 
421   primitive used by observers and dependent keys; usually you will never call
422   this method directly but instead use higher level methods like
423   Ember.addObserver().
424 */
425 Ember.watch = function(obj, keyName) {
426 
427   // can't watch length on Array - it is special...
428   if (keyName === 'length' && Ember.typeOf(obj)==='array') return this;
429   
430   var m = meta(obj), watching = m.watching, desc;
431   keyName = normalizePath(keyName);
432 
433   // activate watching first time
434   if (!watching[keyName]) {
435     watching[keyName] = 1;
436     if (isKeyName(keyName)) {
437       desc = m.descs[keyName];
438       desc = desc ? desc.watched : WATCHED_PROPERTY;
439       if (desc) Ember.defineProperty(obj, keyName, desc);
440     } else {
441       chainsFor(obj).add(keyName);
442     }
443 
444   }  else {
445     watching[keyName] = (watching[keyName]||0)+1;
446   }
447   return this;
448 };
449 
450 Ember.isWatching = function(obj, keyName) {
451   return !!meta(obj).watching[keyName];
452 };
453 
454 Ember.watch.flushPending = flushPendingChains;
455 
456 /** @private */
457 Ember.unwatch = function(obj, keyName) {
458   // can't watch length on Array - it is special...
459   if (keyName === 'length' && Ember.typeOf(obj)==='array') return this;
460 
461   var watching = meta(obj).watching, desc, descs;
462   keyName = normalizePath(keyName);
463   if (watching[keyName] === 1) {
464     watching[keyName] = 0;
465     if (isKeyName(keyName)) {
466       desc = meta(obj).descs[keyName];
467       desc = desc ? desc.unwatched : SIMPLE_PROPERTY;
468       if (desc) Ember.defineProperty(obj, keyName, desc);
469     } else {
470       chainsFor(obj).remove(keyName);
471     }
472 
473   } else if (watching[keyName]>1) {
474     watching[keyName]--;
475   }
476   
477   return this;
478 };

480 /**
481   @private
482 
483   Call on an object when you first beget it from another object.  This will
484   setup any chained watchers on the object instance as needed.  This method is
485   safe to call multiple times.
486 */
487 Ember.rewatch = function(obj) {
488   var m = meta(obj, false), chains = m.chains, bindings = m.bindings, key, b;
489 
490   // make sure the object has its own guid.
491   if (GUID_KEY in obj && !obj.hasOwnProperty(GUID_KEY)) {
492     Ember.generateGuid(obj, 'ember');
493   }  
494 
495   // make sure any chained watchers update.
496   if (chains && chains.value() !== obj) chainsFor(obj);
497 
498   // if the object has bindings then sync them..
499   if (bindings && m.proto!==obj) {
500     for (key in bindings) {
501       b = !DEP_SKIP[key] && obj[key];
502       if (b && b instanceof Ember.Binding) b.fromDidChange(obj);
503     }
504   }
505 
506   return this;
507 };

513 /**
514   This function is called just before an object property is about to change.
515   It will notify any before observers and prepare caches among other things.
516   
517   Normally you will not need to call this method directly but if for some
518   reason you can't directly watch a property you can invoke this method 
519   manually along with `Ember.propertyDidChange()` which you should call just 
520   after the property value changes.
521   
522   @param {Object} obj
523     The object with the property that will change
524     
525   @param {String} keyName
526     The property key (or path) that will change.
527     
528   @returns {void}
529 */
530 Ember.propertyWillChange = function(obj, keyName) {
531   var m = meta(obj, false), proto = m.proto, desc = m.descs[keyName];
532   if (proto === obj) return ;
533   if (desc && desc.willChange) desc.willChange(obj, keyName);
534   dependentKeysWillChange(obj, keyName, m);
535   chainsWillChange(obj, keyName, m);
536   Ember.notifyBeforeObservers(obj, keyName);
537 };

539 /** @private */
540 var propertyWillChange = Ember.propertyWilLChange;

542 /**
543   This function is called just after an object property has changed.
544   It will notify any observers and clear caches among other things.
545   
546   Normally you will not need to call this method directly but if for some
547   reason you can't directly watch a property you can invoke this method 
548   manually along with `Ember.propertyWilLChange()` which you should call just 
549   before the property value changes.
550   
551   @param {Object} obj
552     The object with the property that will change
553     
554   @param {String} keyName
555     The property key (or path) that will change.
556     
557   @returns {void}
558 */
559 Ember.propertyDidChange = function(obj, keyName) {
560   var m = meta(obj, false), proto = m.proto, desc = m.descs[keyName];
561   if (proto === obj) return ;
562   if (desc && desc.didChange) desc.didChange(obj, keyName);
563   dependentKeysDidChange(obj, keyName, m);
564   chainsDidChange(obj, keyName, m);
565   Ember.notifyObservers(obj, keyName);
566 };

571 var NODE_STACK = []
572 
573 /**
574   Tears down the meta on an object so that it can be garbage collected.
575   Multiple calls will have no effect.
576   
577   @param {Object} obj  the object to destroy
578   @returns {void}
579 */
580 Ember.destroy = function (obj) {
581   var meta = obj[META_KEY], node, nodes, key, nodeObject;
582   if (meta) {
583     obj[META_KEY] = null;
584     // remove chainWatchers to remove circular references that would prevent GC
585     node = meta.chains;
586     if (node) {
587       NODE_STACK.push(node);
588       // process tree
589       while (NODE_STACK.length > 0) {
590         node = NODE_STACK.pop();
591         // push children
592         nodes = node._chains;
593         if (nodes) {
594           for (key in nodes) {
595             if (nodes.hasOwnProperty(key)) {
596               NODE_STACK.push(nodes[key]);
597             }
598           }
599         }
600         // remove chainWatcher in node object
601         if (node._watching) {
602           nodeObject = node._object;
603           if (nodeObject) {
604             removeChainWatcher(nodeObject, node._key, node);
605           }
606         }
607 608       }
609     }
610   }
