  8 require('ember-views/views/states/default');
  9 
 10 var get = Ember.get, set = Ember.set, meta = Ember.meta;
 11 
 12 Ember.View.states.inBuffer = {
 13   parentState: Ember.View.states._default,
 14 
 15   $: function(view, sel) {
 16     // if we don't have an element yet, someone calling this.$() is
 17     // trying to update an element that isn't in the DOM. Instead,
 18     // rerender the view to allow the render method to reflect the
 19     // changes.
 20     view.rerender();
 21     return Ember.$();
 22   },
 23 
 24   // when a view is rendered in a buffer, rerendering it simply
 25   // replaces the existing buffer with a new one
 26   rerender: function(view) {
 27     var buffer = meta(view)['Ember.View'].buffer;
 28 
 29     view.clearRenderedChildren();
 30     view.renderToBuffer(buffer, 'replaceWith');
 31   },
 32 
 33   // when a view is rendered in a buffer, appending a child
 34   // view will render that view and append the resulting
 35   // buffer into its buffer.
 36   appendChild: function(view, childView, options) {
 37     var buffer = meta(view)['Ember.View'].buffer;
 38 
 39     childView = this.createChildView(childView, options);
 40     get(view, '_childViews').pushObject(childView);
 41     childView.renderToBuffer(buffer);
 42     return childView;
 43   },
 44 
 45   // when a view is rendered in a buffer, destroying the
 46   // element will simply destroy the buffer and put the
 47   // state back into the preRender state.
 48   destroyElement: function(view) {
 49     view.clearBuffer();
 50     view._notifyWillDestroyElement();
 51     view.transitionTo('preRender');
 52 
 53     return view;
 54   },
 55 
 56   // It should be impossible for a rendered view to be scheduled for
 57   // insertion.
 58   insertElement: function() {
 59     throw "You can't insert an element that has already been rendered";
 60   },
 61 
 62   setElement: function(view, value) {
 63     view.invalidateRecursively('element');
 64 
 65     if (value === null) {
 66       view.transitionTo('preRender');
 67     } else {
 68       view.clearBuffer();
 69       view.transitionTo('hasElement');
 70     }
 71 
 72     return value;
 73   }
 74 };
