 13 var AFTER_OBSERVERS = ':change';
 14 var BEFORE_OBSERVERS = ':before';
 15 var guidFor = Ember.guidFor;
 16 var normalizePath = Ember.normalizePath;
 17 
 18 var suspended = 0;
 19 var array_Slice = Array.prototype.slice;
 20 
 21 /** @private */
 22 var ObserverSet = function(iterateable) {
 23   this.set = {};
 24   if (iterateable) { this.array = []; }
 25 };
 26 
 27 ObserverSet.prototype.add = function(target, name) {
 28   var set = this.set, guid = Ember.guidFor(target), array;
 29 
 30   if (!set[guid]) { set[guid] = {}; }
 31   set[guid][name] = true;
 32   if (array = this.array) {
 33     array.push([target, name]);
 34   }
 35 };
 36 
 37 ObserverSet.prototype.contains = function(target, name) {
 38   var set = this.set, guid = Ember.guidFor(target), nameSet = set[guid];
 39   return nameSet && nameSet[name];
 40 };
 41 
 42 ObserverSet.prototype.empty = function() {
 43   this.set = {};
 44   this.array = [];
 45 };
 46 
 47 ObserverSet.prototype.forEach = function(fn) {
 48   var q = this.array;
 49   this.empty();
 50   q.forEach(function(item) {
 51     fn(item[0], item[1]);
 52   });
 53 };
 54 
 55 var queue = new ObserverSet(true), beforeObserverSet = new ObserverSet();
 56 
 57 /** @private */
 58 function notifyObservers(obj, eventName, forceNotification) {
 59   if (suspended && !forceNotification) {
 60 
 61     // if suspended add to the queue to send event later - but only send 
 62     // event once.
 63     if (!queue.contains(obj, eventName)) {
 64       queue.add(obj, eventName);
 65     }
 66 
 67   } else {
 68     Ember.sendEvent(obj, eventName);
 69   }
 70 }
 71 
 72 /** @private */
 73 function flushObserverQueue() {
 74   beforeObserverSet.empty();
 75 
 76   if (!queue || queue.array.length===0) return ;
 77   queue.forEach(function(target, event){ Ember.sendEvent(target, event); });
 78 }

 80 Ember.beginPropertyChanges = function() {
 81   suspended++;
 82   return this;
 83 };
 84 
 85 Ember.endPropertyChanges = function() {
 86   suspended--;
 87   if (suspended<=0) flushObserverQueue();
 88 };
 89 
 90 /**
 91   Make a series of property changes together in an
 92   exception-safe way.
 93 
 94       Ember.changeProperties(function() {
 95         obj1.set('foo', mayBlowUpWhenSet);
 96         obj2.set('bar', baz);
 97       });
 98 */
 99 Ember.changeProperties = function(cb){
100   Ember.beginPropertyChanges();
101   try {
102     cb();
103   } finally {
104     Ember.endPropertyChanges();
105   }
106 };

108 /** @private */
109 function changeEvent(keyName) {
110   return keyName+AFTER_OBSERVERS;
111 }
112 
113 /** @private */
114 function beforeEvent(keyName) {
115   return keyName+BEFORE_OBSERVERS;
116 }
117 
118 /** @private */
119 function changeKey(eventName) {
120   return eventName.slice(0, -7);
121 }
122 
123 /** @private */
124 function beforeKey(eventName) {
125   return eventName.slice(0, -7);
126 }
127 
128 /** @private */
129 function xformForArgs(args) {
130   return function (target, method, params) {
131     var obj = params[0], keyName = changeKey(params[1]), val;
132     var copy_args = args.slice();
133     if (method.length>2) {
134       val = Ember.getPath(Ember.isGlobalPath(keyName) ? window : obj, keyName);
135     }
136     copy_args.unshift(obj, keyName, val);
137     method.apply(target, copy_args);
138   };
139 }
140 
141 var xformChange = xformForArgs([]);
142 
143 /** @private */
144 function xformBefore(target, method, params) {
145   var obj = params[0], keyName = beforeKey(params[1]), val;
146   if (method.length>2) val = Ember.getPath(obj, keyName);
147   method.call(target, obj, keyName, val);
148 }

150 Ember.addObserver = function(obj, path, target, method) {
151   path = normalizePath(path);
152 
153   var xform;
154   if (arguments.length > 4) {
155     var args = array_Slice.call(arguments, 4);
156     xform = xformForArgs(args);
157   } else {
158     xform = xformChange;
159   }
160   Ember.addListener(obj, changeEvent(path), target, method, xform);
161   Ember.watch(obj, path);
162   return this;
163 };
164 
165 /** @private */
166 Ember.observersFor = function(obj, path) {
167   return Ember.listenersFor(obj, changeEvent(path));
168 };
169 
170 Ember.removeObserver = function(obj, path, target, method) {
171   path = normalizePath(path);
172   Ember.unwatch(obj, path);
173   Ember.removeListener(obj, changeEvent(path), target, method);
174   return this;
175 };
176 
177 Ember.addBeforeObserver = function(obj, path, target, method) {
178   path = normalizePath(path);
179   Ember.addListener(obj, beforeEvent(path), target, method, xformBefore);
180   Ember.watch(obj, path);
181   return this;
182 };
183 
184 /** @private */
185 Ember.beforeObserversFor = function(obj, path) {
186   return Ember.listenersFor(obj, beforeEvent(path));
187 };
188 
189 Ember.removeBeforeObserver = function(obj, path, target, method) {
190   path = normalizePath(path);
191   Ember.unwatch(obj, path);
192   Ember.removeListener(obj, beforeEvent(path), target, method);
193   return this;
194 };

196 /** @private */
197 Ember.notifyObservers = function(obj, keyName) {
198   notifyObservers(obj, changeEvent(keyName));
199 };
200 
201 /** @private */
202 Ember.notifyBeforeObservers = function(obj, keyName) {
203   var guid, set, forceNotification = false;
204 
205   if (suspended) {
206     if (!beforeObserverSet.contains(obj, keyName)) {
207       beforeObserverSet.add(obj, keyName);
208       forceNotification = true;
209     } else {
210       return;
211     }
212   }
213 
214   notifyObservers(obj, beforeEvent(keyName), forceNotification);
215 };
