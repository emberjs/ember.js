 10 /**
 11   @class
 12 
 13   Platform specific methods and feature detectors needed by the framework.
 14 
 15   @name Ember.platform
 16 */
 17 var platform = Ember.platform = {} ;
 18 
 19 /**
 20   Identical to Object.create().  Implements if not available natively.
 21   @memberOf Ember.platform
 22   @name create
 23 */
 24 platform.create = Object.create;
 25 
 26 if (!platform.create) {
 27   /** @private */
 28   var O_ctor = function() {},
 29       O_proto = O_ctor.prototype;
 30 
 31   platform.create = function(obj, descs) {
 32     O_ctor.prototype = obj;
 33     obj = new O_ctor();
 34     O_ctor.prototype = O_proto;
 35 
 36     if (descs !== undefined) {
 37       for(var key in descs) {
 38         if (!descs.hasOwnProperty(key)) continue;
 39         platform.defineProperty(obj, key, descs[key]);
 40       }
 41     }
 42 
 43     return obj;
 44   };
 45 
 46   platform.create.isSimulated = true;
 47 }
 48 
 49 /** @private */
 50 var defineProperty = Object.defineProperty;
 51 var canRedefineProperties, canDefinePropertyOnDOM;
 52 
 53 // Catch IE8 where Object.defineProperty exists but only works on DOM elements
 54 if (defineProperty) {
 55   try {
 56     defineProperty({}, 'a',{get:function(){}});
 57   } catch (e) {
 58     /** @private */
 59     defineProperty = null;
 60   }
 61 }
 62 
 63 if (defineProperty) {
 64   // Detects a bug in Android <3.2 where you cannot redefine a property using
 65   // Object.defineProperty once accessors have already been set.
 66   /** @private */
 67   canRedefineProperties = (function() {
 68     var obj = {};
 69 
 70     defineProperty(obj, 'a', {
 71       configurable: true,
 72       enumerable: true,
 73       get: function() { },
 74       set: function() { }
 75     });
 76 
 77     defineProperty(obj, 'a', {
 78       configurable: true,
 79       enumerable: true,
 80       writable: true,
 81       value: true
 82     });
 83 
 84     return obj.a === true;
 85   })();
 86 
 87   // This is for Safari 5.0, which supports Object.defineProperty, but not
 88   // on DOM nodes.
 89   /** @private */
 90   canDefinePropertyOnDOM = (function(){
 91     try {
 92       defineProperty(document.body, 'definePropertyOnDOM', {});
 93       return true;
 94     } catch(e) { }
 95 
 96     return false;
 97   })();
 98 
 99   if (!canRedefineProperties) {
100     /** @private */
101     defineProperty = null;
102   } else if (!canDefinePropertyOnDOM) {
103     /** @private */
104     defineProperty = function(obj, keyName, desc){
105       var isNode;
106 
107       if (typeof Node === "object") {
108         isNode = obj instanceof Node;
109       } else {
110         isNode = typeof obj === "object" && typeof obj.nodeType === "number" && typeof obj.nodeName === "string";
111       }
112 
113       if (isNode) {
114         // TODO: Should we have a warning here?
115         return (obj[keyName] = desc.value);
116       } else {
117         return Object.defineProperty(obj, keyName, desc);
118       }
119     };
120   }
121 }
122 
123 /**
124   Identical to Object.defineProperty().  Implements as much functionality
125   as possible if not available natively.
126 
127   @memberOf Ember.platform
128   @name defineProperty
129   @param {Object} obj The object to modify
130   @param {String} keyName property name to modify
131   @param {Object} desc descriptor hash
132   @returns {void}
133 */
134 platform.defineProperty = defineProperty;
135 
136 /**
137   Set to true if the platform supports native getters and setters.
138 
139   @memberOf Ember.platform
140   @name hasPropertyAccessors
141 */
142 platform.hasPropertyAccessors = true;
143 
144 if (!platform.defineProperty) {
145   platform.hasPropertyAccessors = false;
146 
147   platform.defineProperty = function(obj, keyName, desc) {
148     ember_assert("property descriptor cannot have `get` or `set` on this platform", !desc.get && !desc.set);
149     obj[keyName] = desc.value;
150   };
151 
152   platform.defineProperty.isSimulated = true;
153 }
