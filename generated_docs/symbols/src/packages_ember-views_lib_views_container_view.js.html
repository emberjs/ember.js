  8 require('ember-views/views/view');
  9 var get = Ember.get, set = Ember.set, meta = Ember.meta;
 10 
 11 var childViewsProperty = Ember.computed(function() {
 12   return get(this, '_childViews');
 13 }).property('_childViews').cacheable();
 14 
 15 Ember.ContainerView = Ember.View.extend({
 16 
 17   init: function() {
 18     var childViews = get(this, 'childViews');
 19     Ember.defineProperty(this, 'childViews', childViewsProperty);
 20 
 21     this._super();
 22 
 23     var _childViews = get(this, '_childViews');
 24 
 25     childViews.forEach(function(viewName, idx) {
 26       var view;
 27 
 28       if ('string' === typeof viewName) {
 29         view = get(this, viewName);
 30         view = this.createChildView(view);
 31         set(this, viewName, view);
 32       } else {
 33         view = this.createChildView(viewName);
 34       }
 35 
 36       _childViews[idx] = view;
 37     }, this);
 38 
 39     // Sets up an array observer on the child views array. This
 40     // observer will detect when child views are added or removed
 41     // and update the DOM to reflect the mutation.
 42     get(this, 'childViews').addArrayObserver(this, {
 43       willChange: 'childViewsWillChange',
 44       didChange: 'childViewsDidChange'
 45     });
 46   },
 47 
 48   /**
 49     Instructs each child view to render to the passed render buffer.
 50 
 51     @param {Ember.RenderBuffer} buffer the buffer to render to
 52     @private
 53   */
 54   render: function(buffer) {
 55     this.forEachChildView(function(view) {
 56       view.renderToBuffer(buffer);
 57     });
 58   },
 59 
 60   /**
 61     When the container view is destroyed, tear down the child views
 62     array observer.
 63 
 64     @private
 65   */
 66   destroy: function() {
 67     get(this, 'childViews').removeArrayObserver(this, {
 68       willChange: 'childViewsWillChange',
 69       didChange: 'childViewsDidChange'
 70     });
 71 
 72     this._super();
 73   },
 74 
 75   /**
 76     When a child view is removed, destroy its element so that
 77     it is removed from the DOM.
 78 
 79     The array observer that triggers this action is set up in the
 80     `renderToBuffer` method.
 81 
 82     @private
 83     @param {Ember.Array} views the child views array before mutation
 84     @param {Number} start the start position of the mutation
 85     @param {Number} removed the number of child views removed
 86   **/
 87   childViewsWillChange: function(views, start, removed) {
 88     if (removed === 0) { return; }
 89 
 90     var changedViews = views.slice(start, start+removed);
 91     this.setParentView(changedViews, null);
 92 
 93     this.invokeForState('childViewsWillChange', views, start, removed);
 94   },
 95 
 96   /**
 97     When a child view is added, make sure the DOM gets updated appropriately.
 98 
 99     If the view has already rendered an element, we tell the child view to
100     create an element and insert it into the DOM. If the enclosing container view
101     has already written to a buffer, but not yet converted that buffer into an
102     element, we insert the string representation of the child into the appropriate
103     place in the buffer.
104 
105     @private
106     @param {Ember.Array} views the array of child views afte the mutation has occurred
107     @param {Number} start the start position of the mutation
108     @param {Number} removed the number of child views removed
109     @param {Number} the number of child views added
110   */
111   childViewsDidChange: function(views, start, removed, added) {
112     var len = get(views, 'length');
113 
114     // No new child views were added; bail out.
115     if (added === 0) return;
116 
117     var changedViews = views.slice(start, start+added);
118     this.setParentView(changedViews, this);
119 
120     // Let the current state handle the changes
121     this.invokeForState('childViewsDidChange', views, start, added);
122   },
123 
124   setParentView: function(views, parentView) {
125     views.forEach(function(view) {
126       set(view, '_parentView', parentView);
127     });
128   },
129 
130   /**
131     Schedules a child view to be inserted into the DOM after bindings have
132     finished syncing for this run loop.
133 
134     @param {Ember.View} view the child view to insert
135     @param {Ember.View} prev the child view after which the specified view should
136                      be inserted
137     @private
138   */
139   _scheduleInsertion: function(view, prev) {
140     if (prev) {
141       prev.get('domManager').after(view);
142     } else {
143       this.get('domManager').prepend(view);
144     }
145   }
146 });
147 
148 // Ember.ContainerView extends the default view states to provide different
149 // behavior for childViewsWillChange and childViewsDidChange.
150 Ember.ContainerView.states = {
151   parent: Ember.View.states,
152 
153   inBuffer: {
154     childViewsDidChange: function(parentView, views, start, added) {
155       var buffer = meta(parentView)['Ember.View'].buffer,
156           startWith, prev, prevBuffer, view;
157 
158       // Determine where to begin inserting the child view(s) in the
159       // render buffer.
160       if (start === 0) {
161         // If views were inserted at the beginning, prepend the first
162         // view to the render buffer, then begin inserting any
163         // additional views at the beginning.
164         view = views[start];
165         startWith = start + 1;
166         view.renderToBuffer(buffer, 'prepend');
167       } else {
168         // Otherwise, just insert them at the same place as the child
169         // views mutation.
170         view = views[start - 1];
171         startWith = start;
172       }
173 
174       for (var i=startWith; i<start+added; i++) {
175         prev = view;
176         view = views[i];
177         prevBuffer = meta(prev)['Ember.View'].buffer;
178         view.renderToBuffer(prevBuffer, 'insertAfter');
179       }
180     }
181   },
182 
183   hasElement: {
184     childViewsWillChange: function(view, views, start, removed) {
185       for (var i=start; i<start+removed; i++) {
186         views[i].remove();
187       }
188     },
189 
190     childViewsDidChange: function(view, views, start, added) {
191       // If the DOM element for this container view already exists,
192       // schedule each child view to insert its DOM representation after
193       // bindings have finished syncing.
194       var prev = start === 0 ? null : views[start-1];
195 
196       for (var i=start; i<start+added; i++) {
197         view = views[i];
198         this._scheduleInsertion(view, prev);
199         prev = view;
200       }
201     }
202   }
203 };
204 
205 Ember.ContainerView.states.inDOM = {
206   parentState: Ember.ContainerView.states.hasElement
207 };
208 
209 Ember.ContainerView.reopen({
210   states: Ember.ContainerView.states
211 });
