  8 var get = Ember.get, set = Ember.set, fmt = Ember.String.fmt;
  9 
 10 /**
 11   @ignore
 12 
 13   Ember.EventDispatcher handles delegating browser events to their corresponding
 14   Ember.Views. For example, when you click on a view, Ember.EventDispatcher ensures
 15   that that view's `mouseDown` method gets called.
 16 */
 17 Ember.EventDispatcher = Ember.Object.extend(
 18 /** @scope Ember.EventDispatcher.prototype */{
 19 
 20   /**
 21     @private
 22 
 23     The root DOM element to which event listeners should be attached. Event
 24     listeners will be attached to the document unless this is overridden.
 25 
 26     Can be specified as a DOMElement or a selector string.
 27 
 28     The default body is a string since this may be evaluated before document.body
 29     exists in the DOM.
 30 
 31     @type DOMElement
 32     @default 'body'
 33   */
 34   rootElement: 'body',
 35 
 36   /**
 37     @private
 38 
 39     Sets up event listeners for standard browser events.
 40 
 41     This will be called after the browser sends a DOMContentReady event. By
 42     default, it will set up all of the listeners on the document body. If you
 43     would like to register the listeners on different element, set the event
 44     dispatcher's `root` property.
 45   */
 46   setup: function(addedEvents) {
 47     var event, events = {
 48       touchstart  : 'touchStart',
 49       touchmove   : 'touchMove',
 50       touchend    : 'touchEnd',
 51       touchcancel : 'touchCancel',
 52       keydown     : 'keyDown',
 53       keyup       : 'keyUp',
 54       keypress    : 'keyPress',
 55       mousedown   : 'mouseDown',
 56       mouseup     : 'mouseUp',
 57       click       : 'click',
 58       dblclick    : 'doubleClick',
 59       mousemove   : 'mouseMove',
 60       focusin     : 'focusIn',
 61       focusout    : 'focusOut',
 62       mouseenter  : 'mouseEnter',
 63       mouseleave  : 'mouseLeave',
 64       submit      : 'submit',
 65       change      : 'change',
 66       dragstart   : 'dragStart',
 67       drag        : 'drag',
 68       dragenter   : 'dragEnter',
 69       dragleave   : 'dragLeave',
 70       dragover    : 'dragOver',
 71       drop        : 'drop',
 72       dragend     : 'dragEnd'
 73     };
 74 
 75     jQuery.extend(events, addedEvents || {});
 76 
 77     var rootElement = Ember.$(get(this, 'rootElement'));
 78 
 79     ember_assert(fmt('You cannot use the same root element (%@) multiple times in an Ember.Application', [rootElement.selector || rootElement[0].tagName]), !rootElement.is('.ember-application'));
 80     ember_assert('You cannot make a new Ember.Application using a root element that is a descendent of an existing Ember.Application', !rootElement.closest('.ember-application').length);
 81     ember_assert('You cannot make a new Ember.Application using a root element that is an ancestor of an existing Ember.Application', !rootElement.find('.ember-application').length);
 82 
 83     rootElement.addClass('ember-application');
 84 
 85     ember_assert('Unable to add "ember-application" class to rootElement. Make sure you the body or an element in the body.', rootElement.is('.ember-application'));
 86 
 87     for (event in events) {
 88       if (events.hasOwnProperty(event)) {
 89         this.setupHandler(rootElement, event, events[event]);
 90       }
 91     }
 92   },
 93 
 94   /**
 95     @private
 96 
 97     Registers an event listener on the document. If the given event is
 98     triggered, the provided event handler will be triggered on the target
 99     view.
100 
101     If the target view does not implement the event handler, or if the handler
102     returns false, the parent view will be called. The event will continue to
103     bubble to each successive parent view until it reaches the top.
104 
105     For example, to have the `mouseDown` method called on the target view when
106     a `mousedown` event is received from the browser, do the following:
107 
108         setupHandler('mousedown', 'mouseDown');
109 
110     @param {String} event the browser-originated event to listen to
111     @param {String} eventName the name of the method to call on the view
112   */
113   setupHandler: function(rootElement, event, eventName) {
114     var self = this;
115 
116     rootElement.delegate('.ember-view', event + '.ember', function(evt, triggeringManager) {
117 
118       var view = Ember.View.views[this.id],
119           result = true, manager = null;
120 
121       manager = self._findNearestEventManager(view,eventName);
122 
123       if (manager && manager !== triggeringManager) {
124         result = self._dispatchEvent(manager, evt, eventName, view);
125       } else if (view) {
126         result = self._bubbleEvent(view,evt,eventName);
127       } else {
128         evt.stopPropagation();
129       }
130 
131       return result;
132     });
133   },
134 
135   /** @private */
136   _findNearestEventManager: function(view, eventName) {
137     var manager = null;
138 
139     while (view) {
140       manager = get(view, 'eventManager');
141       if (manager && manager[eventName]) { break; }
142 
143       view = get(view, 'parentView');
144     }
145 
146     return manager;
147   },
148 
149   /** @private */
150   _dispatchEvent: function(object, evt, eventName, view) {
151     var result = true;
152 
153     handler = object[eventName];
154     if (Ember.typeOf(handler) === 'function') {
155       result = handler.call(object, evt, view);
156       evt.stopPropagation();
157     }
158     else {
159       result = this._bubbleEvent(view, evt, eventName);
160     }
161 
162     return result;
163   },
164 
165   /** @private */
166   _bubbleEvent: function(view, evt, eventName) {
167     return Ember.run(function() {
168       return view.handleEvent(eventName, evt);
169     });
170   },
171 
172   /** @private */
173   destroy: function() {
174     var rootElement = get(this, 'rootElement');
175     Ember.$(rootElement).undelegate('.ember').removeClass('ember-application');
176     return this._super();
177   }
178 });
