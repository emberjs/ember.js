  6 /*globals Handlebars */

  8 require('ember-handlebars/ext');
  9 
 10 var getPath = Ember.getPath;
 11 
 12 /**
 13   `log` allows you to output the value of a value in the current rendering
 14   context.
 15 
 16     {{log myVariable}}
 17 
 18   @name Handlebars.helpers.log
 19   @param {String} property
 20 */
 21 Ember.Handlebars.registerHelper('log', function(property, fn) {
 22   var context = (fn.contexts && fn.contexts[0]) || this;
 23   Ember.Logger.log(getPath(context, property));
 24 });
 25 
 26 /**
 27   The `debugger` helper executes the `debugger` statement in the current
 28   context.
 29 
 30     {{debugger}}
 31 
 32   @name Handlebars.helpers.debugger
 33   @param {String} property
 34 */
 35 Ember.Handlebars.registerHelper('debugger', function() {
 36   debugger;
 37 });
