 14 var rewatch = Ember.rewatch;
 15 var classToString = Ember.Mixin.prototype.toString;
 16 var set = Ember.set, get = Ember.get;
 17 var o_create = Ember.platform.create,
 18     meta = Ember.meta;
 19 
 20 /** @private */
 21 function makeCtor() {
 22 
 23   // Note: avoid accessing any properties on the object since it makes the
 24   // method a lot faster.  This is glue code so we want it to be as fast as
 25   // possible.
 26 
 27   var isPrepared = false, initMixins, init = false, hasChains = false;
 28 
 29   var Class = function() {
 30     if (!isPrepared) { get(Class, 'proto'); } // prepare prototype...
 31     if (initMixins) {
 32       this.reopen.apply(this, initMixins);
 33       initMixins = null;
 34       rewatch(this); // ålways rewatch just in case
 35       this.init.apply(this, arguments);
 36     } else {
 37       if (hasChains) {
 38         rewatch(this);
 39       } else {
 40         this[Ember.GUID_KEY] = undefined;
 41       }
 42       if (init===false) { init = this.init; } // cache for later instantiations
 43       init.apply(this, arguments);
 44     }
 45   };
 46 
 47   Class.toString = classToString;
 48   Class._prototypeMixinDidChange = function() { isPrepared = false; };
 49   Class._initMixins = function(args) { initMixins = args; };
 50 
 51   Ember.defineProperty(Class, 'proto', Ember.computed(function() {
 52     if (!isPrepared) {
 53       isPrepared = true;
 54       Class.PrototypeMixin.applyPartial(Class.prototype);
 55       hasChains = !!meta(Class.prototype, false).chains; // avoid rewatch
 56     }
 57     return this.prototype;
 58   }));
 59 
 60   return Class;
 61 
 62 }
 63 
 64 var CoreObject = makeCtor();
 65 
 66 CoreObject.PrototypeMixin = Ember.Mixin.create(
 67 /** @scope Ember.CoreObject */ {
 68 
 69   reopen: function() {
 70     Ember.Mixin._apply(this, arguments, true);
 71     return this;
 72   },
 73 
 74   isInstance: true,
 75 
 76   init: function() {},
 77 
 78   isDestroyed: false,
 79 
 80   /**
 81     Destroys an object by setting the isDestroyed flag and removing its
 82     metadata, which effectively destroys observers and bindings.
 83 
 84     If you try to set a property on a destroyed object, an exception will be
 85     raised.
 86 
 87     Note that destruction is scheduled for the end of the run loop and does not
 88     happen immediately.
 89 
 90     @returns {Ember.Object} receiver
 91   */
 92   destroy: function() {
 93     set(this, 'isDestroyed', true);
 94     Ember.run.schedule('destroy', this, this._scheduledDestroy);
 95     return this;
 96   },
 97 
 98   /**
 99     Invoked by the run loop to actually destroy the object. This is
100     scheduled for execution by the `destroy` method.
101 
102     @private
103   */
104   _scheduledDestroy: function() {
105     Ember.destroy(this);
106   },
107 
108   bind: function(to, from) {
109     if (!(from instanceof Ember.Binding)) { from = Ember.Binding.from(from); }
110     from.to(to).connect(this);
111     return from;
112   },
113 
114   toString: function() {
115     return '<'+this.constructor.toString()+':'+Ember.guidFor(this)+'>';
116   }
117 });
118 
119 CoreObject.__super__ = null;
120 
121 var ClassMixin = Ember.Mixin.create({
122 
123   ClassMixin: Ember.required(),
124 
125   PrototypeMixin: Ember.required(),
126 
127   isClass: true,
128 
129   isMethod: false,
130 
131   extend: function() {
132     var Class = makeCtor(), proto;
133     Class.ClassMixin = Ember.Mixin.create(this.ClassMixin);
134     Class.PrototypeMixin = Ember.Mixin.create(this.PrototypeMixin);
135 
136     Class.ClassMixin.ownerConstructor = Class;
137     Class.PrototypeMixin.ownerConstructor = Class;
138 
139     var PrototypeMixin = Class.PrototypeMixin;
140     PrototypeMixin.reopen.apply(PrototypeMixin, arguments);
141 
142     Class.superclass = this;
143     Class.__super__  = this.prototype;
144 
145     proto = Class.prototype = o_create(this.prototype);
146     proto.constructor = Class;
147     Ember.generateGuid(proto, 'ember');
148     meta(proto).proto = proto; // this will disable observers on prototype
149     Ember.rewatch(proto); // setup watch chains if needed.
150 
151 
152     Class.subclasses = Ember.Set ? new Ember.Set() : null;
153     if (this.subclasses) { this.subclasses.add(Class); }
154 
155     Class.ClassMixin.apply(Class);
156     return Class;
157   },
158 
159   create: function() {
160     var C = this;
161     if (arguments.length>0) { this._initMixins(arguments); }
162     return new C();
163   },
164 
165   reopen: function() {
166     var PrototypeMixin = this.PrototypeMixin;
167     PrototypeMixin.reopen.apply(PrototypeMixin, arguments);
168     this._prototypeMixinDidChange();
169     return this;
170   },
171 
172   reopenClass: function() {
173     var ClassMixin = this.ClassMixin;
174     ClassMixin.reopen.apply(ClassMixin, arguments);
175     Ember.Mixin._apply(this, arguments, false);
176     return this;
177   },
178 
179   detect: function(obj) {
180     if ('function' !== typeof obj) { return false; }
181     while(obj) {
182       if (obj===this) { return true; }
183       obj = obj.superclass;
184     }
185     return false;
186   },
187 
188   detectInstance: function(obj) {
189     return this.PrototypeMixin.detect(obj);
190   },
191 
192   /**
193     In some cases, you may want to annotate computed properties with additional
194     metadata about how they function or what values they operate on. For example,
195     computed property functions may close over variables that are then no longer
196     available for introspection.
197 
198     You can pass a hash of these values to a computed property like this:
199 
200         person: function() {
201           var personId = this.get('personId');
202           return App.Person.create({ id: personId });
203         }.property().meta({ type: App.Person })
204 
205     Once you've done this, you can retrieve the values saved to the computed
206     property from your class like this:
207 
208         MyClass.metaForProperty('person');
209 
210     This will return the original hash that was passed to `meta()`.
211   */
212   metaForProperty: function(key) {
213     var desc = meta(get(this, 'proto'), false).descs[key];
214 
215     ember_assert("metaForProperty() could not find a computed property with key '"+key+"'.", !!desc && desc instanceof Ember.ComputedProperty);
216     return desc._meta || {};
217   }
218 
219 });
220 
221 CoreObject.ClassMixin = ClassMixin;
222 ClassMixin.apply(CoreObject);
223 
224 /**
225   @class
226 */
227 Ember.CoreObject = CoreObject;
