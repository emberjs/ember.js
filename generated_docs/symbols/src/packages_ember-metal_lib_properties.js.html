 53 /**
 54   Called whenever we want to set the property value.  Should set the value 
 55   and return the actual set value (which is usually the same but may be 
 56   different in the case of computed properties.)
 57   
 58   @param {Object} obj
 59     The object to set the value on.
 60     
 61   @param {String} keyName
 62     The key to set.
 63     
 64   @param {Object} value
 65     The new value
 66     
 67   @returns {Object} value actual set value
 68 */
 69 Dp.set = function(obj, keyName, value) {
 70   obj[keyName] = value;
 71   return value;
 72 };
 73 
 74 /**
 75   Called whenever we want to get the property value.  Should retrieve the 
 76   current value.
 77   
 78   @param {Object} obj
 79     The object to get the value on.
 80     
 81   @param {String} keyName
 82     The key to retrieve
 83     
 84   @returns {Object} the current value
 85 */
 86 Dp.get = function(obj, keyName) {
 87   return w_get(obj, keyName, obj);
 88 };

 90 /**
 91   This is called on the descriptor to set it up on the object.  The 
 92   descriptor is responsible for actually defining the property on the object
 93   here.
 94   
 95   The passed `value` is the transferValue returned from any previous 
 96   descriptor.
 97   
 98   @param {Object} obj
 99     The object to set the value on.
100     
101   @param {String} keyName
102     The key to set.
103     
104   @param {Object} value
105     The transfer value from any previous descriptor.
106   
107   @returns {void}
108 */
109 Dp.setup = setup;
110 
111 /**
112   This is called on the descriptor just before another descriptor takes its
113   place.  This method should at least return the 'transfer value' of the 
114   property - which is the value you want to passed as the input to the new
115   descriptor's setup() method.  
116   
117   It is not generally necessary to actually 'undefine' the property as a new
118   property descriptor will redefine it immediately after this method returns.
119   
120   @param {Object} obj
121     The object to set the value on.
122     
123   @param {String} keyName
124     The key to set.
125     
126   @returns {Object} transfer value
127 */
128 Dp.teardown = function(obj, keyName) {
129   return obj[keyName];
130 };

132 Dp.val = function(obj, keyName) {
133   return obj[keyName];

138 // 
139 
140 // if accessors are disabled for the app then this will act as a guard when
141 // testing on browsers that do support accessors.  It will throw an exception
142 // if you do foo.bar instead of Ember.get(foo, 'bar')
143 
144 // The exception to this is that any objects managed by Ember but not a descendant
145 // of Ember.Object will not throw an exception, instead failing silently. This
146 // prevent errors with other libraries that may attempt to access special
147 // properties on standard objects like Array. Usually this happens when copying
148 // an object by looping over all properties.
149 
150 if (!USE_ACCESSORS) {
151   Ember.Descriptor.MUST_USE_GETTER = function() {
152     if (this instanceof Ember.Object) {
153       ember_assert('Must use Ember.get() to access this property', false);
154     }
155   };
156 
157   Ember.Descriptor.MUST_USE_SETTER = function() {
158     if (this instanceof Ember.Object) {
159       if (this.isDestroyed) {
160         ember_assert('You cannot set observed properties on destroyed objects', false);
161       } else {
162         ember_assert('Must use Ember.set() to access this property', false);
163       }
164     }
165   };
166 }

169   configurable: true,
170   enumerable:   true,
171   set: Ember.Descriptor.MUST_USE_SETTER

174 /** @private */
175 function w_get(obj, keyName, values) {
176   values = values || meta(obj, false).values;
177 
178   if (values) {
179     var ret = values[keyName];
180     if (ret !== undefined) { return ret; }
181     if (obj.unknownProperty) { return obj.unknownProperty(keyName); }
182   }
183 
184 }
185 
186 /** @private */
187 function w_set(obj, keyName, value) {
188   var m = meta(obj), watching;
189   
190   watching = m.watching[keyName]>0 && value!==m.values[keyName];  
191   if (watching) Ember.propertyWillChange(obj, keyName);
192   m.values[keyName] = value;
193   if (watching) Ember.propertyDidChange(obj, keyName);
194   return value;
195 }
196 
197 var WATCHED_GETTERS = {};
198 /** @private */
199 function mkWatchedGetter(keyName) {
200   var ret = WATCHED_GETTERS[keyName];
201   if (!ret) {
202     ret = WATCHED_GETTERS[keyName] = function() { 
203       return w_get(this, keyName); 
204     };
205   }
206   return ret;
207 }
208 
209 var WATCHED_SETTERS = {};
210 /** @private */
211 function mkWatchedSetter(keyName) {
212   var ret = WATCHED_SETTERS[keyName];
213   if (!ret) {
214     ret = WATCHED_SETTERS[keyName] = function(value) {
215       return w_set(this, keyName, value);
216     };
217   }
218   return ret;
219 }

221 /**
222   @private 
223   
224   Private version of simple property that invokes property change callbacks.
225 */
226 WATCHED_PROPERTY = new Ember.Descriptor();
227 
228 if (Ember.platform.hasPropertyAccessors) {
229   WATCHED_PROPERTY.get = w_get ;
230   WATCHED_PROPERTY.set = w_set ;
231 
232   if (USE_ACCESSORS) {
233     WATCHED_PROPERTY.setup = function(obj, keyName, value) {
234       WATCHED_DESC.get = mkWatchedGetter(keyName);
235       WATCHED_DESC.set = mkWatchedSetter(keyName);
236       o_defineProperty(obj, keyName, WATCHED_DESC);
237       WATCHED_DESC.get = WATCHED_DESC.set = null;
238       if (value !== undefined) meta(obj).values[keyName] = value;
239     };
240 
241   } else {
242     WATCHED_PROPERTY.setup = function(obj, keyName, value) {
243       WATCHED_DESC.get = mkWatchedGetter(keyName);
244       o_defineProperty(obj, keyName, WATCHED_DESC);
245       WATCHED_DESC.get = null;
246       if (value !== undefined) meta(obj).values[keyName] = value;
247     };
248   }
249 
250   WATCHED_PROPERTY.teardown = function(obj, keyName) {
251     var ret = meta(obj).values[keyName];
252     delete meta(obj).values[keyName];
253     return ret;
254   };
255 
256 // NOTE: if platform does not have property accessors then we just have to 
257 // set values and hope for the best.  You just won't get any warnings...
258 } else {
259   
260   WATCHED_PROPERTY.set = function(obj, keyName, value) {
261     var m = meta(obj), watching;
262 
263     watching = m.watching[keyName]>0 && value!==obj[keyName];  
264     if (watching) Ember.propertyWillChange(obj, keyName);
265     obj[keyName] = value;
266     if (watching) Ember.propertyDidChange(obj, keyName);
267     return value;
268   };

291 /** @private */
292 function hasDesc(descs, keyName) {
293   if (keyName === 'toString') return 'function' !== typeof descs.toString;
294   else return !!descs[keyName];
295 }
296 
297 /**
298   @private
299 
300   NOTE: This is a low-level method used by other parts of the API.  You almost
301   never want to call this method directly.  Instead you should use Ember.mixin()
302   to define new properties.
303   
304   Defines a property on an object.  This method works much like the ES5 
305   Object.defineProperty() method except that it can also accept computed 
306   properties and other special descriptors. 
307 
308   Normally this method takes only three parameters.  However if you pass an
309   instance of Ember.Descriptor as the third param then you can pass an optional
310   value as the fourth parameter.  This is often more efficient than creating
311   new descriptor hashes for each property.
312   
313   ## Examples
314 
315       // ES5 compatible mode
316       Ember.defineProperty(contact, 'firstName', {
317         writable: true,
318         configurable: false,
319         enumerable: true,
320         value: 'Charles'
321       });
322       
323       // define a simple property
324       Ember.defineProperty(contact, 'lastName', Ember.SIMPLE_PROPERTY, 'Jolley');
325       
326       // define a computed property
327       Ember.defineProperty(contact, 'fullName', Ember.computed(function() {
328         return this.firstName+' '+this.lastName;
329       }).property('firstName', 'lastName').cacheable());
330 */
331 Ember.defineProperty = function(obj, keyName, desc, val) {
332   var m = meta(obj, false), descs = m.descs, watching = m.watching[keyName]>0, override = true;
333 
334   if (val === undefined) {
335     override = false;
336     val = hasDesc(descs, keyName) ? descs[keyName].teardown(obj, keyName) : obj[keyName];
337   } else if (hasDesc(descs, keyName)) {
338     descs[keyName].teardown(obj, keyName);
339   }
340 
341   if (!desc) desc = SIMPLE_PROPERTY;
342   
343   if (desc instanceof Ember.Descriptor) {
344     m = meta(obj, true);
345     descs = m.descs;
346     
347     desc = (watching ? desc.watched : desc.unwatched) || desc; 
348     descs[keyName] = desc;
349     desc.setup(obj, keyName, val, watching);
350 
351   // compatibility with ES5
352   } else {
353     if (descs[keyName]) meta(obj).descs[keyName] = null;
354     o_defineProperty(obj, keyName, desc);
355   }
356 
357   // if key is being watched, override chains that
358   // were initialized with the prototype
359   if (override && watching) Ember.overrideChains(obj, keyName, m);
360 
361   return this;
362 };

364 /**
365   Creates a new object using the passed object as its prototype.  On browsers
366   that support it, this uses the built in Object.create method.  Else one is
367   simulated for you.
368   
369   This method is a better choice thant Object.create() because it will make 
370   sure that any observers, event listeners, and computed properties are 
371   inherited from the parent as well.
372   
373   @param {Object} obj
374     The object you want to have as the prototype.
375     
376   @returns {Object} the newly created object
377 */
378 Ember.create = function(obj, props) {
379   var ret = o_create(obj, props);
380   if (GUID_KEY in ret) Ember.generateGuid(ret, 'ember');
381   if (META_KEY in ret) Ember.rewatch(ret); // setup watch chains if needed.
382   return ret;
383 };
384 
385 /**
386   @private
387 
388   Creates a new object using the passed object as its prototype.  This method
389   acts like `Ember.create()` in every way except that bindings, observers, and
390   computed properties will be activated on the object.  
391   
392   The purpose of this method is to build an object for use in a prototype
393   chain. (i.e. to be set as the `prototype` property on a constructor 
394   function).  Prototype objects need to inherit bindings, observers and
395   other configuration so they pass it on to their children.  However since
396   they are never 'live' objects themselves, they should not fire or make
397   other changes when various properties around them change.
398   
399   You should use this method anytime you want to create a new object for use
400   in a prototype chain.
401 
402   @param {Object} obj
403     The base object.
404 
405   @param {Object} hash
406     Optional hash of properties to define on the object.
407 
408   @returns {Object} new object
409 */
410 Ember.createPrototype = function(obj, props) {
411   var ret = o_create(obj, props);
412   meta(ret, true).proto = ret;
413   if (GUID_KEY in ret) Ember.generateGuid(ret, 'ember');
414   if (META_KEY in ret) Ember.rewatch(ret); // setup watch chains if needed.
415   return ret;
416 };
