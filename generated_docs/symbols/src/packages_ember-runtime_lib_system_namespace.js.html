  7 require('ember-runtime/system/object');
  8 
  9 /**
 10   @private
 11   A Namespace is an object usually used to contain other objects or methods
 12   such as an application or framework.  Create a namespace anytime you want
 13   to define one of these new containers.
 14 
 15   # Example Usage
 16 
 17       MyFramework = Ember.Namespace.create({
 18         VERSION: '1.0.0'
 19       });
 20 
 21 */
 22 Ember.Namespace = Ember.Object.extend({
 23   isNamespace: true,
 24 
 25   init: function() {
 26     Ember.Namespace.NAMESPACES.push(this);
 27     Ember.Namespace.PROCESSED = false;
 28   },
 29 
 30   toString: function() {
 31     Ember.identifyNamespaces();
 32     return this[Ember.GUID_KEY+'_name'];
 33   },
 34 
 35   destroy: function() {
 36     var namespaces = Ember.Namespace.NAMESPACES;
 37     window[this.toString()] = undefined;
 38     namespaces.splice(namespaces.indexOf(this), 1);
 39     this._super();
 40   }
 41 });
 42 
 43 Ember.Namespace.NAMESPACES = [Ember];
 44 Ember.Namespace.PROCESSED = false;
