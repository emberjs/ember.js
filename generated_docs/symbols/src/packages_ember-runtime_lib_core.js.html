 73 /**
 74   Returns a consistent type for the passed item.
 75 
 76   Use this instead of the built-in Ember.typeOf() to get the type of an item.
 77   It will return the same result across all browsers and includes a bit
 78   more detail.  Here is what will be returned:
 79 
 80   | Return Value Constant | Meaning |
 81   | 'string' | String primitive |
 82   | 'number' | Number primitive |
 83   | 'boolean' | Boolean primitive |
 84   | 'null' | Null value |
 85   | 'undefined' | Undefined value |
 86   | 'function' | A function |
 87   | 'array' | An instance of Array |
 88   | 'class' | A Ember class (created using Ember.Object.extend()) |
 89   | 'instance' | A Ember object instance |
 90   | 'error' | An instance of the Error object |
 91   | 'object' | A JavaScript object not inheriting from Ember.Object |
 92 
 93   @param item {Object} the item to check
 94   @returns {String} the type
 95 */
 96 Ember.typeOf = function(item) {
 97   var ret;
 98   
 99   ret = item==null ? String(item) : TYPE_MAP[toString.call(item)]||'object';
100 
101   if (ret === 'function') {
102     if (Ember.Object && Ember.Object.detect(item)) ret = 'class';
103   } else if (ret === 'object') {
104     if (item instanceof Error) ret = 'error';
105     else if (Ember.Object && item instanceof Ember.Object) ret = 'instance';
106     else ret = 'object';
107   }
108   
109   return ret;
110 };

138 /**
139  This will compare two javascript values of possibly different types.
140  It will tell you which one is greater than the other by returning:
141 
142   - -1 if the first is smaller than the second,
143   - 0 if both are equal,
144   - 1 if the first is greater than the second.
145 
146  The order is calculated based on Ember.ORDER_DEFINITION, if types are different.
147  In case they have the same type an appropriate comparison for this type is made.
148 
149  @param {Object} v First value to compare
150  @param {Object} w Second value to compare
151  @returns {Number} -1 if v < w, 0 if v = w and 1 if v > w.
152 */
153 Ember.compare = function (v, w) {
154   if (v === w) { return 0; }
155 
156   var type1 = Ember.typeOf(v);
157   var type2 = Ember.typeOf(w);
158 
159   var Comparable = Ember.Comparable;
160   if (Comparable) {
161     if (type1==='instance' && Comparable.detect(v.constructor)) {
162       return v.constructor.compare(v, w);
163     }
164     
165     if (type2 === 'instance' && Comparable.detect(w.constructor)) {
166       return 1-w.constructor.compare(w, v);
167     }
168   }
169 
170 171   // If we haven't yet generated a reverse-mapping of Ember.ORDER_DEFINITION,
172   // do so now.
173   var mapping = Ember.ORDER_DEFINITION_MAPPING;
174   if (!mapping) {
175     var order = Ember.ORDER_DEFINITION;
176     mapping = Ember.ORDER_DEFINITION_MAPPING = {};
177     var idx, len;
178     for (idx = 0, len = order.length; idx < len;  ++idx) {
179       mapping[order[idx]] = idx;
180     }
181 
182     // We no longer need Ember.ORDER_DEFINITION.
183     delete Ember.ORDER_DEFINITION;
184   }
185 
186   var type1Index = mapping[type1];
187   var type2Index = mapping[type2];
188 
189   if (type1Index < type2Index) { return -1; }
190   if (type1Index > type2Index) { return 1; }
191 
192   // types are equal - so we have to check values now
193   switch (type1) {
194     case 'boolean':
195     case 'number':
196       if (v < w) { return -1; }
197       if (v > w) { return 1; }
198       return 0;
199 
200     case 'string':
201       var comp = v.localeCompare(w);
202       if (comp < 0) { return -1; }
203       if (comp > 0) { return 1; }
204       return 0;
205 
206     case 'array':
207       var vLen = v.length;
208       var wLen = w.length;
209       var l = Math.min(vLen, wLen);
210       var r = 0;
211       var i = 0;
212       var thisFunc = arguments.callee;
213       while (r === 0 && i < l) {
214         r = thisFunc(v[i],w[i]);
215         i++;
216       }
217       if (r !== 0) { return r; }
218 
219       // all elements are equal now
220       // shorter array should be ordered first
221       if (vLen < wLen) { return -1; }
222       if (vLen > wLen) { return 1; }
223       // arrays are equal now
224       return 0;
225 
226     case 'instance':
227       if (Ember.Comparable && Ember.Comparable.detect(v)) { 
228         return v.compare(v, w); 
229       }
230       return 0;
231 
232     default:
233       return 0;
234   }
235 };

237 /** @private */
238 function _copy(obj, deep, seen, copies) {
239   var ret, loc, key;
240 
241   // primitive data types are immutable, just return them.
242   if ('object' !== typeof obj || obj===null) return obj;
243 
244   // avoid cyclical loops
245   if (deep && (loc=seen.indexOf(obj))>=0) return copies[loc];
246   
247   ember_assert('Cannot clone an Ember.Object that does not implement Ember.Copyable', !(obj instanceof Ember.Object) || (Ember.Copyable && Ember.Copyable.detect(obj)));
248 
249   // IMPORTANT: this specific test will detect a native array only.  Any other
250   // object will need to implement Copyable.
251   if (Ember.typeOf(obj) === 'array') {
252     ret = obj.slice();
253     if (deep) {
254       loc = ret.length;
255       while(--loc>=0) ret[loc] = _copy(ret[loc], deep, seen, copies);
256     }
257   } else if (Ember.Copyable && Ember.Copyable.detect(obj)) {
258     ret = obj.copy(deep, seen, copies);
259   } else {
260     ret = {};
261     for(key in obj) {
262       if (!obj.hasOwnProperty(key)) continue;
263       ret[key] = deep ? _copy(obj[key], deep, seen, copies) : obj[key];
264     }
265   }
266   
267   if (deep) {
268     seen.push(obj);
269     copies.push(ret);
270   }
271 
272   return ret;
273 }

275 /**
276   Creates a clone of the passed object. This function can take just about
277   any type of object and create a clone of it, including primitive values
278   (which are not actually cloned because they are immutable).
279 
280   If the passed object implements the clone() method, then this function
281   will simply call that method and return the result.
282 
283   @param {Object} object The object to clone
284   @param {Boolean} deep If true, a deep copy of the object is made
285   @returns {Object} The cloned object
286 */
287 Ember.copy = function(obj, deep) {
288   // fast paths
289   if ('object' !== typeof obj || obj===null) return obj; // can't copy primitives
290   if (Ember.Copyable && Ember.Copyable.detect(obj)) return obj.copy(deep);
291   return _copy(obj, deep, deep ? [] : null, deep ? [] : null);
292 };

294 /**
295   Convenience method to inspect an object. This method will attempt to
296   convert the object into a useful string description.
297 
298   @param {Object} obj The object you want to inspect.
299   @returns {String} A description of the object
300 */
301 Ember.inspect = function(obj) {
302   var v, ret = [];
303   for(var key in obj) {
304     if (obj.hasOwnProperty(key)) {
305       v = obj[key];
306       if (v === 'toString') { continue; } // ignore useless items
307       if (Ember.typeOf(v) === 'function') { v = "function() { ... }"; }
308       ret.push(key + ": " + v);
309     }
310   }
311   return "{" + ret.join(" , ") + "}";
312 };

314 /**
315   Compares two objects, returning true if they are logically equal.  This is 
316   a deeper comparison than a simple triple equal.  For arrays and enumerables
317   it will compare the internal objects.  For any other object that implements
318   `isEqual()` it will respect that method.
319   
320   @param {Object} a first object to compare
321   @param {Object} b second object to compare
322   @returns {Boolean}
323 */
324 Ember.isEqual = function(a, b) {
325   if (a && 'function'===typeof a.isEqual) return a.isEqual(b);
326   return a === b;
327 };

371 /**
372   @class
373 
374   A subclass of the JavaScript Error object for use in Ember.
375 */
376 Ember.Error = function() {
377   var tmp = Error.prototype.constructor.apply(this, arguments);
378 
379   for (var p in tmp) {
380     if (tmp.hasOwnProperty(p)) { this[p] = tmp[p]; }
381   }
382   this.message = tmp.message;
383 };
