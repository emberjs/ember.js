1 var get = Ember.get, set = Ember.set, getPath = Ember.getPath, fmt = Ember.String.fmt;
  2 
  3 require('ember-states/state');
  4 
  5 Ember.LOG_STATE_TRANSITIONS = false;
  6 
  7 /**
  8   @class
  9 */
 10 Ember.StateManager = Ember.State.extend(
 11 /** @scope Ember.State.prototype */ {
 12 
 13   /**
 14     When creating a new statemanager, look for a default state to transition
 15     into. This state can either be named `start`, or can be specified using the
 16     `initialState` property.
 17   */
 18   init: function() {
 19     this._super();
 20 
 21     var initialState = get(this, 'initialState');
 22 
 23     if (!initialState && get(this, 'start')) {
 24       initialState = 'start';
 25     }
 26 
 27     if (initialState) {
 28       this.goToState(initialState);
 29     }
 30  31   },
 32 
 33   currentState: null,
 34 
 35   /**
 36     @property
 37 
 38     If the current state is a view state or the descendent of a view state,
 39     this property will be the view associated with it. If there is no
 40     view state active in this state manager, this value will be null.
 41   */
 42   currentView: Ember.computed(function() {
 43     var currentState = get(this, 'currentState'),
 44         view;
 45 
 46     while (currentState) {
 47       if (get(currentState, 'isViewState')) {
 48         view = get(currentState, 'view');
 49         if (view) { return view; }
 50       }
 51 
 52       currentState = get(currentState, 'parentState');
 53     }
 54 
 55     return null;
 56   }).property('currentState').cacheable(),
 57 
 58   send: function(event, context) {
 59     this.sendRecursively(event, get(this, 'currentState'), context);
 60   },
 61 
 62   sendRecursively: function(event, currentState, context) {
 63     var log = Ember.LOG_STATE_TRANSITIONS;
 64 
 65     var action = currentState[event];
 66 
 67     if (action) {
 68       if (log) { console.log(fmt("STATEMANAGER: Sending event '%@' to state %@.", [event, currentState.name])); }
 69       action.call(currentState, this, context);
 70     } else {
 71       var parentState = get(currentState, 'parentState');
 72       if (parentState) { this.sendRecursively(event, parentState, context); }
 73     }
 74   },
 75 
 76   findStatesByRoute: function(state, route) {
 77     if (!route || route === "") { return undefined; }
 78     var r = route.split('.'), ret = [];
 79 
 80     for (var i=0, len = r.length; i < len; i += 1) {
 81       var states = get(state, 'states') ;
 82 
 83       if (!states) { return undefined; }
 84 
 85       var s = get(states, r[i]);
 86       if (s) { state = s; ret.push(s); }
 87       else { return undefined; }
 88     }
 89 
 90     return ret;
 91   },
 92 
 93   goToState: function(name) {
 94     if (Ember.empty(name)) { return; }
 95 
 96     var currentState = get(this, 'currentState') || this, state, newState;
 97 
 98     var exitStates = [], enterStates;
 99 
100     state = currentState;
101 
102     if (state.routes[name]) {
103       // cache hit
104       exitStates = state.routes[name].exitStates;
105       enterStates = state.routes[name].enterStates;
106       state = state.routes[name].futureState;
107     } else {
108       // cache miss
109 
110       newState = this.findStatesByRoute(currentState, name);
111 
112       while (state && !newState) {
113         exitStates.unshift(state);
114 
115         state = get(state, 'parentState');
116         if (!state) {
117           newState = this.findStatesByRoute(this, name);
118           if (!newState) { return; }
119         }
120         newState = this.findStatesByRoute(state, name);
121       }
122 
123       enterStates = newState.slice(0), exitStates = exitStates.slice(0);
124 
125       if (enterStates.length > 0) {
126         state = enterStates[enterStates.length - 1];
127 
128         while (enterStates.length > 0 && enterStates[0] === exitStates[0]) {
129           enterStates.shift();
130           exitStates.shift();
131         }
132       }
133 
134       currentState.routes[name] = {
135         exitStates: exitStates,
136         enterStates: enterStates,
137         futureState: state
138       };
139     }
140 
141     this.enterState(exitStates, enterStates, state);
142   },
143 
144   getState: function(name) {
145     var state = get(this, name),
146         parentState = get(this, 'parentState');
147 
148     if (state) {
149       return state;
150     } else if (parentState) {
151       return parentState.getState(name);
152     }
153   },
154 
155   asyncEach: function(list, callback, doneCallback) {
156     var async = false, self = this;
157 
158     if (!list.length) {
159       if (doneCallback) { doneCallback.call(this); }
160       return;
161     }
162 
163     var head = list[0];
164     var tail = list.slice(1);
165 
166     var transition = {
167       async: function() { async = true; },
168       resume: function() {
169         self.asyncEach(tail, callback, doneCallback);
170       }
171     };
172 
173     callback.call(this, head, transition);
174 
175     if (!async) { transition.resume(); }
176   },
177 
178   enterState: function(exitStates, enterStates, state) {
179     var log = Ember.LOG_STATE_TRANSITIONS;
180 
181     var stateManager = this;
182 
183     exitStates.reverse();
184     this.asyncEach(exitStates, function(state, transition) {
185       state.exit(stateManager, transition);
186     }, function() {
187       this.asyncEach(enterStates, function(state, transition) {
188         if (log) { console.log("STATEMANAGER: Entering " + state.name); }
189         state.enter(stateManager, transition);
190       }, function() {
191         var startState = state, enteredState, initialState;
192 
193         initialState = get(startState, 'initialState');
194 
195         if (!initialState) {
196           initialState = 'start';
197         }
198 
199         // right now, start states cannot be entered asynchronously
200         while (startState = get(startState, initialState)) {
201           enteredState = startState;
202 
203           if (log) { console.log("STATEMANAGER: Entering " + startState.name); }
204           startState.enter(stateManager);
205 
206           initialState = get(startState, 'initialState');
207 
208           if (!initialState) {
209             initialState = 'start';
210           }
211         }
212 
213         set(this, 'currentState', enteredState || state);
214       });
215     });
216   }
217 });
