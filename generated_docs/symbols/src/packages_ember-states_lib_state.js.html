1 var get = Ember.get, set = Ember.set;
  2 
  3 Ember.State = Ember.Object.extend({
  4   isState: true,
  5   parentState: null,
  6   start: null,
  7 
  8   init: function() {
  9     var states = get(this, 'states'), foundStates;
 10 
 11     // As a convenience, loop over the properties
 12     // of this state and look for any that are other
 13     // Ember.State instances or classes, and move them
 14     // to the `states` hash. This avoids having to
 15     // create an explicit separate hash.
 16 
 17     if (!states) {
 18       states = {};
 19       for (var name in this) {
 20         if (name === "constructor") { continue; }
 21         value = this.setupChild(name, this[name]);
 22 
 23         if (value) {
 24           foundStates = true;
 25           states[name] = value;
 26         }
 27       }
 28 
 29       if (foundStates) { set(this, 'states', states); }
 30     } else {
 31       for (var name in states) {
 32         this.setupChild(name, states[name]);
 33       }
 34     }
 35 
 36     set(this, 'routes', {});
 37   },
 38 
 39   setupChild: function(name, value) {
 40     if (!value) { return false; }
 41 
 42     if (Ember.State.detect(value)) {
 43       value = value.create();
 44     }
 45 
 46     if (value.isState) {
 47       set(value, 'parentState', this);
 48       set(value, 'name', (get(this, 'name') || '') + '.' + name);
 49       return value;
 50     }
 51 
 52     return false;
 53   },
 54 
 55   enter: Ember.K,
 56   exit: Ember.K
 57 });
