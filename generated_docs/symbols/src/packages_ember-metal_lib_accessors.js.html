 18 // GET AND SET

 20 // If we are on a platform that supports accessors we can get use those.
 21 // Otherwise simulate accessors by looking up the property directly on the
 22 // object.

 24 var get, set;
 25 
 26 /** @private */
 27 get = function get(obj, keyName) {
 28   if (keyName === undefined && 'string' === typeof obj) {
 29     keyName = obj;
 30     obj = Ember;
 31   }
 32   
 33   if (!obj) return undefined;
 34   var ret = obj[keyName];
 35   if (ret===undefined && 'function'===typeof obj.unknownProperty) {
 36     ret = obj.unknownProperty(keyName);
 37   }
 38   return ret;
 39 };
 40 
 41 /** @private */
 42 set = function set(obj, keyName, value) {
 43   if (('object'===typeof obj) && !(keyName in obj)) {
 44     if ('function' === typeof obj.setUnknownProperty) {
 45       obj.setUnknownProperty(keyName, value);
 46     } else if ('function' === typeof obj.unknownProperty) {
 47       obj.unknownProperty(keyName, value);
 48     } else obj[keyName] = value;
 49   } else {
 50     obj[keyName] = value;
 51   }
 52   return value;
 53 };
 54 
 55 if (!USE_ACCESSORS) {
 56 
 57   var o_get = get, o_set = set;
 58   
 59   /** @private */
 60   get = function(obj, keyName) {
 61     if (keyName === undefined && 'string' === typeof obj) {
 62       keyName = obj;
 63       obj = Ember;
 64     }
 65 
 66     ember_assert("You need to provide an object and key to `get`.", !!obj && keyName);
 67 
 68     if (!obj) return undefined;
 69     var desc = meta(obj, false).descs[keyName];
 70     if (desc) return desc.get(obj, keyName);
 71     else return o_get(obj, keyName);
 72   };
 73 
 74   /** @private */
 75   set = function(obj, keyName, value) {
 76     ember_assert("You need to provide an object and key to `set`.", !!obj && keyName !== undefined);
 77     var desc = meta(obj, false).descs[keyName];
 78     if (desc) desc.set(obj, keyName, value);
 79     else o_set(obj, keyName, value);
 80     return value;
 81  82   };
 83

 86 /**
 87   @function
 88   
 89   Gets the value of a property on an object.  If the property is computed,
 90   the function will be invoked.  If the property is not defined but the 
 91   object implements the unknownProperty() method then that will be invoked.
 92   
 93   If you plan to run on IE8 and older browsers then you should use this 
 94   method anytime you want to retrieve a property on an object that you don't
 95   know for sure is private.  (My convention only properties beginning with 
 96   an underscore '_' are considered private.)
 97   
 98   On all newer browsers, you only need to use this method to retrieve 
 99   properties if the property might not be defined on the object and you want
100   to respect the unknownProperty() handler.  Otherwise you can ignore this
101   method.
102   
103   Note that if the obj itself is null, this method will simply return 
104   undefined.
105   
106   @param {Object} obj
107     The object to retrieve from.
108     
109   @param {String} keyName
110     The property key to retrieve
111     
112   @returns {Object} the property value or null.
113 */
114 Ember.get = get;

124   If you plan to run on IE8 and older browsers then you should use this

151 /** @private */
152 function normalizePath(path) {
153   ember_assert('must pass non-empty string to normalizePath()', path && path!=='');
154     
155   if (path==='*') return path; //special case...
156   var first = path.charAt(0);
157   if(first==='.') return 'this'+path;
158   if (first==='*' && path.charAt(1)!=='.') return 'this.'+path.slice(1);
159   return path;
160 }
161 
162 // assumes normalized input; no *, normalized path, always a target...
163 /** @private */
164 function getPath(target, path) {
165   var len = path.length, idx, next, key;
166   
167   idx = path.indexOf('*');
168   if (idx>0 && path.charAt(idx-1)!=='.') {
169     return getPath(getPath(target, path.slice(0, idx)), path.slice(idx+1));
170   }
171 
172   idx = 0;
173   while(target && idx<len) {
174     next = path.indexOf('.', idx);
175     if (next<0) next = len;
176     key = path.slice(idx, next);
177     target = key==='*' ? target : get(target, key);
178 
179     if (target && target.isDestroyed) { return undefined; }
180 
181     idx = next+1;
182   }
183   return target ;
184 }

186 var TUPLE_RET = [];
187 var IS_GLOBAL = /^([A-Z$]|([0-9][A-Z$]))/;
188 var IS_GLOBAL_PATH = /^([A-Z$]|([0-9][A-Z$])).*[\.\*]/;
189 var HAS_THIS  = /^this[\.\*]/;
190 var FIRST_KEY = /^([^\.\*]+)/;
191 
192 /** @private */
193 function firstKey(path) {
194   return path.match(FIRST_KEY)[0];
195 }
196 
197 // assumes path is already normalized
198 /** @private */
199 function normalizeTuple(target, path) {
200   var hasThis  = HAS_THIS.test(path),
201       isGlobal = !hasThis && IS_GLOBAL_PATH.test(path),
202       key;
203 
204   if (!target || isGlobal) target = window;
205   if (hasThis) path = path.slice(5);
206   
207   var idx = path.indexOf('*');
208   if (idx>0 && path.charAt(idx-1)!=='.') {
209     
210     // should not do lookup on a prototype object because the object isn't
211     // really live yet.
212     if (target && meta(target,false).proto!==target) {
213       target = getPath(target, path.slice(0, idx));
214     } else {
215       target = null;
216     }
217     path   = path.slice(idx+1);
218 
219   } else if (target === window) {
220     key = firstKey(path);
221     target = get(target, key);
222     path   = path.slice(key.length+1);
223   }
224 
225   // must return some kind of path to be valid else other things will break.
226   if (!path || path.length===0) throw new Error('Invalid Path');
227   
228   TUPLE_RET[0] = target;
229   TUPLE_RET[1] = path;
230   return TUPLE_RET;
231 }

244 /**
245   @private
246 
247   Normalizes a target/path pair to reflect that actual target/path that should
248   be observed, etc.  This takes into account passing in global property 
249   paths (i.e. a path beginning with a captial letter not defined on the 
250   target) and * separators.
251   
252   @param {Object} target
253     The current target.  May be null.
254     
255   @param {String} path
256     A path on the target or a global property path.
257     
258   @returns {Array} a temporary array with the normalized target/path pair.
259 */
260 Ember.normalizeTuple = function(target, path) {
261   return normalizeTuple(target, normalizePath(path));
262 };

266 Ember.getPath = function(root, path, _checkGlobal) {
267   var pathOnly, hasThis, hasStar, isGlobal, ret;
268   
269   // Helpers that operate with 'this' within an #each
270   if (path === '') {
271     return root;
272   }
273 
274   if (!path && 'string'===typeof root) {
275     path = root;
276     root = null;
277     pathOnly = true;
278   }
279 
280   hasStar = path.indexOf('*') > -1;
281 
282   // If there is no root and path is a key name, return that
283   // property from the global object.
284   // E.g. getPath('Ember') -> Ember
285   if (root === null && !hasStar && path.indexOf('.') < 0) { return get(window, path); }
286 
287   // detect complicated paths and normalize them
288   path = normalizePath(path);
289   hasThis  = HAS_THIS.test(path);
290 
291   if (!root || hasThis || hasStar) {
292     ember_deprecate("Fetching globals with Ember.getPath is deprecated (root: "+root+", path: "+path+")", !root || root === window || !IS_GLOBAL.test(path));
293 
294     var tuple = normalizeTuple(root, path);
295     root = tuple[0];
296     path = tuple[1];
297     tuple.length = 0;
298   }
299 
300   ret = getPath(root, path);
301 
302   if (ret === undefined && !pathOnly && !hasThis && root !== window && IS_GLOBAL.test(path) && _checkGlobal !== false) {
303     ember_deprecate("Fetching globals with Ember.getPath is deprecated (root: "+root+", path: "+path+")");
304     return Ember.getPath(window, path);
305   } else {
306     return ret;
307   }
308 };

310 Ember.setPath = function(root, path, value, tolerant) {
311   var keyName;
312   
313   if (arguments.length===2 && 'string' === typeof root) {
314     value = path;
315     path = root;
316     root = null;
317   }
318   
319   path = normalizePath(path);
320   if (path.indexOf('*')>0) {
321     ember_deprecate("Setting globals with Ember.setPath is deprecated (path: "+path+")", !root || root === window || !IS_GLOBAL.test(path));
322 
323     var tuple = normalizeTuple(root, path);
324     root = tuple[0];
325     path = tuple[1];
326     tuple.length = 0;
327   }
328 
329   if (path.indexOf('.') > 0) {
330     keyName = path.slice(path.lastIndexOf('.')+1);
331     path    = path.slice(0, path.length-(keyName.length+1));
332     if (path !== 'this') {
333       // Remove the `false` when we're done with this deprecation
334       root = Ember.getPath(root, path, false);
335       if (!root && IS_GLOBAL.test(path)) {
336         ember_deprecate("Setting globals with Ember.setPath is deprecated (path: "+path+")");
337         root = Ember.getPath(window, path);
338       }
339     }
340 
341   } else {
342     if (IS_GLOBAL.test(path)) throw new Error('Invalid Path');
343     keyName = path;
344   }
345   
346   if (!keyName || keyName.length===0 || keyName==='*') {
347     throw new Error('Invalid Path');
348   }
349 
350   if (!root) {
351     if (tolerant) { return; }
352     else { throw new Error('Object in path '+path+' could not be found or was destroyed.'); }
353   }
354 
355   return Ember.set(root, keyName, value);
356 };
357 
358 /**
359   Error-tolerant form of Ember.setPath. Will not blow up if any part of the
360   chain is undefined, null, or destroyed.
361 
362   This is primarily used when syncing bindings, which may try to update after
363   an object has been destroyed.
364 */
365 Ember.trySetPath = function(root, path, value) {
366   if (arguments.length===2 && 'string' === typeof root) {
367     value = path;
368     path = root;
369     root = null;
370   }
371 
372 373   return Ember.setPath(root, path, value, true);
374 };
