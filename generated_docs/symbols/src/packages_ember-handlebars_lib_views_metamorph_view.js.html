  4 var set = Ember.set, get = Ember.get, getPath = Ember.getPath;

Ember.Metamorph = Ember.Mixin.create({
  7   isVirtual: true,
  8   tagName: '',
  9 
 10   init: function() {
 11     this._super();
 12     set(this, 'morph', Metamorph());
 13   },
 14 
 15   beforeRender: function(buffer) {
 16     var morph = get(this, 'morph');
 17     buffer.push(morph.startTag());
 18   },
 19 
 20   afterRender: function(buffer) {
 21     var morph = get(this, 'morph');
 22     buffer.push(morph.endTag());
 23   },
 24 
 25   createElement: function() {
 26     var buffer = this.renderToBuffer();
 27     set(this, 'outerHTML', buffer.string());
 28     this.clearBuffer();
 29   },
 30 
 31   domManagerClass: Ember.Object.extend({
 32     remove: function(view) {
 33       var morph = getPath(this, 'view.morph');
 34       if (morph.isRemoved()) { return; }
 35       getPath(this, 'view.morph').remove();
 36     },
 37 
 38     prepend: function(childView) {
 39       var view = get(this, 'view');
 40 
 41       childView._insertElementLater(function() {
 42         var morph = get(view, 'morph');
 43         morph.prepend(get(childView, 'outerHTML'));
 44         childView.set('outerHTML', null);
 45       });
 46     },
 47 
 48     after: function(nextView) {
 49       var view = get(this, 'view');
 50 
 51       nextView._insertElementLater(function() {
 52         var morph = get(view, 'morph');
 53         morph.after(get(nextView, 'outerHTML'));
 54         nextView.set('outerHTML', null);
 55       });
 56     },
 57 
 58     replace: function() {
 59       var view = get(this, 'view');
 60       var morph = getPath(this, 'view.morph');
 61 
 62       view.transitionTo('preRender');
 63       view.clearRenderedChildren();
 64       var buffer = view.renderToBuffer();
 65 
 66       Ember.run.schedule('render', this, function() {
 67         if (get(view, 'isDestroyed')) { return; }
 68         view.invalidateRecursively('element');
 69         view._notifyWillInsertElement();
 70         morph.replaceWith(buffer.string());
 71         view.transitionTo('inDOM');
 72         view._notifyDidInsertElement();
 73       });
 74     }
 75   })
 76 });
