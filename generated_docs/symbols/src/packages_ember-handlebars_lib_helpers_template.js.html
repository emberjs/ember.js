  3 /**
  4   `template` allows you to render a template from inside another template.
  5   This allows you to re-use the same template in multiple places. For example:
  6 
  7       <script type="text/x-handlebars">
  8         {{#with loggedInUser}}
  9           Last Login: {{lastLogin}}
 10           User Info: {{template "user_info"}}
 11         {{/with}}
 12       </script>
 13 
 14       <script type="text/x-handlebars" data-template-name="user_info">
 15         Name: <em>{{name}}</em>
 16         Karma: <em>{{karma}}</em>
 17       </script>
 18 
 19   This helper looks for templates in the global Ember.TEMPLATES hash. If you
 20   add <script> tags to your page with the `data-template-name` attribute set,
 21   they will be compiled and placed in this hash automatically.
 22 
 23   You can also manually register templates by adding them to the hash:
 24 
 25       Ember.TEMPLATES["my_cool_template"] = Ember.Handlebars.compile('<b>{{user}}</b>');
 26 
 27   @name Handlebars.helpers.template
 28   @param {String} templateName the template to render
 29 */
 30 
 31 Ember.Handlebars.registerHelper('template', function(name, options) {
 32   var template = Ember.TEMPLATES[name];
 33 
 34   ember_assert("Unable to find template with name '"+name+"'.", !!template);
 35 
 36   Ember.TEMPLATES[name](this, { data: options.data });
 37 });
