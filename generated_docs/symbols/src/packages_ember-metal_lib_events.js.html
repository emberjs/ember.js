 12 var o_create = Ember.platform.create;
 13 var meta = Ember.meta;
 14 var guidFor = Ember.guidFor;

 17 /**
 18   The event system uses a series of nested hashes to store listeners on an
 19   object. When a listener is registered, or when an event arrives, these
 20   hashes are consulted to determine which target and action pair to invoke.
 21 
 22   The hashes are stored in the object's meta hash, and look like this:
 23 
 24       // Object's meta hash
 25       {
 26         listeners: {               // variable name: `listenerSet`
 27           "foo:changed": {         // variable name: `targetSet`
 28             [targetGuid]: {        // variable name: `actionSet`
 29               [methodGuid]: {      // variable name: `action`
 30                 target: [Object object],
 31                 method: [Function function],
 32                 xform: [Function function]
 33               }
 34             }
 35           }
 36         }
 37       }
 38 
 39 */
 40 
 41 /** @private */
 42 var metaPath = Ember.metaPath;
 43 
 44 // Gets the set of all actions, keyed on the guid of each action's
 45 // method property.
 46 /** @private */
 47 function actionSetFor(obj, eventName, target, writable) {
 48   var targetGuid = guidFor(target);
 49   return metaPath(obj, ['listeners', eventName, targetGuid], writable);
 50 }
 51 
 52 // Gets the set of all targets, keyed on the guid of each action's
 53 // target property.
 54 /** @private */
 55 function targetSetFor(obj, eventName) {
 56   var listenerSet = meta(obj, false).listeners;
 57   if (!listenerSet) { return false; }
 58 
 59   return listenerSet[eventName] || false;
 60 }

 62 // TODO: This knowledge should really be a part of the
 63 // meta system.
 64 var SKIP_PROPERTIES = { __ember_source__: true };
 65 
 66 // For a given target, invokes all of the methods that have
 67 // been registered as a listener.
 68 /** @private */
 69 function invokeEvents(targetSet, params) {
 70   // Iterate through all elements of the target set
 71   for(var targetGuid in targetSet) {
 72     if (SKIP_PROPERTIES[targetGuid]) { continue; }
 73 
 74     var actionSet = targetSet[targetGuid];
 75 
 76     // Iterate through the elements of the action set
 77     for(var methodGuid in actionSet) {
 78       if (SKIP_PROPERTIES[methodGuid]) { continue; }
 79 
 80       var action = actionSet[methodGuid];
 81       if (!action) { continue; }
 82 
 83       // Extract target and method for each action
 84       var method = action.method;
 85       var target = action.target;
 86 
 87       // If there is no target, the target is the object
 88       // on which the event was fired.
 89       if (!target) { target = params[0]; }
 90       if ('string' === typeof method) { method = target[method]; }
 91 
 92       // Listeners can provide an `xform` function, which can perform
 93       // arbitrary transformations, such as changing the order of
 94       // parameters.
 95       //
 96       // This is primarily used by ember-runtime's observer system, which
 97       // provides a higher level abstraction on top of events, including
 98       // dynamically looking up current values and passing them into the
 99       // registered listener.
100       var xform = action.xform;
101 
102       if (xform) {
103         xform(target, method, params);
104       } else {
105         method.apply(target, params);
106       }
107     }
108   }
109 }

111 /**
112   The parameters passed to an event listener are not exactly the
113   parameters passed to an observer. if you pass an xform function, it will
114   be invoked and is able to translate event listener parameters into the form
115   that observers are expecting.
116 
117   @memberOf Ember
118 */
119 function addListener(obj, eventName, target, method, xform) {
120   ember_assert("You must pass at least an object and event name to Ember.addListener", !!obj && !!eventName);
121 
122   if (!method && 'function' === typeof target) {
123     method = target;
124     target = null;
125   }
126 
127   var actionSet = actionSetFor(obj, eventName, target, true),
128       methodGuid = guidFor(method), ret;
129 
130   if (!actionSet[methodGuid]) {
131     actionSet[methodGuid] = { target: target, method: method, xform: xform };
132   } else {
133     actionSet[methodGuid].xform = xform; // used by observers etc to map params
134   }
135 
136   if ('function' === typeof obj.didAddListener) {
137     obj.didAddListener(eventName, target, method);
138   }
139 
140   return ret; // return true if this is the first listener.
141 }
142 
143 /** @memberOf Ember */
144 function removeListener(obj, eventName, target, method) {
145   if (!method && 'function'===typeof target) {
146     method = target;
147     target = null;
148   }
149 
150   var actionSet = actionSetFor(obj, eventName, target, true),
151       methodGuid = guidFor(method);
152 
153   // we can't simply delete this parameter, because if we do, we might
154   // re-expose the property from the prototype chain.
155   if (actionSet && actionSet[methodGuid]) { actionSet[methodGuid] = null; }
156 
157   if (obj && 'function'===typeof obj.didRemoveListener) {
158     obj.didRemoveListener(eventName, target, method);
159   }
160 }

162 // returns a list of currently watched events
163 /** @memberOf Ember */
164 function watchedEvents(obj) {
165   var listeners = meta(obj, false).listeners, ret = [];
166 
167   if (listeners) {
168     for(var eventName in listeners) {
169       if (!SKIP_PROPERTIES[eventName] && listeners[eventName]) {
170         ret.push(eventName);
171       }
172     }
173   }
174   return ret;
175 }

177 /** @memberOf Ember */
178 function sendEvent(obj, eventName) {
179 
180   // first give object a chance to handle it
181   if (obj !== Ember && 'function' === typeof obj.sendEvent) {
182     obj.sendEvent.apply(obj, array_Slice.call(arguments, 1));
183   }
184 
185   var targetSet = targetSetFor(obj, eventName);
186   if (!targetSet) { return false; }
187 
188   invokeEvents(targetSet, arguments);
189   return true;
190 }

192 /** @memberOf Ember */
193 function hasListeners(obj, eventName) {
194   var targetSet = targetSetFor(obj, eventName);
195   if (!targetSet) { return false; }
196 
197   for(var targetGuid in targetSet) {
198     if (SKIP_PROPERTIES[targetGuid] || !targetSet[targetGuid]) { continue; }
199 
200     var actionSet = targetSet[targetGuid];
201 
202     for(var methodGuid in actionSet) {
203       if (SKIP_PROPERTIES[methodGuid] || !actionSet[methodGuid]) { continue; }
204       return true; // stop as soon as we find a valid listener
205     }
206   }
207 
208   // no listeners!  might as well clean this up so it is faster later.
209   var set = metaPath(obj, ['listeners'], true);
210   set[eventName] = null;
211 
212   return false;
213 }
214 
215 /** @memberOf Ember */
216 function listenersFor(obj, eventName) {
217   var targetSet = targetSetFor(obj, eventName), ret = [];
218   if (!targetSet) { return ret; }
219 
220   var info;
221   for(var targetGuid in targetSet) {
222     if (SKIP_PROPERTIES[targetGuid] || !targetSet[targetGuid]) { continue; }
223 
224     var actionSet = targetSet[targetGuid];
225 
226     for(var methodGuid in actionSet) {
227       if (SKIP_PROPERTIES[methodGuid] || !actionSet[methodGuid]) { continue; }
228       info = actionSet[methodGuid];
229       ret.push([info.target, info.method]);
230     }
231   }
232 
233   return ret;
234 }
