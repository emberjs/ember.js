  8 var get = Ember.get, set = Ember.set;
  9 
 10 /**
 11   @class
 12 
 13   Ember.RenderBuffer gathers information regarding the a view and generates the
 14   final representation. Ember.RenderBuffer will generate HTML which can be pushed
 15   to the DOM.
 16 
 17   @extends Ember.Object
 18 */
 19 Ember.RenderBuffer = function(tagName) {
 20   return Ember._RenderBuffer.create({ elementTag: tagName });
 21 };
 22 
 23 Ember._RenderBuffer = Ember.Object.extend(
 24 /** @scope Ember.RenderBuffer.prototype */ {
 25 
 26   /**
 27     Array of class-names which will be applied in the class="" attribute
 28 
 29     You should not maintain this array yourself, rather, you should use
 30     the addClass() method of Ember.RenderBuffer.
 31 
 32     @type Array
 33     @default []
 34   */
 35   elementClasses: null,
 36 
 37   /**
 38     The id in of the element, to be applied in the id="" attribute
 39 
 40     You should not set this property yourself, rather, you should use
 41     the id() method of Ember.RenderBuffer.
 42 
 43     @type String
 44     @default null
 45   */
 46   elementId: null,
 47 
 48   /**
 49     A hash keyed on the name of the attribute and whose value will be
 50     applied to that attribute. For example, if you wanted to apply a
 51     data-view="Foo.bar" property to an element, you would set the
 52     elementAttributes hash to {'data-view':'Foo.bar'}
 53 
 54     You should not maintain this hash yourself, rather, you should use
 55     the attr() method of Ember.RenderBuffer.
 56 
 57     @type Hash
 58     @default {}
 59   */
 60   elementAttributes: null,
 61 
 62   /**
 63     The tagname of the element an instance of Ember.RenderBuffer represents.
 64 
 65     Usually, this gets set as the first parameter to Ember.RenderBuffer. For
 66     example, if you wanted to create a `p` tag, then you would call
 67 
 68       Ember.RenderBuffer('p')
 69 
 70     @type String
 71     @default null
 72   */
 73   elementTag: null,
 74 
 75   /**
 76     A hash keyed on the name of the style attribute and whose value will
 77     be applied to that attribute. For example, if you wanted to apply a
 78     background-color:black;" style to an element, you would set the
 79     elementStyle hash to {'background-color':'black'}
 80 
 81     You should not maintain this hash yourself, rather, you should use
 82     the style() method of Ember.RenderBuffer.
 83 
 84     @type Hash
 85     @default {}
 86   */
 87   elementStyle: null,
 88 
 89   /**
 90     Nested RenderBuffers will set this to their parent RenderBuffer
 91     instance.
 92 
 93     @type Ember._RenderBuffer
 94   */
 95   parentBuffer: null,
 96 
 97   /** @private */
 98   init: function() {
 99     this._super();
100 
101     set(this ,'elementClasses', Ember.A());
102     set(this, 'elementAttributes', {});
103     set(this, 'elementStyle', {});
104     set(this, 'childBuffers', []);
105     set(this, 'elements', {});
106   },
107 
108   /**
109     Adds a string of HTML to the RenderBuffer.
110 
111     @param {String} string HTML to push into the buffer
112     @returns {Ember.RenderBuffer} this
113   */
114   push: function(string) {
115     get(this, 'childBuffers').push(String(string));
116     return this;
117   },
118 
119   /**
120     Adds a class to the buffer, which will be rendered to the class attribute.
121 
122     @param {String} className Class name to add to the buffer
123     @returns {Ember.RenderBuffer} this
124   */
125   addClass: function(className) {
126     get(this, 'elementClasses').addObject(className);
127     return this;
128   },
129 
130   /**
131     Sets the elementID to be used for the element.
132 
133     @param {String} id
134     @returns {Ember.RenderBuffer} this
135   */
136   id: function(id) {
137     set(this, 'elementId', id);
138     return this;
139   },
140 
141   // duck type attribute functionality like jQuery so a render buffer
142   // can be used like a jQuery object in attribute binding scenarios.
143 
144   /**
145     Adds an attribute which will be rendered to the element.
146 
147     @param {String} name The name of the attribute
148     @param {String} value The value to add to the attribute
149     @returns {Ember.RenderBuffer|String} this or the current attribute value
150   */
151   attr: function(name, value) {
152     var attributes = get(this, 'elementAttributes');
153 
154     if (arguments.length === 1) {
155       return attributes[name]
156     } else {
157       attributes[name] = value;
158     }
159 
160     return this;
161   },
162 
163   /**
164     Remove an attribute from the list of attributes to render.
165 
166     @param {String} name The name of the attribute
167     @returns {Ember.RenderBuffer} this
168   */
169   removeAttr: function(name) {
170     var attributes = get(this, 'elementAttributes');
171     delete attributes[name];
172 
173     return this;
174   },
175 
176   /**
177     Adds a style to the style attribute which will be rendered to the element.
178 
179     @param {String} name Name of the style
180     @param {String} value
181     @returns {Ember.RenderBuffer} this
182   */
183   style: function(name, value) {
184     get(this, 'elementStyle')[name] = value;
185     return this;
186   },
187 
188   /**
189     Create a new child render buffer from a parent buffer. Optionally set
190     additional properties on the buffer. Optionally invoke a callback
191     with the newly created buffer.
192 
193     This is a primitive method used by other public methods: `begin`,
194     `prepend`, `replaceWith`, `insertAfter`.
195 
196     @private
197     @param {String} tagName Tag name to use for the child buffer's element
198     @param {Ember._RenderBuffer} parent The parent render buffer that this
199       buffer should be appended to.
200     @param {Function} fn A callback to invoke with the newly created buffer.
201     @param {Object} other Additional properties to add to the newly created
202       buffer.
203   */
204   newBuffer: function(tagName, parent, fn, other) {
205     var buffer = Ember._RenderBuffer.create({
206       parentBuffer: parent,
207       elementTag: tagName
208     });
209 
210     if (other) { buffer.setProperties(other); }
211     if (fn) { fn.call(this, buffer); }
212 
213     return buffer;
214   },
215 
216   /**
217     Replace the current buffer with a new buffer. This is a primitive
218     used by `remove`, which passes `null` for `newBuffer`, and `replaceWith`,
219     which passes the new buffer it created.
220 
221     @private
222     @param {Ember._RenderBuffer} buffer The buffer to insert in place of
223       the existing buffer.
224   */
225   replaceWithBuffer: function(newBuffer) {
226     var parent = get(this, 'parentBuffer');
227     if (!parent) { return; }
228 
229     var childBuffers = get(parent, 'childBuffers');
230 
231     var index = childBuffers.indexOf(this);
232 
233     if (newBuffer) {
234       childBuffers.splice(index, 1, newBuffer);
235     } else {
236       childBuffers.splice(index, 1);
237     }
238   },
239 
240   /**
241     Creates a new Ember.RenderBuffer object with the provided tagName as
242     the element tag and with its parentBuffer property set to the current
243     Ember.RenderBuffer.
244 
245     @param {String} tagName Tag name to use for the child buffer's element
246     @returns {Ember.RenderBuffer} A new RenderBuffer object
247   */
248   begin: function(tagName) {
249     return this.newBuffer(tagName, this, function(buffer) {
250       get(this, 'childBuffers').push(buffer);
251     });
252   },
253 
254   /**
255     Prepend a new child buffer to the current render buffer.
256 
257     @param {String} tagName Tag name to use for the child buffer's element
258   */
259   prepend: function(tagName) {
260     return this.newBuffer(tagName, this, function(buffer) {
261       get(this, 'childBuffers').splice(0, 0, buffer);
262     });
263   },
264 
265   /**
266     Replace the current buffer with a new render buffer.
267 
268     @param {String} tagName Tag name to use for the new buffer's element
269   */
270   replaceWith: function(tagName) {
271     var parentBuffer = get(this, 'parentBuffer');
272 
273     return this.newBuffer(tagName, parentBuffer, function(buffer) {
274       this.replaceWithBuffer(buffer);
275     });
276   },
277 
278   /**
279     Insert a new render buffer after the current render buffer.
280 
281     @param {String} tagName Tag name to use for the new buffer's element
282   */
283   insertAfter: function(tagName) {
284     var parentBuffer = get(this, 'parentBuffer');
285 
286     return this.newBuffer(tagName, parentBuffer, function(buffer) {
287       var siblings = get(parentBuffer, 'childBuffers');
288       var index = siblings.indexOf(this);
289       siblings.splice(index + 1, 0, buffer);
290     });
291   },
292 
293   /**
294     Closes the current buffer and adds its content to the parentBuffer.
295 
296     @returns {Ember.RenderBuffer} The parentBuffer, if one exists. Otherwise, this
297   */
298   end: function() {
299     var parent = get(this, 'parentBuffer');
300     return parent || this;
301   },
302 
303   remove: function() {
304     this.replaceWithBuffer(null);
305   },
306 
307   /**
308     @returns {DOMElement} The element corresponding to the generated HTML
309       of this buffer
310   */
311   element: function() {
312     return Ember.$(this.string())[0];
313   },
314 
315   /**
316     Generates the HTML content for this buffer.
317 
318     @returns {String} The generated HTMl
319   */
320   string: function() {
321     var id = get(this, 'elementId'),
322         classes = get(this, 'elementClasses'),
323         attrs = get(this, 'elementAttributes'),
324         style = get(this, 'elementStyle'),
325         tag = get(this, 'elementTag'),
326         content = '',
327         styleBuffer = [], prop;
328 
329     if (tag) {
330       var openTag = ["<" + tag];
331 
332       if (id) { openTag.push('id="' + id + '"'); }
333       if (classes.length) { openTag.push('class="' + classes.join(" ") + '"'); }
334 
335       if (!jQuery.isEmptyObject(style)) {
336         for (prop in style) {
337           if (style.hasOwnProperty(prop)) {
338             styleBuffer.push(prop + ':' + style[prop] + ';');
339           }
340         }
341 
342         openTag.push('style="' + styleBuffer.join("") + '"');
343       }
344 
345       for (prop in attrs) {
346         if (attrs.hasOwnProperty(prop)) {
347           openTag.push(prop + '="' + attrs[prop] + '"');
348         }
349       }
350 
351       openTag = openTag.join(" ") + '>';
352     }
353 
354     var childBuffers = get(this, 'childBuffers');
355 
356     childBuffers.forEach(function(buffer) {
357       var stringy = typeof buffer === 'string';
358       content += (stringy ? buffer : buffer.string());
359     });
360 
361     if (tag) {
362       return openTag + content + "</" + tag + ">";
363     } else {
364       return content;
365     }
366   }
367 
368 });
