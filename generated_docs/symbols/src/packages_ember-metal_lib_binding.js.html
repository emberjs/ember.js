 77 // Coerces a non-array value into an array.
 78 /** @private */
 79 function MULTIPLE(val) {
 80   if (val instanceof Array) return val;
 81   if (val === undefined || val === null) return [];
 82   return [val];
 83 }

 85 // Treats a single-element array as the element. Otherwise
 86  87 // returns a placeholder.
 88 /** @private */
 89 function SINGLE(val, placeholder) {
 90   if (val instanceof Array) {
 91     if (val.length>1) return placeholder;
 92     else return val[0];
 93   }
 94   return val;
 95 }

118 // Applies a binding's transformations against a value.
119 /** @private */
120 function getTransformedValue(binding, val, obj, dir) {
121 
122   // First run a type transform, if it exists, that changes the fundamental
123   // type of the value. For example, some transforms convert an array to a
124   // single object.
125 
126   var typeTransform = binding._typeTransform;
127   if (typeTransform) { val = typeTransform(val, binding._placeholder); }
128 
129   // handle transforms
130   var transforms = binding._transforms,
131       len        = transforms ? transforms.length : 0,
132       idx;
133 
134   for(idx=0;idx<len;idx++) {
135     var transform = transforms[idx][dir];
136     if (transform) { val = transform.call(this, val, obj); }
137 138   }
139   return val;
140 }

147 /** @private */
148 function getPathWithGlobals(obj, path) {
149   return getPath(isGlobalPath(path) ? window : obj, path);
150 }
151 
152 /** @private */
153 function getTransformedFromValue(obj, binding) {
154   var operation = binding._operation,
155       fromValue;
156   if (operation) {
157     fromValue = operation(obj, binding._from, binding._operand);
158   } else {
159     fromValue = getPathWithGlobals(obj, binding._from);
160   }
161   return getTransformedValue(binding, fromValue, obj, 'to');
162 }
163 
164 /** @private */
165 function getTransformedToValue(obj, binding) {
166   var toValue = getPath(obj, binding._to);
167   return getTransformedValue(binding, toValue, obj, 'from');
168 }

170 /** @private */
171 var AND_OPERATION = function(obj, left, right) {
172   return getPathWithGlobals(obj, left) && getPathWithGlobals(obj, right);
173 };
174 
175 /** @private */
176 var OR_OPERATION = function(obj, left, right) {
177   return getPathWithGlobals(obj, left) || getPathWithGlobals(obj, right);
178 };

268     Adds the specified transform to the array of transform functions.

270     A transform is a hash with `to` and `from` properties. Each property

272     forward or back direction.
273 
274     The functions you pass must have the following signature:

276           function(value) {};

430   /**
431     Attempts to connect this binding instance so that it can receive and relay
432     changes. This method will raise an exception if you have not set the
433     from/to properties yet.
434 
435     @param {Object} obj
436       The root object for this binding.
437 
438     @param {Boolean} preferFromParam
439       private: Normally, `connect` cannot take an object if `from` already set
440       an object. Internally, we would like to be able to provide a default object
441       to be used if no object was provided via `from`, so this parameter turns
442       off the assertion.
443 
444     @returns {Ember.Binding} this
445   */
446   connect: function(obj) {
447     ember_assert('Must pass a valid object to Ember.Binding.connect()', !!obj);
448 
449     var oneWay = this._oneWay, operand = this._operand;
450 
451     // add an observer on the object to be notified when the binding should be updated
452     Ember.addObserver(obj, this._from, this, this.fromDidChange);
453 
454     // if there is an operand, add an observer onto it as well
455     if (operand) { Ember.addObserver(obj, operand, this, this.fromDidChange); }
456 
457     // if the binding is a two-way binding, also set up an observer on the target
458     // object.
459     if (!oneWay) { Ember.addObserver(obj, this._to, this, this.toDidChange); }
460 
461     if (Ember.meta(obj,false).proto !== obj) { this._scheduleSync(obj, 'fwd'); }
462 
463     this._readyToSync = true;
464     return this;
465   },
466 
467   /**
468     Disconnects the binding instance. Changes will no longer be relayed. You
469     will not usually need to call this method.
470 
471     @param {Object} obj
472       The root object you passed when connecting the binding.
473 
474     @returns {Ember.Binding} this
475   */
476   disconnect: function(obj) {
477     ember_assert('Must pass a valid object to Ember.Binding.disconnect()', !!obj);
478 
479     var oneWay = this._oneWay, operand = this._operand;
480 
481     // remove an observer on the object so we're no longer notified of
482     // changes that should update bindings.
483     Ember.removeObserver(obj, this._from, this, this.fromDidChange);
484 
485     // if there is an operand, remove the observer from it as well
486     if (operand) Ember.removeObserver(obj, operand, this, this.fromDidChange);
487 
488     // if the binding is two-way, remove the observer from the target as well
489     if (!oneWay) Ember.removeObserver(obj, this._to, this, this.toDidChange);
490 
491     this._readyToSync = false; // disable scheduled syncs...
492     return this;
493   },

499   /** @private - called when the from side changes */
500   fromDidChange: function(target) {
501     this._scheduleSync(target, 'fwd');
502   },
503 
504   /** @private - called when the to side changes */
505   toDidChange: function(target) {
506     this._scheduleSync(target, 'back');

509   /** @private */
510   _scheduleSync: function(obj, dir) {
511     var guid = guidFor(obj), existingDir = this[guid];
512 
513     // if we haven't scheduled the binding yet, schedule it
514     if (!existingDir) {
515       Ember.run.schedule('sync', this, this._sync, obj);
516       this[guid] = dir;
517     }
518 
519     // If both a 'back' and 'fwd' sync have been scheduled on the same object,
520     // default to a 'fwd' sync so that it remains deterministic.
521     if (existingDir === 'back' && dir === 'fwd') {
522       this[guid] = 'fwd';
523     }
524   },
525 
526   /** @private */
527   _sync: function(obj) {
528     var log = Ember.LOG_BINDINGS;
529 
530     // don't synchronize destroyed objects or disconnected bindings
531     if (obj.isDestroyed || !this._readyToSync) { return; }
532 
533     // get the direction of the binding for the object we are
534     // synchronizing from
535     var guid = guidFor(obj), direction = this[guid];
536 
537     var fromPath = this._from, toPath = this._to;
538 
539     delete this[guid];
540 
541     // apply any operations to the object, then apply transforms
542     var fromValue = getTransformedFromValue(obj, this);
543     var toValue   = getTransformedToValue(obj, this);
544 
545     if (toValue === fromValue) { return; }
546 
547     // if we're synchronizing from the remote object...
548     if (direction === 'fwd') {
549       if (log) { Ember.Logger.log(' ', this.toString(), toValue, '->', fromValue, obj); }
550       Ember.trySetPath(Ember.isGlobalPath(toPath) ? window : obj, toPath, fromValue);
551 
552     // if we're synchronizing *to* the remote object
553     } else if (direction === 'back') {// && !this._oneWay) {
554       if (log) { Ember.Logger.log(' ', this.toString(), toValue, '<-', fromValue, obj); }
555       Ember.trySetPath(Ember.isGlobalPath(fromPath) ? window : obj, fromPath, toValue);
556     }
557   }

689 /**
690   @class
691 
692   A binding simply connects the properties of two objects so that whenever the
693   value of one property changes, the other property will be changed also. You
694   do not usually work with Binding objects directly but instead describe
695   bindings in your class definition using something like:
696 
697         valueBinding: "MyApp.someController.title"
698 
699   This will create a binding from `MyApp.someController.title` to the `value`
700   property of your object instance automatically. Now the two values will be
701   kept in sync.
702 
703   ## Customizing Your Bindings
704 
705   In addition to synchronizing values, bindings can also perform some basic
706   transforms on values. These transforms can help to make sure the data fed
707   into one object always meets the expectations of that object regardless of
708   what the other object outputs.
709 
710   To customize a binding, you can use one of the many helper methods defined
711   on Ember.Binding like so:
712 
713         valueBinding: Ember.Binding.single("MyApp.someController.title")
714 
715   This will create a binding just like the example above, except that now the
716   binding will convert the value of `MyApp.someController.title` to a single
717   object (removing any arrays) before applying it to the `value` property of
718   your object.
719 
720   You can also chain helper methods to build custom bindings like so:
721 
722         valueBinding: Ember.Binding.single("MyApp.someController.title").notEmpty("(EMPTY)")
723 
724   This will force the value of MyApp.someController.title to be a single value
725   and then check to see if the value is "empty" (null, undefined, empty array,
726   or an empty string). If it is empty, the value will be set to the string
727   "(EMPTY)".
728 
729   ## One Way Bindings
730 
731   One especially useful binding customization you can use is the `oneWay()`
732   helper. This helper tells Ember that you are only interested in
733   receiving changes on the object you are binding from. For example, if you
734   are binding to a preference and you want to be notified if the preference
735   has changed, but your object will not be changing the preference itself, you
736   could do:
737 
738         bigTitlesBinding: Ember.Binding.oneWay("MyApp.preferencesController.bigTitles")
739 
740   This way if the value of MyApp.preferencesController.bigTitles changes the
741   "bigTitles" property of your object will change also. However, if you
742   change the value of your "bigTitles" property, it will not update the
743   preferencesController.
744 
745   One way bindings are almost twice as fast to setup and twice as fast to
746   execute because the binding only has to worry about changes to one side.
747 
748   You should consider using one way bindings anytime you have an object that
749   may be created frequently and you do not intend to change a property; only
750   to monitor it for changes. (such as in the example above).
751 
752   ## Adding Custom Transforms
753 
754   In addition to using the standard helpers provided by Ember, you can
755   also defined your own custom transform functions which will be used to
756   convert the value. To do this, just define your transform function and add
757   it to the binding with the transform() helper. The following example will
758   not allow Integers less than ten. Note that it checks the value of the
759   bindings and allows all other values to pass:
760 
761         valueBinding: Ember.Binding.transform(function(value, binding) {
762           return ((Ember.typeOf(value) === 'number') && (value < 10)) ? 10 : value;
763         }).from("MyApp.someController.value")
764 
765   If you would like to instead use this transform on a number of bindings,
766   you can also optionally add your own helper method to Ember.Binding. This
767   method should simply return the value of `this.transform()`. The example
768   below adds a new helper called `notLessThan()` which will limit the value to
769   be not less than the passed minimum:
770 
771       Ember.Binding.reopen({
772         notLessThan: function(minValue) {
773           return this.transform(function(value, binding) {
774             return ((Ember.typeOf(value) === 'number') && (value < minValue)) ? minValue : value;
775           });
776         }
777       });
778 
779   You could specify this in your core.js file, for example. Then anywhere in
780   your application you can use it to define bindings like so:
781 
782         valueBinding: Ember.Binding.from("MyApp.someController.value").notLessThan(10)
783 
784   Also, remember that helpers are chained so you can use your helper along
785   with any other helpers. The example below will create a one way binding that
786   does not allow empty values or values less than 10:
787 
788         valueBinding: Ember.Binding.oneWay("MyApp.someController.value").notEmpty().notLessThan(10)
789 
790   Finally, it's also possible to specify bi-directional transforms. To do this,
791   you can pass a hash to `transform` with `to` and `from`. In the following
792   example, we are expecting a lowercase string that we want to transform to
793   uppercase.
794 
795         valueBinding: Ember.Binding.transform({
796           to:   function(value, binding) { return value.toUpperCase(); },
797           from: function(value, binding) { return value.toLowerCase(); }
798 
799   ## How to Manually Adding Binding
800 
801   All of the examples above show you how to configure a custom binding, but
802   the result of these customizations will be a binding template, not a fully
803   active binding. The binding will actually become active only when you
804   instantiate the object the binding belongs to. It is useful however, to
805   understand what actually happens when the binding is activated.
806 
807   For a binding to function it must have at least a "from" property and a "to"
808   property. The from property path points to the object/key that you want to
809   bind from while the to path points to the object/key you want to bind to.
810 
811   When you define a custom binding, you are usually describing the property
812   you want to bind from (such as "MyApp.someController.value" in the examples
813   above). When your object is created, it will automatically assign the value
814   you want to bind "to" based on the name of your binding key. In the
815   examples above, during init, Ember objects will effectively call
816   something like this on your binding:
817 
818         binding = Ember.Binding.from(this.valueBinding).to("value");
819 
820   This creates a new binding instance based on the template you provide, and
821   sets the to path to the "value" property of the new object. Now that the
822   binding is fully configured with a "from" and a "to", it simply needs to be
823   connected to become active. This is done through the connect() method:
824 
825         binding.connect(this);
826 
827   Note that when you connect a binding you pass the object you want it to be
828   connected to.  This object will be used as the root for both the from and
829   to side of the binding when inspecting relative paths.  This allows the
830   binding to be automatically inherited by subclassed objects as well.
831 
832   Now that the binding is connected, it will observe both the from and to side
833   and relay changes.
834 
835   If you ever needed to do so (you almost never will, but it is useful to
836   understand this anyway), you could manually create an active binding by
837   using the Ember.bind() helper method. (This is the same method used by
838   to setup your bindings on objects):
839 
840         Ember.bind(MyApp.anotherObject, "value", "MyApp.someController.value");
841 
842   Both of these code fragments have the same effect as doing the most friendly
843   form of binding creation like so:
844 
845         MyApp.anotherObject = Ember.Object.create({
846           valueBinding: "MyApp.someController.value",
847 
848           // OTHER CODE FOR THIS OBJECT...
849 
850         });
851 
852   Ember's built in binding creation method makes it easy to automatically
853   create bindings for you. You should always use the highest-level APIs
854   available, even if you understand how to it works underneath.
855 
856   @since Ember 0.9
857 */
858 Ember.Binding = Binding;
859 
860 /**
861   Global helper method to create a new binding.  Just pass the root object
862   along with a to and from path to create and connect the binding.  The new
863   binding object will be returned which you can further configure with
864   transforms and other conditions.
865 
866   @param {Object} obj
867     The root object of the transform.
868 
869   @param {String} to
870     The path to the 'to' side of the binding.  Must be relative to obj.
871 
872   @param {String} from
873     The path to the 'from' side of the binding.  Must be relative to obj or
874     a global path.
875 
876   @returns {Ember.Binding} binding instance
877 */
878 Ember.bind = function(obj, to, from) {
879   return new Ember.Binding(to, from).connect(obj);
880 };
881 
882 Ember.oneWay = function(obj, to, from) {
883   return new Ember.Binding(to, from).oneWay().connect(obj);
884 };
