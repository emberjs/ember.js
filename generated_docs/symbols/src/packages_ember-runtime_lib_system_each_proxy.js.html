 12 var set = Ember.set, get = Ember.get, guidFor = Ember.guidFor;
 13 
 14 var EachArray = Ember.Object.extend(Ember.Array, {
 15 
 16   init: function(content, keyName, owner) {
 17     this._super();
 18     this._keyName = keyName;
 19     this._owner   = owner;
 20     this._content = content;
 21   },
 22 
 23   objectAt: function(idx) {
 24     var item = this._content.objectAt(idx);
 25     return item && get(item, this._keyName);
 26   },
 27 
 28   length: Ember.computed(function() {
 29     var content = this._content;
 30     return content ? get(content, 'length') : 0;
 31   }).property('[]').cacheable()
 32 
 33 });
 34 
 35 var IS_OBSERVER = /^.+:(before|change)$/;
 36 
 37 /** @private */
 38 function addObserverForContentKey(content, keyName, proxy, idx, loc) {
 39   var objects = proxy._objects, guid;
 40   if (!objects) objects = proxy._objects = {};
 41 
 42   while(--loc>=idx) {
 43     var item = content.objectAt(loc);
 44     if (item) {
 45       Ember.addBeforeObserver(item, keyName, proxy, 'contentKeyWillChange');
 46       Ember.addObserver(item, keyName, proxy, 'contentKeyDidChange');
 47 
 48       // keep track of the indicies each item was found at so we can map
 49       // it back when the obj changes.
 50       guid = guidFor(item);
 51       if (!objects[guid]) objects[guid] = [];
 52       objects[guid].push(loc);
 53     }
 54   }
 55 }
 56 
 57 /** @private */
 58 function removeObserverForContentKey(content, keyName, proxy, idx, loc) {
 59   var objects = proxy._objects;
 60   if (!objects) objects = proxy._objects = {};
 61   var indicies, guid;
 62 
 63   while(--loc>=idx) {
 64     var item = content.objectAt(loc);
 65     if (item) {
 66       Ember.removeBeforeObserver(item, keyName, proxy, 'contentKeyWillChange');
 67       Ember.removeObserver(item, keyName, proxy, 'contentKeyDidChange');
 68 
 69       guid = guidFor(item);
 70       indicies = objects[guid];
 71       indicies[indicies.indexOf(loc)] = null;
 72     }
 73   }
 74 }

 76 /**
 77   @private
 78   @class
 79 
 80   This is the object instance returned when you get the @each property on an
 81   array.  It uses the unknownProperty handler to automatically create
 82   EachArray instances for property names.
 83 
 84   @extends Ember.Object
 85 */
 86 Ember.EachProxy = Ember.Object.extend({
 87 
 88   init: function(content) {
 89     this._super();
 90     this._content = content;
 91     content.addArrayObserver(this);
 92 
 93     // in case someone is already observing some keys make sure they are
 94     // added
 95     Ember.watchedEvents(this).forEach(function(eventName) {
 96       this.didAddListener(eventName);
 97     }, this);
 98   },
 99 
100   /**
101     You can directly access mapped properties by simply requesting them.
102     The unknownProperty handler will generate an EachArray of each item.
103   */
104 105   unknownProperty: function(keyName, value) {
106     var ret;
107     ret = new EachArray(this._content, keyName, this);
108     new Ember.Descriptor().setup(this, keyName, ret);
109     this.beginObservingContentKey(keyName);
110     return ret;
111   },
112 
113   // ..........................................................
114   // ARRAY CHANGES
115   // Invokes whenever the content array itself changes.
116 
117   arrayWillChange: function(content, idx, removedCnt, addedCnt) {
118     var keys = this._keys, key, array, lim;
119 
120     lim = removedCnt>0 ? idx+removedCnt : -1;
121     Ember.beginPropertyChanges(this);
122 
123     for(key in keys) {
124       if (!keys.hasOwnProperty(key)) { continue; }
125 
126       if (lim>0) removeObserverForContentKey(content, key, this, idx, lim);
127 
128       Ember.propertyWillChange(this, key);
129     }
130 
131     Ember.propertyWillChange(this._content, '@each');
132     Ember.endPropertyChanges(this);
133   },
134 
135   arrayDidChange: function(content, idx, removedCnt, addedCnt) {
136     var keys = this._keys, key, array, lim;
137 
138     lim = addedCnt>0 ? idx+addedCnt : -1;
139     Ember.beginPropertyChanges(this);
140 
141     for(key in keys) {
142       if (!keys.hasOwnProperty(key)) { continue; }
143 
144       if (lim>0) addObserverForContentKey(content, key, this, idx, lim);
145 
146       Ember.propertyDidChange(this, key);
147     }
148 
149     Ember.propertyDidChange(this._content, '@each');
150     Ember.endPropertyChanges(this);
151   },
152 
153   // ..........................................................
154   // LISTEN FOR NEW OBSERVERS AND OTHER EVENT LISTENERS
155   // Start monitoring keys based on who is listening...
156 
157   didAddListener: function(eventName) {
158     if (IS_OBSERVER.test(eventName)) {
159       this.beginObservingContentKey(eventName.slice(0, -7));
160     }
161   },
162 
163   didRemoveListener: function(eventName) {
164     if (IS_OBSERVER.test(eventName)) {
165       this.stopObservingContentKey(eventName.slice(0, -7));
166     }
167   },
168 
169   // ..........................................................
170   // CONTENT KEY OBSERVING
171   // Actual watch keys on the source content.
172 
173   beginObservingContentKey: function(keyName) {
174     var keys = this._keys;
175     if (!keys) keys = this._keys = {};
176     if (!keys[keyName]) {
177       keys[keyName] = 1;
178       var content = this._content,
179           len = get(content, 'length');
180       addObserverForContentKey(content, keyName, this, 0, len);
181     } else {
182       keys[keyName]++;
183     }
184   },
185 
186   stopObservingContentKey: function(keyName) {
187     var keys = this._keys;
188     if (keys && (keys[keyName]>0) && (--keys[keyName]<=0)) {
189       var content = this._content,
190           len     = get(content, 'length');
191       removeObserverForContentKey(content, keyName, this, 0, len);
192     }
193   },
194 
195   contentKeyWillChange: function(obj, keyName) {
196     Ember.propertyWillChange(this, keyName);
197   },
198 
199   contentKeyDidChange: function(obj, keyName) {
200     Ember.propertyDidChange(this, keyName);
201   }
202 
203 });
