287       // is the view by default. A hash of data is also passed that provides

300   invokeForState: function(name) {
301     var parent = this, states = parent.states;
302     var stateName = get(this, 'state'), state;
303 
304     while (states) {
305       state = states[stateName];
306 
307       while (state) {
308         var fn = state[name];
309 
310         if (fn) {
311           var args = a_slice.call(arguments, 1);
312           args.unshift(this);
313 
314           return fn.apply(this, args);
315         }
316 
317         state = state.parentState;
318       }
319 
320       states = states.parent;
321     }

358   /**
359     @private
360 
361     Iterates over the view's `classNameBindings` array, inserts the value
362     of the specified property into the `classNames` array, then creates an
363     observer to update the view's element if the bound property ever changes
364     in the future.
365   */
366   _applyClassNameBindings: function() {
367     var classBindings = get(this, 'classNameBindings'),
368         classNames = get(this, 'classNames'),
369         elem, newClass, dasherizedClass;
370 
371     if (!classBindings) { return; }
372 
373     // Loop through all of the configured bindings. These will be either
374     // property names ('isUrgent') or property paths relative to the view
375     // ('content.isUrgent')
376     classBindings.forEach(function(binding) {
377 
378       // Variable in which the old class value is saved. The observer function
379       // closes over this variable, so it knows which string to remove when
380       // the property changes.
381       var oldClass, property;
382 
383       // Set up an observer on the context. If the property changes, toggle the
384       // class name.
385       var observer = function() {
386         // Get the current value of the property
387         newClass = this._classStringForProperty(binding);
388         elem = this.$();
389 
390         // If we had previously added a class to the element, remove it.
391         if (oldClass) {
392           elem.removeClass(oldClass);
393           // Also remove from classNames so that if the view gets rerendered,
394           // the class doesn't get added back to the DOM.
395           classNames.removeObject(oldClass);
396         }
397 
398         // If necessary, add a new class. Make sure we keep track of it so
399         // it can be removed in the future.
400         if (newClass) {
401           elem.addClass(newClass);
402           oldClass = newClass;
403         } else {
404           oldClass = null;
405         }
406       };
407 
408       // Get the class name for the property at its current value
409       dasherizedClass = this._classStringForProperty(binding);
410 
411       if (dasherizedClass) {
412         // Ensure that it gets into the classNames array
413         // so it is displayed when we render.
414         classNames.push(dasherizedClass);
415 
416         // Save a reference to the class name so we can remove it
417         // if the observer fires. Remember that this variable has
418         // been closed over by the observer.
419         oldClass = dasherizedClass;
420       }
421 
422       // Extract just the property name from bindings like 'foo:bar'
423       property = binding.split(':')[0];
424       addObserver(this, property, observer);
425     }, this);
426   },

428   /**
429     Iterates through the view's attribute bindings, sets up observers for each,
430     then applies the current value of the attributes to the passed render buffer.
431 
432     @param {Ember.RenderBuffer} buffer
433   */
434   _applyAttributeBindings: function(buffer) {
435     var attributeBindings = get(this, 'attributeBindings'),
436         attributeValue, elem, type;
437 
438     if (!attributeBindings) { return; }
439 
440     attributeBindings.forEach(function(attributeName) {
441       // Create an observer to add/remove/change the attribute if the
442       // JavaScript property changes.
443       var observer = function() {
444         elem = this.$();
445         attributeValue = get(this, attributeName);
446 
447         Ember.View.applyAttributeBindings(elem, attributeName, attributeValue)
448       };
449 
450       addObserver(this, attributeName, observer);
451 
452       // Determine the current value and add it to the render buffer
453       // if necessary.
454       attributeValue = get(this, attributeName);
455       Ember.View.applyAttributeBindings(buffer, attributeName, attributeValue);
456     }, this);
457   },

459   /**
460     @private
461 
462     Given a property name, returns a dasherized version of that
463     property name if the property evaluates to a non-falsy value.
464 
465     For example, if the view has property `isUrgent` that evaluates to true,
466     passing `isUrgent` to this method will return `"is-urgent"`.
467   */
468   _classStringForProperty: function(property) {
469     var split = property.split(':'),
470         property = split[0],
471         className = split[1];
472 
473     // TODO: Remove this `false` when the `getPath` globals support is removed
474     var val = Ember.getPath(this, property, false);
475     if (val === undefined && Ember.isGlobalPath(property)) {
476       val = Ember.getPath(window, property);
477     }
478 
479     // If value is a Boolean and true, return the dasherized property
480     // name.
481     if (val === YES) {
482       if (className) { return className; }
483 
484       // Normalize property path to be suitable for use
485       // as a class name. For exaple, content.foo.barBaz
486       // becomes bar-baz.
487       var parts = property.split('.');
488       return Ember.String.dasherize(parts[parts.length-1]);
489 
490     // If the value is not NO, undefined, or null, return the current
491     // value of the property.
492     } else if (val !== NO && val !== undefined && val !== null) {
493       return val;
494 
495     // Nothing to display. Return null so that the old class is removed
496     // but no new class is added.
497     } else {
498       return null;
499     }
500   },
