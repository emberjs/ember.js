  9 /**
 10   @private
 11 
 12   Defines a namespace that will contain an executable application.  This is
 13   very similar to a normal namespace except that it is expected to include at
 14   least a 'ready' function which can be run to initialize the application.
 15   
 16   Currently Ember.Application is very similar to Ember.Namespace.  However, this
 17   class may be augmented by additional frameworks so it is important to use
 18   this instance when building new applications.
 19   
 20   # Example Usage
 21   
 22       MyApp = Ember.Application.create({
 23         VERSION: '1.0.0',
 24         store: Ember.Store.create().from(Ember.fixtures)
 25       });
 26       
 27       MyApp.ready = function() { 
 28         //..init code goes here...
 29       }
 30       
 31 */
 32 Ember.Application = Ember.Namespace.extend();
