1 require('ember-handlebars/ext');
  2 
  3 var EmberHandlebars = Ember.Handlebars, getPath = Ember.Handlebars.getPath;
  4 
  5 var ActionHelper = EmberHandlebars.ActionHelper = {};
  6 
  7 ActionHelper.registerAction = function(actionName, eventName, target, view) {
  8   var actionId = (++jQuery.uuid).toString(),
  9       existingHandler = view[eventName];
 10 
 11   function handler(event) {
 12     if (Ember.$(event.target).closest('[data-ember-action]').attr('data-ember-action') === actionId) {
 13       if ('function' === typeof target.send) {
 14         return target.send(actionName);
 15       } else {
 16         return target[actionName](event);
 17       }
 18     }
 19   }
 20 
 21   if (existingHandler) {
 22     view[eventName] = function(event) {
 23       var ret = handler.call(view, event);
 24       return ret !== false ? existingHandler.call(view, event) : ret;
 25     };
 26   } else {
 27     view[eventName] = handler;
 28   }
 29 
 30   view.reopen({
 31     rerender: function() {
 32       if (existingHandler) {
 33         view[eventName] = existingHandler;
 34       } else {
 35         view[eventName] = null;
 36       }
 37       return this._super();
 38     }
 39   });
 40 
 41   return actionId;
 42 };
 43 
 44 EmberHandlebars.registerHelper('action', function(actionName, options) {
 45   var hash = options.hash || {},
 46       eventName = options.hash.on || "click",
 47       view = options.data.view,
 48       target;
 49 
 50   if (view.isVirtual) { view = view.get('parentView'); }
 51   target = options.hash.target ? getPath(this, options.hash.target) : view;
 52 
 53   var actionId = ActionHelper.registerAction(actionName, eventName, target, view);
 54   return new EmberHandlebars.SafeString('data-ember-action="' + actionId + '"');
 55 });
