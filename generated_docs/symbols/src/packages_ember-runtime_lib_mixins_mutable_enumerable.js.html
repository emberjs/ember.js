  8 require('ember-runtime/mixins/enumerable');
  9 
 10 /**
 11   @class
 12 
 13   This mixin defines the API for modifying generic enumerables.  These methods
 14   can be applied to an object regardless of whether it is ordered or 
 15   unordered.
 16   
 17   Note that an Enumerable can change even if it does not implement this mixin.
 18   For example, a MappedEnumerable cannot be directly modified but if its 
 19   underlying enumerable changes, it will change also.
 20 
 21   ## Adding Objects
 22   
 23   To add an object to an enumerable, use the addObject() method.  This 
 24   method will only add the object to the enumerable if the object is not 
 25   already present and the object if of a type supported by the enumerable.
 26   
 27       javascript:
 28       set.addObject(contact);
 29       
 30   ## Removing Objects
 31   
 32   To remove an object form an enumerable, use the removeObject() method.  This
 33   will only remove the object if it is already in the enumerable, otherwise
 34   this method has no effect.
 35   
 36       javascript:
 37       set.removeObject(contact);
 38       
 39   ## Implementing In Your Own Code
 40   
 41   If you are implementing an object and want to support this API, just include
 42   this mixin in your class and implement the required methods.  In your unit
 43   tests, be sure to apply the Ember.MutableEnumerableTests to your object.
 44   
 45   @extends Ember.Mixin
 46   @extends Ember.Enumerable
 47 */
 48 Ember.MutableEnumerable = Ember.Mixin.create(Ember.Enumerable, 
 49   /** @scope Ember.MutableEnumerable.prototype */ {
 50   
 51   /**
 52     __Required.__ You must implement this method to apply this mixin.
 53     
 54     Attempts to add the passed object to the receiver if the object is not 
 55     already present in the collection. If the object is present, this method
 56     has no effect. 
 57     
 58     If the passed object is of a type not supported by the receiver (for 
 59     example if you pass an object to an IndexSet) then this method should 
 60     raise an exception.
 61     
 62     @param {Object} object
 63       The object to add to the enumerable.
 64       
 65     @returns {Object} the passed object
 66   */
 67   addObject: Ember.required(Function),
 68 
 69   /**
 70     Adds each object in the passed enumerable to the receiver.
 71 
 72     @param {Ember.Enumerable} objects the objects to remove
 73     @returns {Object} receiver
 74   */
 75   addObjects: function(objects) {
 76     Ember.beginPropertyChanges(this);
 77     objects.forEach(function(obj) { this.addObject(obj); }, this);
 78     Ember.endPropertyChanges(this);
 79     return this;
 80   },
 81 
 82   /**
 83     __Required.__ You must implement this method to apply this mixin.
 84     
 85     Attempts to remove the passed object from the receiver collection if the
 86     object is in present in the collection.  If the object is not present,
 87     this method has no effect.
 88     
 89     If the passed object is of a type not supported by the receiver (for 
 90     example if you pass an object to an IndexSet) then this method should 
 91     raise an exception.
 92     
 93     @param {Object} object
 94       The object to remove from the enumerable.
 95       
 96     @returns {Object} the passed object
 97   */
 98   removeObject: Ember.required(Function),
 99   
100   
101   /**
102     Removes each objects in the passed enumerable from the receiver.
103 
104     @param {Ember.Enumerable} objects the objects to remove
105     @returns {Object} receiver
106   */
107   removeObjects: function(objects) {
108     Ember.beginPropertyChanges(this);
109     objects.forEach(function(obj) { this.removeObject(obj); }, this);
110     Ember.endPropertyChanges(this);
111     return this;
112   }
113     
114 });
