  3 Ember.Select = Ember.View.extend({
  4   tagName: 'select',
  5   template: Ember.Handlebars.compile(
  6     '{{#if prompt}}<option>{{prompt}}</option>{{/if}}' +
  7     '{{#each content}}{{view Ember.SelectOption contentBinding="this"}}{{/each}}'
  8   ),
  9 
 10   content: null,
 11   selection: null,
 12   prompt: null,
 13 
 14   optionLabelPath: 'content',
 15   optionValuePath: 'content',
 16 
 17 
 18   didInsertElement: function() {
 19     var selection = get(this, 'selection');
 20 
 21     if (selection) { this.selectionDidChange(); }
 22 
 23     this.change();
 24   },
 25 
 26   change: function() {
 27     var selectedIndex = this.$()[0].selectedIndex,
 28         content = get(this, 'content'),
 29         prompt = get(this, 'prompt');
 30 
 31     if (!content) { return; }
 32     if (prompt && selectedIndex === 0) { set(this, 'selection', null); return; }
 33 
 34     if (prompt) { selectedIndex -= 1; }
 35     set(this, 'selection', content.objectAt(selectedIndex));
 36   },
 37 
 38   selectionDidChange: Ember.observer(function() {
 39     var el = this.$()[0],
 40         content = get(this, 'content'),
 41         selection = get(this, 'selection'),
 42         selectionIndex = content.indexOf(selection),
 43         prompt = get(this, 'prompt');
 44 
 45     if (prompt) { selectionIndex += 1; }
 46     if (el) { el.selectedIndex = selectionIndex; }
 47   }, 'selection')
 48 });

 50 Ember.SelectOption = Ember.View.extend({
 51   tagName: 'option',
 52   template: Ember.Handlebars.compile("{{label}}"),
 53   attributeBindings: ['value'],
 54 
 55   init: function() {
 56     this.labelPathDidChange();
 57     this.valuePathDidChange();
 58 
 59     this._super();
 60   },
 61 
 62   labelPathDidChange: Ember.observer(function() {
 63     var labelPath = getPath(this, 'parentView.optionLabelPath');
 64 
 65     if (!labelPath) { return; }
 66 
 67     Ember.defineProperty(this, 'label', Ember.computed(function() {
 68       return getPath(this, labelPath);
 69     }).property(labelPath).cacheable());
 70   }, 'parentView.optionLabelPath'),
 71 
 72   valuePathDidChange: Ember.observer(function() {
 73     var valuePath = getPath(this, 'parentView.optionValuePath');
 74 
 75     if (!valuePath) { return; }
 76 
 77     Ember.defineProperty(this, 'value', Ember.computed(function() {
 78       return getPath(this, valuePath);
 79     }).property(valuePath).cacheable());
 80   }, 'parentView.optionValuePath')
 81 });
