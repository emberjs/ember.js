  7 require('ember-runtime/system/array_proxy');
  8 
  9 /**
 10   @class
 11 
 12   Ember.ArrayController provides a way for you to publish an array of objects for
 13   Ember.CollectionView or other controllers to work with.  To work with an
 14   ArrayController, set the content property to the array you want the controller
 15   to manage.  Then work directly with the controller object as if it were the
 16   array itself.
 17 
 18   For example, imagine you wanted to display a list of items fetched via an XHR
 19   request. Create an Ember.ArrayController and set its `content` property:
 20 
 21       MyApp.listController = Ember.ArrayController.create();
 22 
 23       $.get('people.json', function(data) {
 24         MyApp.listController.set('content', data);
 25       });
 26 
 27   Then, create a view that binds to your new controller:
 28 
 29     {{#collection contentBinding="MyApp.listController"}}
 30       {{content.firstName}} {{content.lastName}}
 31     {{/collection}}
 32 
 33   The advantage of using an array controller is that you only have to set up
 34   your view bindings once; to change what's displayed, simply swap out the
 35   `content` property on the controller.
 36 
 37   @extends Ember.ArrayProxy
 38 */
 39 
 40 Ember.ArrayController = Ember.ArrayProxy.extend();
