  8 require('ember-handlebars/ext');
  9 
 10 var getPath = Ember.Handlebars.getPath;
 11 
 12 /**
 13   `unbound` allows you to output a property without binding. *Important:* The 
 14   output will not be updated if the property changes. Use with caution.
 15 
 16       <div>{{unbound somePropertyThatDoesntChange}}</div>
 17 
 18   @name Handlebars.helpers.unbound
 19   @param {String} property
 20   @returns {String} HTML string
 21 */
 22 Ember.Handlebars.registerHelper('unbound', function(property, fn) {
 23   var context = (fn.contexts && fn.contexts[0]) || this;
 24   return getPath(context, property);
 25 });
