  8 require('ember-views/views/states/default');
  9 
 10 var destroyedError = "You can't call %@ on a destroyed view", fmt = Ember.String.fmt;
 11 
 12 Ember.View.states.destroyed = {
 13   parentState: Ember.View.states._default,
 14 
 15   appendChild: function() {
 16     throw fmt(destroyedError, ['appendChild']);
 17   },
 18   rerender: function() {
 19     throw fmt(destroyedError, ['rerender']);
 20   },
 21   destroyElement: function() {
 22     throw fmt(destroyedError, ['destroyElement']);
 23   },
 24 
 25   setElement: function() {
 26     throw fmt(destroyedError, ["set('element', ...)"]);
 27   },
 28 
 29   // Since element insertion is scheduled, don't do anything if
 30   // the view has been destroyed between scheduling and execution
 31   insertElement: Ember.K
 32 };
